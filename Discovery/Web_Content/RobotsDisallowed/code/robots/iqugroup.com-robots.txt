	(function () {
		// Inline this so that it's immediate available for load view
		// Class bits taken from classie: https://github.com/desandro/classie

		var _body = document.getElementById('body');
		var _world_intro = document.getElementsByClassName('intro')[0];
		var _slider_container = document.getElementsByClassName('sliders-container')[0];
		var _slider_container_sliders = document.getElementsByClassName('slider');
		var _spaceship_thruster_right = document.getElementsByClassName('thruster-right')[0];
		var _preloader_bg = document.getElementsByClassName('preloader-bg')[0];
		_scrolled = false;
		classListSupport=false;

		function classReg(className) {
			return new RegExp('(^|\\s+)' + className + '(\\s+|$)');
		}

		function _hasClass(element, className) {
			return classReg(className).test(element.className);
		}

		function _addClass(element, className) {
			if (classListSupport) {
				element.classList.add(className);
			} else if (!_hasClass(element, className)) {
				element.className = element.className + ' ' + className;
			}
		}

		function _removeClass(element, className) {
			if (classListSupport) {
				element.classList.remove(className);
			} else {
				element.className = element.className.replace(classReg(className), ' ');
			}
		}

		function update_screen_size() {
			window_size_h = window.innerHeight;
			window_size_w = window.innerWidth;

			if( window_size_w < 1024 )
			{
				window_size_w = 1024;
			}
		}

		function update_positions() {
			st = window.pageYOffset || document.documentElement.scrollTop;
			sb = Math.round(st + window_size_h);
			_body.setAttribute("data-st", st);
			_body.setAttribute("data-sb", sb);
			if(st>0){
				_scrolled = true;
			}
		}

		function update_slider_size() {

			_preloader_bg.style.width = window_size_w+'px';
			_preloader_bg.style.height = window_size_h+'px';

			_world_intro.style.width = window_size_w+'px';
			_world_intro.style.height = (window_size_h - 120 )+'px';

			_slider_container.style.width = window_size_w+'px';
			_slider_container.style.height = (window_size_h - 120 )+'px';

			for( var i=0 ; i<_slider_container_sliders.length ; i++ )
			{
				_slider_container_sliders[i].style.width = window_size_w+'px';
				_slider_container_sliders[i].style.height = (window_size_h - 120 )+'px';
			}
		}

		function update_sliders_positions() {
			for( var i=0 ; i<_slider_container_sliders.length ; i++ )
			{
				_slider_container_sliders[i].style.opacity = 0;
			}
		}

		function init() {
			update_screen_size();
			update_positions();
			update_slider_size();
		}

		function resize() {
			init();
			refresh_about_us();
			init_content_sliders();

			init_logo_grid();
			init_media_grid();
			init_game_grid();
		}

		function scrolling() {
			update_positions();
			_scrolled = true;
		}

		if(window.isSupported)
		{
			_addClass(_body,'desktop');
		}
		else
		{
			_addClass(_body,'mobile');
		}

		
		window.addEventListener('resize', resize, false);
		window.addEventListener('orientationchange', resize, false);
		window.addEventListener('scroll', scrolling, false);
		

		init();

		update_sliders_positions();
		
	})();
