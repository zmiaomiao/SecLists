function rewriteLinks(id){
	var target = document.getElementById(id);
	if(target){
		var arr = target.getElementsByTagName("li");
		for(var i = 0; i < arr.length; i++){
			var spanlist = arr[i].getElementsByTagName("span");
			if(spanlist[0] != undefined && spanlist.length==1){
				var link = spanlist[0].getAttribute('title');
				var linktext = spanlist[0].getAttribute('alt');			
				arr[i].innerHTML = "<a href='" + link + "'>" + linktext + "</a>";
			}
		}
	}
}
