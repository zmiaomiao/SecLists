function vatMultiply(string, multi)
{
	var regex = /^([^.0-9]+\s?)([0-9\.]+)(\s)?(\/(.+))?$/;
	var matches = string.match(regex);

	// Currency symbol in front
	if ( matches )
	{
		var returnStr = matches[1] + (parseFloat(matches[2])*multi).toFixed(2);
		if (typeof(matches[3]) != 'undefined')
			returnStr += matches[3];
		if (typeof(matches[4]) != 'undefined')
			returnStr += matches[4];

		return returnStr;
	}

	regex = /^([0-9\.]+)(\s)?([^.]+)?(\/(.+))?$/;
	matches = string.match(regex);

	// Currency symbol in da bak!
	if ( matches )
	{
		var returnStr = (parseFloat(matches[1])*multi).toFixed(2);
		if (typeof(matches[2]) != 'undefined')
			returnStr += matches[2];
		if (typeof(matches[3]) != 'undefined')
			returnStr += matches[3];
		if (typeof(matches[4]) != 'undefined')
			returnStr += matches[4];
		return returnStr;
	}

	return false;
}

function toggleVatIncluded()
{
	if (vatSwitchInProgress)
		return;

	vatSwitchInProgress = true;

	var enabled = !inPageVatStatus;

	document.cookie = "includingVat=" + (enabled ? 1 : 0) + ";expires=0";

	var vatPercent = 21;
	var multi = enabled ? ((vatPercent+100)/100) : 1/((vatPercent+100)/100);
	$('.vatPriceElement').each(function()
	{
		$(this).text(vatMultiply($(this).text(),multi));
	});

	if (enabled)
	{
		$('.vatNoticeText').each(function()
		{
			$(this).text($(this).text().replace('excl','incl').replace('Excl','Incl'));
		});
		$('#vatSwitcher').children('strong').text('Prices include VAT');
		$('#vatSwitcher a strong').text('Show prices without VAT');
	}
	else
	{
		$('.vatNoticeText').each(function()
		{
			$(this).text($(this).text().replace('incl','excl').replace('Incl','Excl'));
		});
		$('#vatSwitcher').children('strong').text('Prices exclude VAT');
		$('#vatSwitcher a strong').text('Show prices with VAT');
	}

	inPageVatStatus = !inPageVatStatus;
	vatSwitchInProgress = false;
	return;
}

function fixedHeader()
{
	if (width <= 980 && (orientation == 0 || orientation == 180))
	{
		if (($(document).scrollTop() >= 1))
		{
			if (headerFixedClass == false)
			{
				$('#header').addClass('fixed');
				headerFixedClass = true;
			}
		}
		else
		{
			if (headerFixedClass == true)
			{
				$('#header').removeClass('fixed');
				headerFixedClass = false;
			}
		}
	}

	if ((orientation == -90 || orientation == 90) && headerFixedClass == true)
	{
		$('#header').removeClass('fixed');
		headerFixedClass = false;
	}
}
