        /*Начало: скрипт слайдера*/
        $element_num = 0;
        $max_slide = 0;
        $element_width = 0;
        $slider_interval = 4000;
        
        function slider_scrool_element(){
            if ( !$('.one_slide[slide_num="'+$element_num+'"]').length )
                if ($element_num < 0) { $element_num = $max_slide; } else { $element_num = 0; }
            
            $to_slide_x = $element_num * (-$element_width);
            $('.slides_wrap').stop().animate({ marginLeft: $to_slide_x }, 700);
            $(".slider_controls_btns .slider_one_controls_btn").removeClass("slider_one_controls_btn_active");
            $(".slider_controls_btns .slider_one_controls_btn[el_num='"+$element_num+"']").addClass("slider_one_controls_btn_active");
            $(".slide_text_block").html( $(".slider_imgs_full_text div[el_num='"+$element_num+"']").html() );
            }
        
        $(".one_slide").each( function(){
            if ( $max_slide < parseInt($(this).attr("slide_num")) ) $max_slide = parseInt($(this).attr("slide_num"));
            $element_width = parseInt( $(this).css("width") );
        } );
        
        $(".slider_left_btn, .slider_right_btn").click(function(){
            if ( $(this).hasClass("slider_right_btn") ) {$element_num = $element_num + 1;} else $element_num = $element_num - 1;
            
            slider_scrool_element();
        });
        
        $(".slider_controls_btns .slider_one_controls_btn").click(function(){ $element_num = parseInt( $(this).attr("el_num") ); slider_scrool_element(); });
        
        $(".main_slider_wrap").everyTime($slider_interval, function(i) {$element_num = $element_num + 1; slider_scrool_element();});
            
        $(".main_slider_wrap").hover(function(){ $(".main_slider_wrap").stopTime(); }, function(){ $(".main_slider_wrap").everyTime($slider_interval, function(i) {$element_num = $element_num + 1; slider_scrool_element();}); })
