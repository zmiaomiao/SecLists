  var searchNav = {
    max_items_to_display: 5,
    htmlSearchSkeleton:
      '<div><li>\
        <a class="search-result-item-link" href="">\
          <div class="product-row">\
            <div class="product-image">\
              <img src="">\
            </div>\
            <div class="product-title">Foxy In Black-Scarlet Red</div>\
            <div class="search-result-product-price">$58.50</div>\
          </div>\
        </a>\
      </li></div>',
    selectors: {
      questions: '#questions',
      spinning_bar: '#search-nav .spinning-bar'
    },
    className: {
      greenHighlight: 'green-highlight'
    },
    attachEvents: function() {
      var selectors = searchNav.selectors;
      var xhr = null;

      var removeAllHighlight = function(resultDropDown) {
        resultDropDown.removeClass(searchNav.className.greenHighlight);
      };

      var highlightContent = function(resultDropDown, loc){
        removeAllHighlight(resultDropDown);
        var container = $(resultDropDown[loc]);
        container.addClass(searchNav.className.greenHighlight);
      };

      var navigateSearchResult = function() {
        var resultDropDown = $(".has-result ul li");
        var max_item_index = resultDropDown.length - 1;
        var loc = -1;
        $(selectors.questions).off('keydown').on('keydown', function (e) {
          switch(e.which) {
            case 13: //return
              var url = $(resultDropDown[loc]).find('a').attr("href");
              if (url) {
                window.location.href = url;
                document.URL = url;
              } else {
                $("#search-form").submit();
              }
              break;

            case 38: // up
              if (-1 >= --loc) {
                loc = max_item_index;
              }
              highlightContent(resultDropDown, loc);
              break;

            case 40: // down
              if (max_item_index < ++loc) {
                loc = 0;
              }
              highlightContent(resultDropDown, loc);
              break;
            default: return; // exit this handler for other keys
          }

          e.preventDefault(); // prevent the default action (scroll / move caret)
        });
      };

      var updateSearchWithResult = function(responseText, loadUrl) {
        var domObj = $(responseText);
        var results = domObj.find('#col-main .results');
        var liObjs = results.find(".product-grid-collection-page li");
        var lstObjHTMLPrepared = $(searchNav.htmlSearchSkeleton);
        var searchResultObj = $('#search-result');
        var $searchIcon = $('.search-icon');
        var ulObj = searchResultObj.find('ul');
        ulObj.empty();

        var productVariantIds = [];

        liObjs.each(function (index, value) {
          var valObj = $(this);
          var productLinkObj = valObj.find('.product-link');
          var imageLink = productLinkObj.find('.collection-photo').attr('src');
          var title = valObj.find('.product-name').html();

          var href = productLinkObj.attr('href');
          var price = valObj.find('.collection-product-sale-price').html();
          lstObjHTMLPrepared.find('.product-image img').attr('src', imageLink);
          lstObjHTMLPrepared.find('.product-title').html(title);
          lstObjHTMLPrepared.find('.search-result-product-price').html(price);
          lstObjHTMLPrepared.find('a').attr('href', href);
          searchResultObj.find('.link a').attr('href', loadUrl);
          var html = lstObjHTMLPrepared.html();
          ulObj.append(html);

          productVariantIds.push(valObj.find('div').attr("id").split('-')[1]);

          if (index == searchNav.max_items_to_display) {
            return false;
          }
        });

        if (liObjs.size() == 0) {
          searchResultObj.removeClass("hidden");
          searchResultObj.find(".has-result").addClass("hidden");
          searchResultObj.find(".no-result").removeClass("hidden");
          $searchIcon.addClass('no-border');
        } else {
          searchResultObj.removeClass("hidden");
          searchResultObj.find(".has-result").removeClass("hidden");
          searchResultObj.find(".no-result").addClass("hidden");
          $searchIcon.addClass('no-border');

          window.criteo_q = window.criteo_q || [];
          window.criteo_q.push(
                { event: "setAccount", account: 16081 },
            	{ event: "setSiteType", type: window.siteType },
                { event: "viewList", item: productVariantIds });
        }
      };

      var keyUpOnSeachNav = function(){
        var previous_query = "";
        $(selectors.questions).keyup(function () {
          var query = $(this).val().trim();
          if (query !== previous_query) {
            previous_query = query;
            if (query !== "") {
              $(selectors.spinning_bar).show();
              var loadUrl = "/search?type=product&q=" + query;
              if (xhr) {
                xhr.abort();
              }
              xhr = $.get(
                loadUrl,
                function (responseText) {
                  updateSearchWithResult(responseText, loadUrl);
                  navigateSearchResult();
                  $(selectors.spinning_bar).hide();
                },
                "html"
              );
            }
          }
        });
      };

      keyUpOnSeachNav();

      // click outside the search result
      $(document).mouseup(function(e) {
        var searchResultObj = $('#search-result');
        var searchNavObj = $('#search-nav');
        if (searchResultObj.is(':visible') // if the search result is visible
          && !searchResultObj.is(e.target) // if the target of the click isn't the search result
          && searchResultObj.has(e.target).length === 0 // nor a descendent of the search result
          && !searchNavObj.is(e.target) // and the target of the click isn't the search nav
          && searchNavObj.has(e.target).length === 0 // nor a descendent of the search nav
        ) {
          searchResultObj.addClass("hidden");
          $('.search-icon').removeClass('no-border');
        }
      });

      $('#questions').on('focusin', function(){
        $('#search-nav').addClass('focused');
      });

      $('#questions').on('focusout', function(){
        $('#search-nav').removeClass('focused');
      });

      $('#search-nav .search-icon img').on('click', function() {
        $('form#search-form').submit();
      });

      $('form#search-form').on('submit', function(event) {
        if (!$('#questions').val()) {
          event.preventDefault();
        }
      });
    }
  };

  $(document).ready(function() {
    searchNav.attachEvents();
  });

var DateUtil = {
   calcTime: function(offset) {
    // create Date object for current location
    d = new Date();
    // convert to msec
    // add local time zone offset 
    // get UTC time in msec
    utc = d.getTime() + (d.getTimezoneOffset() * 60000);    
    nd = new Date(utc + (3600000*offset));
    return nd;
  },
  convertToDate: function(time) {
    var hours = time % 100;
    var time = Math.floor(time/100);
    var days = time % 100;
    time = Math.floor(time/100);
    var month = time % 100 - 1;
    time = Math.floor(time/100);
    var year = time;
    var d = new Date(year, month, days, hours);
    return d;
  },
  getTimeBreakDown: function(startDate, endDate) {
    total_seconds = Math.floor(endDate - startDate)/1000;
    minutes = Math.floor(total_seconds/60);
    hours = Math.floor(minutes/60);
    days = Math.floor(hours/24);

    hours = hours - (days * 24);
    minutes = minutes - (days * 24 * 60) - (hours * 60);
    seconds = total_seconds - (days * 24 * 60 * 60) - (hours * 60 * 60) - (minutes * 60);
    return {
      'days': days,
      'hours': hours,
      'minutes': minutes,
      'seconds': seconds,
      'totalseconds': total_seconds
    };
  }
}

/**

function searchPlaceholder() {
  if (!Modernizr.input.placeholder) {
    $('#top-search-input').focus(function () {
      var input = $(this);
      if (input.val() == input.attr('placeholder')) {
        input.val('');
        input.removeClass('placeholder');
      }
    }).blur(function () {
      var input = $(this);
      if (input.val() == '' || input.val() == input.attr('placeholder')) {
        input.addClass('placeholder');
        input.val(input.attr('placeholder'));
      }
    }).blur();
    $('[placeholder]').parents('form').submit(function () {
      $(this).find('[placeholder]').each(function () {
        var input = $(this);
        if (input.val() == input.attr('placeholder')) {
          input.val('');
        }
      })
    });
  }
}

    } else if (days > 1) {
      message += days + " days";
    } else if (hours == 1) {
      message += hours + " hour";
    } else if (hours > 1) {
      message += hours + " hours";
    } else {
      message += "0 hour";
    }
    $(this).find('.sale-date-banner').html(message);

    $(this).mouseenter(function() {
      var that = $(this);
      if (hours > 0 || days > 0) {
        that.find('.sale-count-down-image').removeClass('hidden');
      }
    });
    $(this).mouseleave(function() {
      var that = $(this);
      that.find('.sale-count-down-image').addClass('hidden');
    });
  });

});

</script>
