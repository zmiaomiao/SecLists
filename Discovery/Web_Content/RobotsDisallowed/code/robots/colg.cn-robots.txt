    //$('#navigation');
    $('#navigation').addClass('index_nav');
    $('#footer').css({'bottom': 0 - 50}).addClass('simple');

    function resizeFrame() {
        $('.page').width($(window).width());
        $('.page').height($(window).height());
    }

    function getSharpArt(url) {
        var arg = url.split('#');
        return arg;
    }

    var page = {'index': 0, 'dnf': 1, 'tos': 2, 'sd': 3, 'ms2': 4};
    function rollSubPage(id, subid) {
        $('#footer').css({'bottom': 0 - 50}).addClass('simple');
        if (typeof (id) == 'undefined') {
            id = 'index';
        }

        if (typeof (subid) == 'undefined') {
            if (id == 'index') {
                subid = 'cover';
            } else {
                subid = 'index';
            }
        }

        if (id == 'index') {
            $('#index-area').animate({'left' : 0}, 700);
            $('.page.index').find('iframe').attr('src', '/index/main#' + subid);
            if (subid == 'cover') {
                if ($('#navigation').attr('class') != 'index_nav') {
                    $('#navigation').stop().animate({'top': 0 - 50}, 250);
                    setTimeout(function () {
                        $('#navigation').addClass('index_nav').css({'top': $(window).height()});
                        $('#navigation').stop().animate({'top': $(window).height() - 50}, 250);
                    }, 350);
                }
            } else if (subid != 'videos') {
                $('#navigation').css({'top': $(window).height() - 50}).removeClass('index_nav');
                $('#navigation').stop().animate({'top': $(window).height()}, 250);

                setTimeout(function () {
                    $('#navigation').removeClass('index_nav').css({'top': 0 - 50});
                    $('#navigation').stop().animate({'top': 0}, 250);
                }, 350);
            }
        } else if (id == 'dnf') {
            $('#index-area').animate({'left' : '-100%'}, 700);
            $('.page.dnf').find('iframe').attr('src', '/index/dnf#' + subid);
        } else if (id == 'tos') {
            $('#index-area').animate({'left' : '-200%'}, 700);
            $('.page.tos').find('iframe').attr('src', '/index/tos#' + subid);
        } else if (id == 'sd') {
            $('#index-area').animate({'left' : '-300%'}, 700);
            $('.page.sd').find('iframe').attr('src', '/index/sd#' + subid);
        } else if (id == 'ms2') {
            $('#index-area').animate({'left' : '-400%'}, 700);
            $('.page.ms2').find('iframe').attr('src', '/index/ms2#' + subid);
        }

        if (id != 'index' && $('#navigation').attr('class') == 'index_nav') {
            $('#navigation').css({'top': $(window).height() - 50}).removeClass('index_nav');
            $('#navigation').stop().animate({'top': $(window).height()}, 250);

            setTimeout(function () {
                $('#navigation').removeClass('index_nav').css({'top': 0 - 50});
                $('#navigation').stop().animate({'top': 0}, 250);
            }, 350);
        }
        if (subid != 'videos') {
            //$('#footer').hide();
            $('#footer').stop(true).animate({'bottom': 0 - 50}, 250);
        } else {
            //$('#footer').show();
            $('#footer').stop(true).animate({'bottom': 0}, 250);
        }
    }

    $(document).ready(function () {
        var page_name = getSharpArt(location.href);
        rollSubPage(page_name[1], page_name[2]);
        resizeFrame();
    });

    $(window).resize(function () {
        resizeFrame();
        if ($('#navigation').attr('class') == 'index_nav') {
            $('#navigation').css({'top': $(window).height() - 50});
        } else {
            $('#navigation').css({'top': 0});
        }
    });
</script>
