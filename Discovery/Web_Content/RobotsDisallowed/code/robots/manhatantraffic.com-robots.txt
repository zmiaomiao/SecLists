				function calculat()

					var pid=$("#planid").val();

							var rate=Number($("#earningrate"+pid+"").val());

							$("#profit").html(parseFloat(totalamount-(price*position*1)).toFixed(4));
							$("#totalreturn").html(totalamount);
						}
						else
						{
							
							var rate2=$("#earningrate"+pid+"").val().split('[@]');
							
							for(key1 in rate2)

									break;
								}
								
							}

							$("#profit").html(parseFloat(totalamount-(price*position*1)).toFixed(4));
							$("#totalreturn").html(totalamount);
						}
					}
