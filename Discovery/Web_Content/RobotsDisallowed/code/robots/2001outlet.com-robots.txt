    function getByteLength(data) {
        var len = 0;
        var str = data.substring(0);
        if (str == null) return 0;
        for (var i = 0; i < str.length; i++) {
            var ch = escape(str.charAt(i));
            if (ch.length == 1) len++;
            else if (ch.indexOf("%u") != -1) len += 2;
            else if (ch.indexOf("%") != -1) len += ch.length / 3;

        return len;
