				});			
			});
        },
        update: function() {
			var btn = $("#btnnnidclublogin"),
				usr = $("a.user"),
				border = $(".logoutborder");		
            $.post(
                club.urls.updateStatus,
                function(result) {
                    json = $.parseJSON(result);
                    club.login.status = json.user;
                    if (club.login.status.userName) {											
	                    btn.html(btn.data("logout"))
							.removeClass("cboxElement btn login white arrow_right small filter gradient login_box_link inline")
							.addClass("logout login arrow_right")						
							.off()
							.on('click', function(e) {
								e.preventDefault();
								nintendo.club.login.logout();
							});
	                    usr.html(json.user.userName)
							.show();
	                    border.show();	                    	                    
	                } else {
	                    btn.html(btn.data("login"))
							.removeClass("logout login arrow_right")
							.addClass("cboxElement btn login white arrow_right small filter gradient login_box_link inline")
							.off();
						usr.html("")
							.hide();
						border.hide();		
	                }
                }
            );
        }		
    }    
})(nintendo.club, nintendo);
