        function setCookie_header(name, value, exdays, path, domain, secure){
            var d = new Date();
            d.setTime(d.getTime() + (exdays*24*60*60*1000)); // expire day = 1 day
            document.cookie= name + "=" + escape(value) +
            ((exdays) ? "; expires=" + d.toGMTString() : "") +
            ("; path=/") +       //you having wrong quote here
            ((domain) ? "; domain=" + domain : "") +
            ((secure) ? "; secure" : "");
        }

        function getCookie_header(cname) {
            var name = cname + "=";
            var ca = document.cookie.split(';');

            for(var i=0; i<ca.length; i++) {
                var c = ca[i];
                while (c.charAt(0)==' ') {c = c.substring(1);}

                if (c.indexOf(name) != -1) {
                    return c.substring(name.length,c.length);
                }
            }

            return "";
        }

	function l_popup_default_options(input_options){
		var input_options = input_options ? input_options : {};

			speed: 400,
			transition: 'slideDown',
// 			follow: [true, false],

	function l_popup_set_close_btn($box, close_btn){
		if (close_btn) {
			$box.prepend('<span class="b-close"><i class="fa fa-times"></i></span>');
		}
	}

	// button callback attach
	// do some function when click $btn ... then close popup!
	function l_popup_set_click_func($box, $btn, func){
		$btn.off();
		$btn.on('click', function(){ $box.bPopup().close(); });
		$btn.on('click', func);
	}

	// set the html to $elem
	function l_popup_set_html($elem, html){
		if(html){ $elem.html(html); }
	}

		l_popup_set_html($('#l-confirm-btn-no'), options['noText']);

		l_popup_set_click_func($box, $('#l-confirm-btn-yes'), options['yesFunc']);
		l_popup_set_click_func($box, $('#l-confirm-btn-no'), options['noFunc']);
		// popup now!
		$box.bPopup(options);
