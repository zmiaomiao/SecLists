function getParam(name)
{
	var match = new RegExp(name + "=([^&]+)", "i").exec(location.search);
	if (match == null)
	{
		match = new RegExp(name + "=(.+)", "i").exec(location.search);
	}

	if (match == null)
	{
		return '';
	}

	match = match + "";
	result = match.split(",");

	return result[1];
}
