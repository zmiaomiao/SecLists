function menuFixer(){
	var fixerId='menuFixer';
	var objId='menu'
	var topLevel=2; //px
	
	if(document.getElementById(fixerId).getBoundingClientRect().top<topLevel){
		document.getElementById(objId).style.top=topLevel+'px';
		document.getElementById(objId).style.position='fixed';
		
		document.getElementById(fixerId).style.height=document.getElementById(objId).clientHeight+'px';
	}
	else{
		document.getElementById(objId).style.top='inherit';
		document.getElementById(objId).style.position='inherit';

		document.getElementById(fixerId).style.height='0px';
	}
}
function sidebarFixer(){
	var fixerId='sidebarFixer';
	var objId='sidebarObject'
	var topLevel=2; //px
	
	var body = document.body, html = document.documentElement;
	var height = Math.max( body.scrollHeight, body.offsetHeight, html.clientHeight, html.scrollHeight, html.offsetHeight );
	
	if(document.getElementById(fixerId).getBoundingClientRect().top<topLevel){
		
		var bottomPos=(height-window.scrollY-1120);
		
		
		document.getElementById(objId).style.top=(topLevel+(bottomPos<0?bottomPos:0))+'px';
		
		
		document.getElementById(objId).style.position='fixed';
		
		document.getElementById(fixerId).style.height=document.getElementById(objId).clientHeight+'px';
	}
	else{
		document.getElementById(objId).style.top='inherit';
		document.getElementById(objId).style.position='inherit';

		document.getElementById(fixerId).style.height='0px';
	}
}
