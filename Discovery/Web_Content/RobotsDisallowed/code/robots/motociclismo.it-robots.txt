    function header_loadMenuArticoli(menuId) {
        var subMenu = $("#headerSubMenu_" + menuId);
        if (subMenu.text().trim() !== "" &&
                $("#headerSubMenuContent_" + menuId).text().trim() === "") {
            $("#headerSubMenuContent_" + menuId).load(
                    '/articoli/menu/' + menuId
                    );

        }
    }

    function header_showMenu(menuId) {
        //alert(menuId);

        $(".menu-current").removeClass("menu-current");
        var subMenu = $("#headerSubMenu_" + menuId);

        if (subMenu.text().trim() !== "") {

            header_loadMenuArticoli(menuId);

            $(".headerSubMenuContent").hide();
            $(".headerSubMenu").hide();

            $("#headerSubMenuRow").show();
            $("#headerSubMenuContent_" + menuId).show();

            //seleziona <li> menu
            $("a[data-menuid='" + menuId + "']").parent().addClass("menu-current");
            subMenu.show();
        } else {

            $("#headerSubMenuRow").hide();
        }
    }

    function header_closeSubMenu() {
        $(".menu-current").removeClass("menu-current");
        $("#headerSubMenuRow").hide();
    }

    function header_changeCopertina() {
        var currentCopertina = -1;
        $(".headerCopertina").each(function () {
            if ($(this).css("display") !== "none") {
                currentCopertina = $(this).attr("id").substr("headerCopertina_".length);
            }
        });
        if (currentCopertina >= $(".headerCopertina").length - 1) {
            currentCopertina = 0;
        } else {
            currentCopertina++;
        }

        $(".headerCopertina").hide();
        $("#headerCopertina_" + currentCopertina).show();

    }
