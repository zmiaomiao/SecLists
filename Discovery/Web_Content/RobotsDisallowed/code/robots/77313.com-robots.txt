		pwd.style.display = "";
		pwd.value = "";
		pwd.focus();
		pwd.style.color ='#000000';
	}
	pwd.onblur = function() {
		if(this.value != "") return;
		this.style.display = "none";
		txtpwd.style.display = "";
		txtpwd.value = "密码";
		pwd.style.color ='#999999';
	}
}
</script>
