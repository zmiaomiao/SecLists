<script type="text/javascript">

	function loginLayover_getPosition(theObj){
		x = y = 0;
		h = theObj.offsetHeight;
		w = theObj.offsetWidth;
		
		while(theObj){
			x += theObj.offsetLeft;
			y += theObj.offsetTop;
			theObj = theObj.offsetParent;
		}
		return {height:h,width:w,x:x,y:y}
	}

	function loginLayover_showLoginLayover(){
		refPoint = loginLayover_getPosition(document.getElementById('setting_links'));	
		var loginLayoverDiv = document.getElementById('loginLayoverDiv');
		loginLayoverDiv.style.display='block';
		loginLayoverDiv.style.top = refPoint.y - 20 + 'px';
		loginLayoverDiv.style.left = refPoint.x - 200 + 'px';
		
		if( $("#layoverUsername").val() == '' )
		{
			$("#layoverUsername").focus();
		}
	}
	
	$("document").ready( function() {
		$('#loginLayover_ForgotPasswordLink').click(function()
		{
			window.location = '/web/chatr.portal?_nfpb=true&_pageLabel=ConsumerRegistration&forward=goToForgotPasswordPage&username=' + $('#layoverUsername').val();
		});
		
		var actionVal = $("#loginLayoverForm").attr('action');
		$("#loginLayoverForm").attr('action', '');
		
		 validator = $("#loginLayoverForm").validate({
			onkeyup: false,
			rules:
			{
				"layoverUsername":
				{
					minlength: 6,
					maxlength: 20
				},
				"layoverPassword":
				{
					minlength: 7,
					maxlength: 17
				}
			},
			submitHandler: function( form ) 
			{
				var errorText = '';

				if ( !isValidUserName( $("#layoverUsername").val() ) )
				{					
					errorText += "Invalid username/chatr phone number.";
				}

				if ( errorText != '' )
				{
					$('.loginLayoverError').html( errorText );
				}
				else
				{
					$("#loginLayoverForm").attr('action', actionVal);
					form.submit();
				}
			},
			errorPlacement: function( error, element )
			{
				element.addClass( 'error' );
				$('.' + $(element).attr('name') ).addClass('error');
				$('.loginLayoverError').text("Please correct fields in red");
			},
			unhighlight: function( element, errorClass )
			{
				$(element).removeClass( 'error' );
				$('.' + $(element).attr('name') ).removeClass('error');
			}
		});
			
		$("#loginLayoverClose").click( function(){ 
			$('#loginLayoverDiv').hide();
		});
		
		$('#loginLayoverDiv').mouseover(function(){
			loginLayover_showLoginLayover();
		});
		
		$('#loginLayoverDiv').mousemove(function(){
			loginLayover_showLoginLayover();
		});
	});
