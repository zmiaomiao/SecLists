<script>
(function() {

        

	function getRedirectionUrlFromLocation() {
		var redirectionUrl;

		if ((window.location.hash !== "#prtInvalid") && (/\?.*goto=/).test(window.location.href)) {
				redirectionUrl = unescape(window.location.href.replace(/^.*goto=/, ''));
				redirectionUrl = redirectionUrl.replace(/^(http(s)*:\/\/[^\/]*usabilitytools.com):[0-9]*\//g, '$1/');

				return redirectionUrl;
		} else {
				return false;
		}
	}

	function redirect(url) { //ugly, but handles hashes in goto urls and server can't
		var def = 'http://mobile.usabilitytools.com/app/webapp/',
				fromLocation = getRedirectionUrlFromLocation();

		url = (fromLocation ? fromLocation : (url ? url : def));

		window.location.href = url;
	}

	var $form = $('#login-form, #remind-form');
	$('#login-form').parsley().subscribe('parsley:form:validate', function() {
		$('#login-form .error').hide();
	});
	$('#remind-form').parsley().subscribe('parsley:form:validate', function() {
		$('#remind-form .error').hide();
	});
	$form.on('submit', function(ev) {

		var $form = $(this);

		$form.addClass('sending');

		$.ajax($form.attr('action'), {
			type: 'POST',
			dataType: 'json',
			xhrFields: {
				withCredentials: true
			},
			data: $form.is('#remind-form') ? $form.find('[name=email]').serialize() : $form.find('[name=username], [name=password]').serialize(),
			success: function(addr) {
			  if ($form.is('#remind-form')) {
					$form.find('.success').show();
					$form.removeClass('sending');
			  } else {
					redirect(addr.url);
				}
			},
			error: function(jqXHR) {
				if (jqXHR.status === 401) {
					$form.find('.error').show();
					$form.find('[name=password]').focus();
				} else if ((jqXHR.status === 404) && ($form.is('#remind-form'))) {
					$form.find('.error').show();
					$form.find('[name=email]').focus();
				} else {
					alert("We're currently updating UsabilityTools to a new version (it usually takes an hour). Don't worry - your data is safe.");
				}
				$form.removeClass('sending');
			}
		});

		//ev.preventDefault();
		return false;
	});

	$('.forgot-password').on('click', function() {
		$('.login-box .login').hide();
		$('.login-box .remind').show().find('[name=email]').val($('.login-box .login [name=username]').val()).focus();
		$('.login-box').addClass('remind-box');
		return false;
	});
	$('.forgot-password-back').on('click', function() {
		$('.login-box .remind').hide();
		$('.login-box .login').show().find('[name=username]').focus();
		$('.login-box').removeClass('remind-box');
		return false;
	});


})();

					$form.find('[name=password]').focus();

						'MISSING_DATA': 'Please fill in both fields.',

					var errors = $.parseJSON(jqXHR.responseText);

							var error = errors[errFieldName];
							if (error) { // because backend likes to send all fields, not just the ones that have messages
									$form.find('.error').text(errorMessages[error] || (error + '')).show();
							}
					}

				} else {
					alert("We're currently updating UsabilityTools to a new version (it usually takes an hour). Don't worry - your data is safe.");
				}
				$form.removeClass('sending');
			}
		});

		//ev.preventDefault();

})();
