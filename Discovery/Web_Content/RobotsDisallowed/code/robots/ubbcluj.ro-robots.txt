				$('.back-to-top').removeClass('absoluteTop');

			    $('.back-to-top').addClass('absoluteTop');
			    $('.back-to-top').removeClass('fixedTop');

				$('.back-to-top').fadeIn();

				$('.back-to-top').fadeOut();

	$('.back-to-top').click(function () {
			$('body,html').animate({

			}, 800);

		});
