<script type="text/javascript">

    function redirectToHttps() {

        var httpURL = window.location.hostname + window.location.pathname + location.search;
        if (httpURL.toLowerCase().indexOf('writefilecontent') == -1) {
            var httpsURL = "https://" + httpURL;
            window.location = httpsURL;
        }
    }

    redirectToHttps();
 
</script>
