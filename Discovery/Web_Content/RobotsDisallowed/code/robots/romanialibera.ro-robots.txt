        var promoWidgetApp = {

            selectorArticles : $('.widget-promo1 .widget-main-container .main'),
            selectorSliderContainer : $('.widget-promo1 .widget-main-container'),
            getRelatedSelectorActive : function() {
                return $('.widget-promo1 .more .active');
            },

            initialArticleBoxWidth : null, // setat in init
            sliderWidth : null, // setati in init

            setInitialArticleBoxWidth : function() {
                this.initialArticleBoxWidth = parseInt( this.selectorArticles.first().width() );
            },
            getArticlesNumber : function() {
                return parseInt(this.selectorArticles.length);
            },
            setSliderWidth : function() { // apelat 1 data in init
                this.setInitialArticleBoxWidth();
                var sliderWidth = this.getArticlesNumber() * this.initialArticleBoxWidth;

                this.sliderWidth = sliderWidth;
            },
            leftPos: null, // setat in init

            makeAnimation : function() {
                // animation
                var fn = this;

                // set related selector active
                var curentRelatedSelector = fn.getRelatedSelectorActive();
                curentRelatedSelector.removeClass('active');

                if( curentRelatedSelector.next().is('li') ) {
                    curentRelatedSelector.next().addClass('active');
                }
                else if(curentRelatedSelector.next().next().is('li')) {
                    curentRelatedSelector.next().next().addClass('active');
                }
                else {
                    $('.widget-promo1 .more ul li').first().addClass('active');
                }
                // end

                fn.selectorSliderContainer.animate({
                    left: '-=' + fn.leftPos
                }, 1200, function() {

                    var curLeft = parseInt($(this).css('left').replace('px',''));

                    if(fn.leftPos < 0) {
                        // daca este mai mic ca zero a fost resetat
                        // deci resetam leftPos la dimensiune normala
                        fn.leftPos = fn.initialArticleBoxWidth; return;
                    }

                    if(-curLeft+fn.initialArticleBoxWidth >= fn.sliderWidth) {
                        // daca curLeft mai mare ca sliderWidth inseamna ca a ajuns la capat
                        // deci setam left post la dimensiunea totala curLeft
                        fn.leftPos = curLeft;
                    }

                });


            },

            init : function() {
                this.setSliderWidth();
                this.leftPos = this.initialArticleBoxWidth;
                // set selectorSlider width
                this.selectorSliderContainer.css('width', this.sliderWidth + 'px');


                setInterval('promoWidgetApp.makeAnimation()', 7000);
            }

        };

        $(document).ready(function(){
            promoWidgetApp.init();
        })




    </script>
