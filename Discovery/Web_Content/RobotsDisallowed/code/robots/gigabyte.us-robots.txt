		function setMenuRow2Position() {
			var maxHeight = 0;
			$('.sub_menu>ul>li>div:first-child').each(function(i,k) {
				var _this = $(this);
				var _contentHeight = 0;
				if (i == 3) {
					return false;
				}
				var x = 0;
				$(this).children('a, div').each(function() {
					var _me = $(this);
					_contentHeight += _me.height();
					x += $(this).innerHeight() - $(this).height();
				});

				_contentHeight += x;

				maxHeight = (_contentHeight > maxHeight) ? _contentHeight : maxHeight;
				
			});
			$('.sub_menu>ul>li>div:first-child').each(function (i,k) {
				var _this = $(this);
				_this.css('min-height', maxHeight + 'px');
			});

			var sheight = 0;
			$('.sub_menu > ul > li').each(function () {
				var _lih = $(this).height();
				if (_lih > sheight)
					sheight = _lih;
			});

			if (userAgent.match("Chrome")) {
				var eheight = sheight - $('.menu_pc_item').height() * 1;
				$('.sub_menu>ul>li:last-child').css("height", eheight + "px");

			}

			if (userAgent.match("MSIE 8")) {
				var eheight = $('.sub_menu>ul div[cg="7"]').height();
				var cheight = $('.menu_pc_item').height() * 3;
				var q = sheight - eheight - cheight;			
				$('.menu_pc_item').css("margin-top", q + 'px');
				$('.sub_menu > ul > li:last-child').prev().css('width','24%');
				$('.sub_menu > ul > li:last-child').css('border-left', '1px solid #333333');
			}
			else {
				sheight = sheight - $('.menu_pc_item').height() * 3.5;
				$('.menu_pc_item').css({ 'bottom': '0px', 'padding-bottom': '2%' });
			}
		}
