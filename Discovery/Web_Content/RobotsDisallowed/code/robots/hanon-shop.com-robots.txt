function headerClick(e){
	if (e.target.getParent('.ks-popout') !== null) {
		//click inside the header item
		return;
	}
	e.stop();
	closeAllHeader();
}


function closeAllHeader(){
	$$('.ks-popout').removeClass('open');
	document.removeEvent('click', headerClick);
