            /*Sets a date in the Departure calendar to whatever it is currently selected in the matching dropdowns*/
            function AdjustDepartureCalendar(colorName) {
                var dayValue = $("#day").val();
                var monthYear = $("#month").val();

                if (dayValue != null && monthYear != null && dayValue != "0" && monthYear != "0") {
                    var departureDate = ConstructDateFromDropdownValues(dayValue, monthYear);
                    ApplyDateToCalendar(departureDate, "#departure_date_picker");
                }
            }

            /*Sets a date in  the Return calendar to whatevet it is currently selected in the matching dropdowns*/
            function AdjustReturnCalendar(colorName) {
                var dayValue = $("#day_ret").val();
                var monthYear = $("#month_ret").val();

                if (dayValue != null && monthYear != null && dayValue != "0" && monthYear != "0") {
                    var returnDate = ConstructDateFromDropdownValues(dayValue, monthYear);
                    ApplyDateToCalendar(returnDate, "#return_date_picker");
                }
            }


            /*Applies a new date to a specified calendar*/
            function ApplyDateToCalendar(dateObj, calendarId) {
                if (calendarId.toString().charAt(0) == "#")
                    $(calendarId).datepicker("setDate", dateObj);
                else
                    $("#" + calendarId).datepicker("setDate", dateObj);
            }

            /*Converts the values from the date dropdowns to an actual date object */
            function ConstructDateFromDropdownValues(dayValue, monthYearValue) {
                var dayValue = new Number(dayValue);

                //if the monthYear has a lenght of three the the month is less then 10 (only one char)
                if (monthYearValue.toString().length == 3) {
                    var monthValue = new Number(monthYearValue.toString().charAt(0));
                    var y1 = ("" + monthYearValue).toString().charAt(1);
                    var y2 = ("" + monthYearValue).toString().charAt(2);
                    var yearValue = new Number("20" + y1 + y2);

                    var resultDate = new Date(yearValue, monthValue - 1, dayValue); // minus one to month because they are Zero based ie. first month has a value of 0
                    return resultDate;
                }
                else {
                    //length of monthyear is 4, which means month has two digits ie 10 or more
                    var monthValue = new Number(monthYearValue.toString().charAt(0) + monthYearValue.toString().charAt(1) );
                    var y1 = ("" + monthYearValue).toString().charAt(2);
                    var y2 = ("" + monthYearValue).toString().charAt(3);
                    var yearValue = new Number("20" + y1 + y2);

                    var resultDate = new Date(yearValue, monthValue - 1, dayValue);
                    return resultDate;
                }
            }
