function getAbsoluteURL()
{
	if (absoluteURL != "") {
		var currentURL = document.URL;
		var currentProtocol = "";
	
		if (currentURL.indexOf("://") != -1) {
			currentProtocol = currentURL.substring(0, currentURL.indexOf("://"));
		}
		
		var savedProtocol = "";
		if (absoluteURL.indexOf("://") != -1) {
			savedProtocol = absoluteURL.substring(0, absoluteURL.indexOf("://"));
		}
		
		if (currentProtocol != savedProtocol) {
			absoluteURL = currentProtocol + absoluteURL.substring(absoluteURL.indexOf("://"));
		}
	}
	
	return absoluteURL;
}

    function  ReadCookie(name)
		{
		  	var a_all_cookies = document.cookie.split( ';' );
			var a_temp_cookie = '';
			var cookie_name = '';
			var cookie_value = '';
			var b_cookie_found = false; // set boolean t/f default f
			
			for ( i = 0; i < a_all_cookies.length; i++ ) {
				a_temp_cookie = a_all_cookies[i].split( '=' );
				cookie_name = a_temp_cookie[0].replace(/^\s+|\s+$/g, '');
				if ( cookie_name == name ) {
					b_cookie_found = true;
					if ( a_temp_cookie.length > 1 ) {
						cookie_value = unescape( a_temp_cookie[1].replace(/^\s+|\s+$/g, '') );
					}
					return cookie_value;
					break;
				}
				a_temp_cookie = null;
				cookie_name = '';
			}
			if ( !b_cookie_found ) {
				return null;
			}
		}

}

function isWelcome() {

	var c_name = 'wlcme';
	if (document.cookie.length > 0) {
		c_start = document.cookie.indexOf(c_name + "=");
		
		
		
		if (c_start != -1) {
			c_start = c_start + c_name.length + 1;
			c_end = document.cookie.indexOf(";", c_start);
			if (c_end == -1) c_end = document.cookie.length;
			return unescape(document.cookie.substring(c_start, c_end));
		}
	}
	return "";
}

 * DISPLAYING OR NOT DISPALAYING EMAIL SIGNUP MODAL.
 * Check if we need to display 'email signup' modal.
 * We display it only if either one of 'email signup' cookies is available:
 * tcp_email_signup_modal_persistent (user was logged in or subscribed already using email signup modal)
 * tcp_email_signup_modal_session (User already saw email signup modal using the same browser)

 * The HTML for modal is part of e-spot EmailSignupModal (e-spot content EmailSignupModal_Content)

function displayEmailSignupModal() {
	if(window.location.href.indexOf('ResetPasswordGuestErrorView') > 0){
		return;
	}
	var tcp_email_signup_modal_persistent = ReadCookie("tcp_email_signup_modal_persistent");   
	var tcp_email_signup_modal_session = ReadCookie("tcp_email_signup_modal_session"); 
	
	if(tcp_email_signup_modal_persistent == null)
		tcp_email_signup_modal_persistent = false;				   
	if(tcp_email_signup_modal_session == null) 
		tcp_email_signup_modal_session = false;
			      
	//console.debug("tcp_email_signup_modal_persistent = " + tcp_email_signup_modal_persistent + 
	//				"  tcp_email_signup_modal_session = " + tcp_email_signup_modal_session);                             
          
            
	if(tcp_email_signup_modal_persistent == false  &&  tcp_email_signup_modal_session == false) {
			 
	 	// Need to display email signup modal because neither cookies are set:
		// tcp_email_signup_modal_persistent
		// tcp_email_signup_modal_session 
		
		
		// cookie domain will be retrieved from common.properties 
		// DOMAIN_NAME=.childrensplace.com		 
		var cookieDomain = '.childrensplace.com';
		
		
		// For cookie value we use date when it is created.
		// Example: Thu, 12 Jun 2014 19:36:07
		var currentDate = new Date();
		var cookieValue = currentDate.toGMTString();
		
		 
		//console.debug("Need to display email signup modal because neither cookies are set. " + 
		//				" tcp_email_signup_modal_persistent = " + tcp_email_signup_modal_persistent + 
		//				" tcp_email_signup_modal_session = " + tcp_email_signup_modal_session); 			
	
		//console.debug("Setting cookie for email signup modal. " + 
		//				" cookieName = " + "tcp_email_signup_modal_session" + 
		//				" cookieValue = " + cookieValue + 
		//				" cookieDomain = " + cookieDomain);			
		
		// Email signup modal will be displayed now, so we need to set cookie.
		// As a result email signup modal will be not displayed for next request.
		setTCPSessionCookie("tcp_email_signup_modal_session", cookieValue, cookieDomain);
		               
		
		// open email signup modal.
		// The HTML for modal is part of e-spot EmailSignupModal (e-spot content EmailSignupModal_Content)
		$('#email-signup-modal-top').dialog({				
			modal: 'true',
			open: function(event, ui){
				// make sure text box with 'Enter Email Address' message does not have focus.
				// (this is common issue with modals)
				// http://stackoverflow.com/questions/1202079/prevent-jquery-ui-dialog-from-setting-focus-to-first-textbox
		        $('#email-signup-modal-text-field').blur();
		        $('#email-signup-modal-text-field').val('Enter Email Address');
		        // $(this).scrollTop(0); 

		        
		        
			},
			
										
			width: 480
		});
			
				
				
	} else {
			
		//console.debug("Need to hide email signup modal because one of cookies is set. " +
		//				" tcp_email_signup_modal_persistent = " + tcp_email_signup_modal_persistent + 
		//				" tcp_email_signup_modal_session = " + tcp_email_signup_modal_session);
	
		// We do not need to display email signup modal.
	
		// espot is placing <div> which represents emailsignup modal on the page.
		// If we do not need to display modal, we need to make sure <div> is invisible.
		$('#email-signup-modal-top').hide();
			
	}

}
