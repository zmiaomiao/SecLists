		$('select[name=country_id]').on('change', function() {

			//clear stateProvince out so we don't have conflicting data
			$('input[name=stateProvince_id]').val('');
			$('input[name=city]').val('');
			$('input[name=zip]').val('');

			//show newsletter option for NL
			if($(this).val() == 155) {
				$('#mail_optin').show();
			} else {
				$('#mail_optin').hide();
			}
		});
	}

	function checkForm() {
		var isError = '';

		$('input.required:visible, select.required:visible').each(function() {
			$(this).closest('.form-group').removeClass('has-error');
			if(!$(this).val().trim()) {
				//errStr += $(this).attr('placeholder') + ' is required<br />';
				$(this).closest('.form-group').addClass('has-error');
				isError = true;
			}
		});

		if(isError) return false;
		return true;
	}


	function initializePopup(acid) {
		if(acid) $('#account_creator_id').val(acid);

		$(':input:first').focus();
		initCountries();

		//try to automatically determine olson timezone data
		var tzone = jstz.determine_timezone().timezone;
		if(typeof tzone.olson_tz != 'undefined' && tzone.olson_tz != '') {
			$('#timezone').val(zone_map[tzone.olson_tz]);
		}

	}
