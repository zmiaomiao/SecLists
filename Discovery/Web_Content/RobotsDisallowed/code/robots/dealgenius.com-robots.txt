    /**
     * jQuery widget to infinite scroll frontpage deals.  Wraps the functionality
     * of jquery.aydus.iscroll.js.
     * 
     * Was originally going to return JSON, but server returns HTML
     */
    (function(jQuery) {
        'use strict';
        jQuery.widget('aydus.dealsiscroll', {
            
            // AJAX endpoint to hit
            callbackUrl: location.protocol + '//' + document.domain + '//deals/deal/moreDeals/',
            // list of cat ids
            categoryIdList: [43,15,16,18,36,12,10,13,28,17,35,23,34,20,26,33,19,21,14,30,32,37,41,40],
            // jQuery selector to use for iscroll wrapper
            targetSelector: 'body.catalog-product-view div.col-main',
            // persistent storage for current element index for categoryIdList
            catIdIndex: 0,
            // aydus.iscroll instance
            iscrollInstance: '',
                    
            _create: function(target) {
                var that = this;
                var target = jQuery(that.targetSelector);
                target.iscroll({
                    scrollOffset: 200,
                    triggerOnStart: false,
                    loaderHtml: '<div class="loading"><img src="http://cdn.dealgenius.com/skin/frontend/dealgenius/master/images/loader.gif" alt="loading" /></div>',
                    load: function(event, ui) {
                        that._get();
                    }
                });
                
                // store a reference to the iscroll widget
                var instance = target.data("aydus-iscroll");
                that.iscrollInstance = instance;
                
            },
                    
            _get: function(plugin) {

                var that = this;
                
                // get a category id from index
                var catId = that.categoryIdList[that.catIdIndex];
                
                // concat ajax url
                var targetUrl = that.callbackUrl + 'categoryId/' + catId;
                
                // http back to server to get more content
                jQuery.ajax({
                    type: 'get',
                    url: targetUrl,
                    success: function(data) {
                        
                        // hide loading placeholder via iscroll plugin
                        that.iscrollInstance.loadSuccess();
                        
                        // create markup from JSON and append
                        jQuery(that.targetSelector).append(data);
                        
                        // if you're out of category ids, no reason to request more data
                        if (that.catIdIndex < (that.categoryIdList.length - 1)) {
                            that.catIdIndex++;
                        } else {
                            that.iscrollInstance.end();
                        }
                    }
                });
            },
        });
    }(jQuery));
