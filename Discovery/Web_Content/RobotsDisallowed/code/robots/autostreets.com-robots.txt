//通过省市一级联动由id查询更新为按省市名查询
function selectCity(provinceName) {
    $("#sellCarCity").val("所在城市").siblings("div.form-input-show").find("span").text("所在城市");
    var sellCarCityUl = jQuery("#sellCarCityUl").empty().append('<li>所在城市</li>');
    if (provinceName!="") {
        $.ajax({
            url :"/dic/selectCity",
            type: "get",
            data : {provinceName : provinceName},
            success : function(data) {
                $.each(data, function(index, item){
                    sellCarCityUl.append("<li>" + item.name + "</li>");
                });
            }
        });
    } 
}

//查找省份内经销商所在市区
function selectNewCarCity(provinceName, city) {
    var newCarCityUl = jQuery("#newCarCityUl").empty();
    $.ajax({
       url: "http://www.autostreets.com/selectModelCity",
       dataType: "json",
       data: {province: provinceName}
    }).done(function(data) {
        if (data && data.length) {
            $("#newCarCity").val(data[0].city+ ","+ data[0].abbr).siblings("div.form-input-show").find("span").text(data[0].abbr);
            $(data).each(function(index, item){
                newCarCityUl.append("<li>"+item.city+"<input type='hidden' value='"+ item.city+ ","+ item.abbr +"'></li>");
                if (city && city == item.city) {
                    $("#newCarCity").val(item).siblings("div.form-input-show").find("span").text(item.city);
                }
            });
        }
    });
}
