function getCookie(name) {
    var array = document.cookie.split(/[ ;]/);
    var c;
    var n = array.length;
    var cookie = {};
    for (var i = 0; i < n; i++) {
        c = array[i].split("=");

    }
    return cookie[name];
}

        timeout: 5000

    //

    var ExciteMailInfo = function(config) {
        // レスポンスメール
        this.mail = config.response;
        this.init();
        // 未読数リンクは必ず出す
        this.menuHtml = this.getMenuHtml();
    };

    ExciteMailInfo.prototype = {

        init: function(){
            if(!this.mail.is_maintenance){
                // 未読数
                this.num = this.mail.unread_count.count ? this.mail.unread_count.count : 0;

                // 件名表示は3件まで（見た目の問題なのでフロントで制御することにした）
                this.limit = this.num >= 3 ? 3 : this.mail.unread_count.count;
                // 受信箱
                this.mails = [];
                if(this.num > 0) {
                    var resMails =  this.mail.unread_mail;
                    var subject = '', mail = {};
                    for (var i = 0; i < this.limit; i++) {
                        mail = resMails[i];
                        subject = this.getSubject(mail.subject);
                        link = mail.link;
                        this.mails[i] = {subject: subject,link: link};
                        subject = '';
                        mail = {};
                    }
                }
            }
        },
        getSubject: function(subject) {
            var result = (subject == '') ? '[件名なし]' : subject;
            return result;
        },
        // メール未読N件メニューリンク
        getMenuHtml: function() {
            var html = '';
            html += '<a href="' + this.mail.mailtop_url + '" data-ex="IDC:mail" class="mailIcon" id="excitemail">メール</a>' + "\n";
            if(!this.mail.is_maintenance){
                var newsMailClassName = (this.num == 0) ? 'newMail0' : 'newMail';
                html += '<a id="unreadNumLink" href="' + this.mail.mailbox_url + '" data-ex="IDC:mail" class="' + newsMailClassName +'">未読' + this.num +'件</a>' + "\n";
            }
            return html;
        }
    };

    // 
    // フレンズミニメ
    //
    // {"status":"0","newminimenum":"10","subject":["subject01","subject02","subject03","subject04","subject05"]}
    // status 0: 正常, newminimenum ミニメ未読件数, subject 未読の件名を新しい順から最大５件
    //
    var FriendsMailInfo = function(config) {
        this.friends = config.response;
        this.init();
        // 未読数リンクは必ず出す
        this.menuHtml = this.getMenuHtml();
    };

    FriendsMailInfo.prototype = {
        init: function(){
            // 未読数
            this.num = this.friends.newminimenum;
            // 件名表示は5件まで
            this.limit = this.num >= 3 ? 3 : this.friends.newminimenum;
            this.mails = [];
            if(this.num > 0) {
                var resMails =  this.friends.subject;
                var subject = '';
                for (var i = 0; i < this.limit; i++) {
                    subject = resMails[i];
                    // ミニメには必ず件名がある仕様
                    this.mails[i] = {subject: subject}; 
                    subject = '';
                }
            }
        },
        // ミニメ未読N件メニューリンク
        getMenuHtml: function() {
            var html = '';
            var newsMailClassName = (this.num == 0) ? 'newMail0' : 'newMail';
            html += '<a href="http://friends.excite.co.jp/mail/receive/" data-ex="IDC:friendsmail" class="mailIcon" id="friendsmail">ミニメ</a>' + "\n";
            html += '<a id="miniMailNumLink" href="http://friends.excite.co.jp/mail/receive/" data-ex="IDC:friendsmail" class="' + newsMailClassName +'">未読' + this.num +'件</a>' + "\n";
            return html;
        }
    };

    // 
    // エキサイトメールとフレンズミニメ表示メニュークラス
    //
    // メール関係が皆無のユーザの場合、このクラスのオブジェクトを生成してはいけない。
    //
    //
    var existsTooltipOn = false;  // global
    var DisplayMailMenus = function(config) {
        this.exMailInfo = config.hasOwnProperty('exMailInfo')  ? config.exMailInfo : null;
        this.frMailInfo = config.hasOwnProperty('frMailInfo') ? config.frMailInfo : null;
    };

    DisplayMailMenus.prototype = {
        exMailRender: function() {
            var exMailInfo = this.exMailInfo;
            $('div#exMail').css('display', 'block');
            $('div#exMail p#exMailMenu').append(exMailInfo.menuHtml);
            if((exMailInfo.num > 0) && !this.exMailInfo.mail.is_maintenance) {
                var on  = this.makeFunc(this.tooltipExMailActive, this);
                var off = this.makeFunc(this.tooltipExMailInactive, this);
                $('div#exMail').hover(on, off);
            }
        },
        frMailRender: function() {
            var frMailInfo = this.frMailInfo;
            $('div#frMail').css('display', 'block');
            $('div#frMail p#frMailMenu').append(frMailInfo.menuHtml);
            if(frMailInfo.num > 0) {
                var on = this.makeFunc(this.tooltipFrMailActive, this);
                var off = this.makeFunc(this.tooltipFrMailInactive, this);
                $('div#frMail').hover(on, off);
            }
        },
        makeFunc: function(f, o) {
            // obj.funcname()は obj.funcname.apply(obj) のシンタックスシュガー。
            // apply の第一引数が実行時コンテキストになる。
            var result = function(){
                f.apply(o);
            };
            return result;
        },
        // ツールチップの実処理
        tooltipExMailActive: function() {
            if(existsTooltipOn === true || this.exMailInfo.is_maintenance) {
                return;
            }
            existsTooltipOn = true;
            var exMailInfo = this.exMailInfo;
            $('div#exMailInbox').empty();
            $('div#exMailInbox').prepend('<ul></ul>');
            var html = '';
            for (var i = 0; i < exMailInfo.limit; i++) {
                html += '<li>';
                html += '<a href="' + exMailInfo.mails[i].link + '" data-ex="IDC:mail-subject">';
                html += exMailInfo.mails[i].subject;
                html += '</a>';
                html += '</li>' + "\n";
            }
            html += '<p class="tooltipLink"><a href="' +  exMailInfo.mail.mailbox_url + '" data-ex="IDC:mail-subject">メール受信箱を見る</a></p>';
            $('div#exMailInbox ul').append(html);

            // IE8 でNG
            //$('div#exMailPreview').stop(true).animate({opacity: 1}, "fast").css("display", "block");
            // IE6,7 でNG
            //$('div#exMailPreview').stop(true,true).fadeIn('fast').css("display", "block");
            $('div#exMailPreview').stop(true,true).css("display", "block");
        },
        tooltipExMailInactive: function() {
            existsTooltipOn = false;
            $('div#exMailInbox').empty();

            // IE8 でNG
            //$('div#exMailPreview').stop(true).animate({opacity: 0}, "fast").css("display", "none");
            // IE6,7 でNG
            //$('div#exMailPreview').stop(true,true).fadeOut('fast').css("display", "none");
            $('div#exMailPreview').stop(true,true).css("display", "none");
        },
        tooltipFrMailActive: function() {
            if(existsTooltipOn === true) {
                return;
            }
            existsTooltipOn = true;
            var frMailInfo = this.frMailInfo;
            $('div#frMailInbox').empty();
            $('div#frMailInbox').prepend('<ul></ul>');
            var html = '';
            var href = 'http://friends.excite.co.jp/mail/receive/" data-ex="IDC:friendsmail-subject';
            for (var i = 0; i < frMailInfo.limit; i++) {
                html += '<li>';
                html += '<a href="' + href +'">';
                html += frMailInfo.mails[i].subject;
                html += '</a>';
                html += '</li>' + "\n";
            }
            html += '<p class="tooltipLink"><a href="' + href +'">ミニメを見る</a></p>';
            $('div#frMailInbox ul').append(html);

            // IE8 でNG
            //$('div#frMailPreview').stop(true).animate({opacity: 1}, "fast").css("display", "block");
            $('div#frMailPreview').stop(true,true).css("display", "block");
        },
        tooltipFrMailInactive: function() {
            existsTooltipOn = false;
            $('div#frMailInbox').empty();

            // IE8 でNG
            //$('div#frMailPreview').stop(true).animate({opacity: 0}, "fast").css("display", "none");
            $('div#frMailPreview').stop(true,true).css("display", "none");
        }
    };

    // JSONP取得成功時処理

    var asyncApiRequestSuccess = function(responseOrigi) {
        var response = exciteApp.getAsyncLoginResponse(responseOrigi);
        if(! response) {
            return;
        }

        setFriendsAvatar(response);
        setFavBlog(response);

        var exUserServiceInfo = new ExUserServiceInfo(response);
        var userInfo = exUserServiceInfo.serviceStatus;
        if(userInfo.exMail === true) {
            var exMail = new ExciteMailInfo({response: response.mail});
            new DisplayMailMenus({exMailInfo: exMail}).exMailRender();
        }
        if(userInfo.frMail === true) {
            var frMail = new FriendsMailInfo({response: response.friends});
            new DisplayMailMenus({frMailInfo: frMail}).frMailRender();
        }
    };

    $.ajax({
        url: asyncApiRequestConfig.uri,
        data: {
            'type': asyncApiRequestConfig.type
        },
        dataType: 'jsonp',
        cache: true,
        timeout: asyncApiRequestConfig.timeout,
        success: function(response){
            asyncApiRequestSuccess(response);
        },
        error: function(XMLHttpRequest, status, errorThrown) {
        }
    });

    function setFavBlog(response){
        if (!response || !response.blog || !response.blog.update) {
            return;
        }
        var tag = '';
        tag += '<dl>';
        $.each(response.blog.update,function(key,res) {
            if (res.blogurl && res.subject && res.subject) {
                tag += '<dt><a href=\"' + res.blogurl + '\">' + res.subject +'<\/a><\/dt>';
                tag += '<dd>by <a href=\"' + res.blogurl + '\">' + res.blognm +'<\/a><\/dd>';
            }
        });
        tag += '<\/dl>';
        $('div#after_exblog_favorite').append(tag).parent().show();
        $('div#login_exblog').css('display','');
    }
