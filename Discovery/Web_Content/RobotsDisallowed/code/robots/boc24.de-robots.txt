  //allow navigation by cursors
  function navSuggestions(e) {
    arrowKey = e.which||e.keyCode;
    var text="";
    switch(arrowKey) {
      case 38:
        if (sS.suggestLinkCount > 0){
        	sS.decLinkCount();
        }
        else if (sS.suggestLinkCount == 0) sS.suggestLinkCount=sS.totalSuggestions;
        if (sS.suggestLinkCount > 0){
          text = jQuery(".term a#"+"autoSelectOption_"+sS.suggestLinkCount+" span").text();  
        }
        else  text=sS.lastTerm;
        setSearchText(text);
        break;
      case 40:
        //alert('down arrow');
        if (sS.suggestLinkCount < sS.totalSuggestions){
          //console.info("holds down");
          sS.incLinkCount();
        }
        else if (sS.suggestLinkCount == sS.totalSuggestions) sS.suggestLinkCount=0;
        if (sS.suggestLinkCount > 0){
          text = jQuery(".term a#"+"autoSelectOption_"+sS.suggestLinkCount+" span").text();    
        }
        else text=sS.lastTerm;
        setSearchText(text);
        break;
    }

      sS.clearLinkCount();
