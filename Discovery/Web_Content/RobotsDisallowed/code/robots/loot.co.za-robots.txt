<script>
$(function()
{
	$("#subscribe-modal").dialog({
	    autoOpen: false,
	    resizable: false,
	    position: 'center',
	    stack: true,
	    height: 'auto',
	    width: 'auto',
	    modal: true
	});

	$("#subscribe_button").click(function()
	{  
		// validate and process form here  
		$('#subscribe_error').hide();  
		var name = $("input#name").val().trim();
		var pos = name.indexOf(' ');
		var surname;
		if (pos < 0)
		{
			$("#subscribe_error").html("Please enter both your name and surname");
			$('#subscribe_heading').hide();
			$("#subscribe_error").show();  
			$("input#name").focus();  
			return false;  
	    }  
		surname = name.substring(pos+1);
		name = name.substring(0,pos);
		var email = $("input#email").val().trim();  
		if (email == "")
		{
			$("#subscribe_error").html("Please enter a valid email address");
			$('#subscribe_heading').hide();
			$("#subscribe_error").show();  
			$("input#email").focus();  
			return false;
	    }  
		var dataString = encodeURI('name='+ name + '&surname=' + surname + '&email=' + email);  
		$.ajax({  
			type: "GET",  
			url: "/shop/subscribe",  
			dataType : "text",
			data: dataString,  
			success: function(response)
			{
				if (response == "OK")
				{
					$("input#name").val("");  
					$("input#email").val("");
					$('#subscribe_heading').show();
					$("#subscribe_error").hide(); 
					$("#subscribe-modal").dialog('open');
				}
				else
				{
					$("#subscribe_error").html(response);
					$('#subscribe_heading').hide();
					$("#subscribe_error").show();  
					$("input#name").focus();  
				}
			}
		});  
		return false; 
	});  
});  
</script>
