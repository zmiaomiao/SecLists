function updateWeatherWidget(){
	if( ( navigator.userAgent.match(/Chrome/i) || navigator.userAgent.match(/Firefox/i) || navigator.userAgent.match(/Safari/i) )){
		var serviceUrl = "/sitewidgets/fd3d/";
		var browserIE = false;
	}else{
		var serviceUrl = "/sitewidgets/fd/";
		var browserIE = true;
	}
	//console.log('serviceUrl',serviceUrl);
	var ajaxLocationId = $.cookie(currentDomain+'_settings_location');
	if(ajaxLocationId == '' || ajaxLocationId == null || ajaxLocationId == undefined){
		ajaxLocationId = globalSettings.defaultCookieValues.locationID;
		$('.autocomplete').val(globalSettings.defaultCookieValues.locationTitle);
		//set the default for domain
		$.cookie(currentDomain+'_settings_location',globalSettings.defaultCookieValues.locationID,{path:'/'});
		$.cookie(currentDomain+'_settings_location_title',globalSettings.defaultCookieValues.locationTitle,{path:'/'});
		$.cookie(currentDomain+'_settings_administrative',globalSettings.defaultCookieValues.adminID,{path:'/'});
		$.cookie(currentDomain+'_settings_administrative_title',globalSettings.defaultCookieValues.adminTitle,{path:'/'});
	}
	$.ajax({
		url:serviceUrl+ajaxLocationId,
		dataType:'html',
		beforeSend:function(){
			$('#homepage-weather-block').html('');
			$('#homepage-weather-loading').fadeIn(400);
		}
	}).done(function(data){
		$('#homepage-weather-block').html(data).fadeIn(500);
		$('#homepage-weather-loading').fadeOut(500);
		//next ten days button action
		$('#next-ten-days-link').mouseenter(function() {
			if($(window).width() > 1175){
				$(this).animate({'right': '970px'},200);
			}
		}).mouseleave(function() {
			if($(window).width() > 1175){
				$(this).animate({'right': '891px'},200);
			}
		});
		
		if(browserIE){
			// Event: mouse over : weather widget columns 
			$.fn.expandColomn = function(){
				if($(this).attr('id') == 'current-day'){
					var expandedHeightDay = '197px';
					var expandedHeightNight = '142px';
				}else{
					var expandedHeightDay = '240px';
					var expandedHeightNight = '190px';
				}
				$(this).children('.day-weather-container').animate({height: expandedHeightDay}, 250);
				$(this).children('.night-weather-container').animate({height: expandedHeightNight}, 250);
			}
			// Event : mouse out : weather widget columns 
			$.fn.unexpandColomn = function(){
				var originalHeightDay = '147px';
				var originalHeightNight = '92px';
				$(this).children('.day-weather-container').animate({height: originalHeightDay}, 250);
				$(this).children('.night-weather-container').animate({height: originalHeightNight}, 250);
			}
			// hide all columns 
			function hideColomns(){
				$('#current-day').hide();
				$('#next-day-1').hide();
				$('#next-day-2').hide();
				$('#next-day-3').hide();
				$('#next-day-4').hide();
				$('#next-ten-days-link').hide();
			}
			// Show all columns
			function showColomns(){
				$('#current-day').slideDown(500);
				$('#next-day-1').slideDown(650);
				$('#next-day-2').slideDown(800);
				$('#next-day-3').slideDown(950);
				$('#next-day-4').slideDown(1000 , function(){
					$('#next-ten-days-link').fadeIn(250);
				});
			}
			$('.wrappernot3d .hp-weather-colomn').mouseenter(function(e) {
				$(this).expandColomn();
				});
			$('.wrappernot3d .hp-weather-colomn').mouseleave(function(e) {
				$(this).unexpandColomn();
			});
		}//end if browserIE
		
	}).fail(function(err){
		//alert('error');
		console.log('error !',err);
	});
}//EOF: updateWeatherWidget()

function updateTextForecast(){
	$('.text-forecast-container').stop().slideUp(400);
	textServiceUrl = '/weather_text/'+$.cookie(currentDomain+'_settings_location');
	//console.log('textServiceUrl : '+textServiceUrl);
	$.ajax({
	  url:textServiceUrl,
	  type: 'GET',
	  dataType:'jsonp'
	}).done(function(response){
	  //console.log(response);
	  $('.text-forecast-text').html(response.note);
	  $('.text-forecast-container').slideDown(500);
	  
	  animateTextForecast();
	  
	});
}//EOF: updateTextForecast

function animateTextForecast(){
	clearInterval(textInterval);
	if($('.text-forecast-text').height() > 30){
		if($('.text-forecast-text .three-dots').length == 0){
			$('.text-forecast-text').css('position','relative').append('<div class="three-dots" style="position:absolute;top:0;left:0;background:#fff;line-height:28px;padding:0 0 0 10px;">...</div>');
		}
		//two lines
		var textInterval = setInterval(function(){
			if($('.text-forecast-text').css('margin-top') == '0px' || $('.text-forecast-text').css('margin-top') == '0' || $('.text-forecast-text').css('margin-top') == 0){
				$('.text-forecast-text').animate({marginTop:'-28px'});
			}else{
				$('.text-forecast-text').animate({marginTop:'0'});
			}
		},5000);
	}else{
		//single line do nothing..
	}
	
}

function infiniteLoad(){
  var scrollPositionTop = $(window).scrollTop();
  var documentHeight = $('html').height();
  var windowHeight = $(window).innerHeight();
  if(scrollPositionTop  >= documentHeight - windowHeight - 50){
	//alert('limit',scrollPositionTop + windowHeight,documentHeight);
	//limit
	var url = '/module/misc/page/'+(currentMiscPage+1)+'?max=6';
	if(currentMiscPage+1 ==5)
		 url +='?max=7';
	else
		 url +='?max=6';
	if(currentMiscPage < maxNumberOfPages){
	  //middle colomn
	  $.ajax({
		url:url,
		dataType:'html',
		async:false,
		beforeSend:function(){
		  $('.colomn-middle .colomn-loading').remove();
		  $('.colomn-middle').append('<div class="colomn-loading"></div>');
		}
	  }).done(function(data){
		  currentMiscPage++;
		  $('.colomn-middle .colomn-loading').remove();
		  $('.colomn-middle').append(data);
		  isNewAddAppend = false;
	  });
	}
	if(currentNewsPage < maxNumberOfPages){
	
	  url = '/module/tax/page/'+(currentNewsPage+1);
	  if(currentNewsPage==2){
			url +='?type=home&native3=1';
		}
		else if(currentNewsPage==3){
			url +='?type=home&native4=1';
		}
		else if(currentNewsPage==4){
			url +='?type=home&native5=1';
		}
		console.log(url);
		
	  //right colomn
	  $.ajax({
		url:url,
		dataType:'html',
		async:false,
		beforeSend:function(){
		  $('.colomn-right .colomn-loading').remove();
		  $('.colomn-right').append('<div class="colomn-loading"></div>');
		}
	  }).done(function(data){
		  currentNewsPage++;
		  $('.colomn-right .colomn-loading').remove();
		  $('.colomn-right').append(data);
		  
				//console.log("LOAD MORE","ROWS=",$(".colomn-right>div").length,"PAGE NUMBER",currentNewsPage);
				if(currentNewsPage==3){
					AwShowAdUnit("native-dfp-block-3");
				}
				else if(currentNewsPage==4){
					AwShowAdUnit("native-dfp-block-4");
				}
				else if(currentNewsPage==5){
					AwShowAdUnit("native-dfp-block-5");
					//addNativeAd(dfpNativeParams.native5);
				}
		
		  
	  }); 
	}
  }else{
    //console.log(scrollPositionTop,documentHeight,windowHeight);
  }
}//EOF: infiniteLoad()
