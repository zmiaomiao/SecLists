function $type(obj){
		if (obj == undefined) return false;
		if (obj.$family) return (obj.$family.name == 'number' && !isFinite(obj)) ? false : obj.$family.name;
		if (obj.nodeName){
			switch (obj.nodeType){
				case 1: return 'element';
				case 3: return (/\S/).test(obj.nodeValue) ? 'textnode' : 'whitespace';
			}
		} else if (typeof obj.length == 'number'){
			if (obj.callee) return 'arguments';
			else if (obj.item) return 'collection';
		}
		return typeof obj;
	};

function $clear(timer){
		clearTimeout(timer);
		clearInterval(timer);
		return null;
	};

function isEmpty(elementOrId){
	var e = $(elementOrId);
	if (e){
		HomeUtil.empty(e);
		var value = e.value;
		return value.blank(); 	
	}
	return true;
}

HomeUtil = {};

HomeUtil.DEFAULT_VALUES = {};

HomeUtil.contains = function (e){
	var txtValue = e.value;
	var defValue = HomeUtil.DEFAULT_VALUES[e.id];
	if (txtValue.search(defValue) > 0 || txtValue == ''){
		e.value = defValue;
	}
};

HomeUtil.empty = function (e){
	var txtValue = e.value;
	var defValue = HomeUtil.DEFAULT_VALUES[e.id];	
	if (txtValue == defValue ){
		e.value = '';		
	}	
};

HomeUtil.isEmpty = function(elementOrId){
	var e = $(elementOrId);
	if (e){
		HomeUtil.empty(e);
		var value = e.value;		
		return value.blank(); 	
	}
	return true;
};

HomeUtil.submitForm = function(formId){
	var submit = true;
	var elsForm = $(formId).getElements();
	elsForm.each(
			function(el){
				if (el.id != ''){
					HomeUtil.empty(el); 
					if (el.value == ''){
						submit = false;
					}	
				}							
			});
	if (submit) { 
		$(formId).submit();
	} else {
		HomeUtil.setFormToDefaultValue();
	}	
	return false;
};


HomeUtil.submitLoginForm = function(formId){
	$('txt_usrname_was').value = '';
	$('txt_password_was').value = '';
	// Se i campi della form sono vuoti	
	var isEmptyUsername = HomeUtil.isEmpty($('txt_userName'));
	var isEmptyPw = HomeUtil.isEmpty($('txt_passWord'));	
	if (!isEmptyUsername && !isEmptyPw){				
		$('txt_usrname_was').value = $('txt_userName').value;
		$('txt_password_was').value = $('txt_passWord').value;
		// resetto i campi per la sicurezza
		$('txt_userName').value = HomeUtil.DEFAULT_VALUES['txt_userName'];
		$('txt_passWord').value = HomeUtil.DEFAULT_VALUES['txt_passWord'];
		$(formId).submit();	
	} else {
		HomeUtil.setFormToDefaultValue();
	}
	return false;
};

HomeUtil.setFormToDefaultValue = function(){

	HomeUtil.DEFAULT_VALUES['txt_search'] = 'Cerca';
	HomeUtil.DEFAULT_VALUES['txt_search_sx'] = 'Cerca';
	HomeUtil.DEFAULT_VALUES['txt_userName'] = 'Username';
	HomeUtil.DEFAULT_VALUES['txt_passWord'] = 'Password';
	
	if ($('txt_search') && ($('txt_search').value == null || $('txt_search').value == '')){ 
		$('txt_search').value = HomeUtil.DEFAULT_VALUES['txt_search'];
	}

	if ($('txt_search_sx') && ($('txt_search_sx').value == null || $('txt_search_sx').value == '')){ 
		$('txt_search_sx').value = HomeUtil.DEFAULT_VALUES['txt_search_sx'];
	}
	if ($('txt_userName') && ($('txt_userName').value == null || $('txt_userName').value == '')){ 
		$('txt_userName').value = HomeUtil.DEFAULT_VALUES['txt_userName'];
	}
		
	if ($('txt_passWord') && ($('txt_passWord').value.blank())){ 
		$('txt_passWord').value = HomeUtil.DEFAULT_VALUES['txt_passWord'];		
	}


	if ($('txt_usrname_was')){		
		$('txt_usrname_was').value = '';
	}
	if ($('txt_password_was')){
		$('txt_password_was').value = '';
	}	
};

HomeUtil.createSlideNews = function (containerId, className, btnBackId, btnNextId){
	var newsContainer = $(containerId);
	var sizeDim = 241;
	if (isBrowserIE6){

	}
};

	}
	return false;	
}

HomeUtil.setLocationHref = function (target, url){
	if ('_self' == target)	{
		document.location.href = url;
	} else {
		window.open(url, '_blank');
	}
};

/**
* divContainer indica il contenitore
* sourceMedia è il percorso del video o flash
* type è il tipo di elemento (video|flash)

HomeUtil.updateMediaElement = function( aElementId, divContainer, source, type, url, index) {

	var aElement = $(aElementId);
	if (aElement) {
		// Imposto la classe no_select sul link
		$$('#thumb_bar .link').each(
				function(aHtml){
					if (!aHtml.hasClassName('no_select')){
						aHtml.addClassName('no_select');		
					}								
				});
		
		
		// Tolgo la classe select 		
		$$('#thumb_bar .select').each(
			function(imgHtml){
				imgHtml.removeClassName('select');										
			});

		// Add class select all'elemento corrente
		aElement.removeClassName('no_select');
		var imgEl = $(aElement.id+'_img');
		if (imgEl){
			imgEl.addClassName('select');			
		}
	}
	
	if ('image' == type){
		$(divContainer).removeClassName('backGroundFlash');
		var currentImage = new Element('img', { src: source });
		$(divContainer).update(currentImage);
		// Nel caso di immagine imposto il link
		var divLinkElement = $('bannerUrl');
		if (divLinkElement){
			if (url != null && url != ''){													
				var target = '_self';
				if (url.search('www') > 0 || url.startsWith('http') > 0){
					target = '_blank';
				}		
				
				divLinkElement.stopObserving('click'); 
					
				divLinkElement.observe('click', function(){
					HomeUtil.setLocationHref(target, url);  
					 });
									
				//divLinkElement.writeAttribute({onclik : });
				divLinkElement.setStyle({cursor: 'pointer'});
			} else {
				//linkElement.writeAttribute({href: 'javascript:void(0);', target: '_self'});
				/*divLinkElement.observe('click', function(){
					HomeUtil.setLocationHref('_self', 'javascript:void(0);');  
					 });*/
				divLinkElement.stopObserving('click'); 				
			}
		}
		
	} else if ( 'flash' == type){
		
		if (index == 1){
			$(divContainer).addClassName('backGroundFlash');
		} else {
			$(divContainer).removeClassName('backGroundFlash');
		}
			
		var flashContainer = $('flashContainer');

		if (flashContainer == null){
			flashContainer =  new Element('div', { id: 'flashContainer' });
			$(divContainer).update(flashContainer);
		}

		var divLinkElement = $('bannerUrl');
		/*divLinkElement.observe('click', function(){
			HomeUtil.setLocationHref('_self', 'javascript:void(0);');  
			 }); */

				640, 272, "9.0.0", "/cms-web/js/swfobject/expressInstall.swf",
				flashVars, attributes, params);	
	}

};
