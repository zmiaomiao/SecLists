    function login(msgholder, callback, subFix) {
        if (subFix == undefined) {
            subFix = "";
        }
        var loginId = $("#login-id" + subFix).val();
        if (loginId == "" || loginId == "请输入邮箱") {
            var msg = "请输入邮箱或者用户名进行登录";
            if (msgholder != undefined) {
                $("#" + msgholder).html("<span class='warning'>" + msg + "</span>");
            } else {
                alert(msg);
            }
            $("#login-id" + subFix).val("");
            $("#login-id" + subFix).focus();
            return false;
        }

        var pwd = $("#password" + subFix).val();
        if ($("#password" + subFix).val() == "") {
            var msg = "请输入密码";
            if (msgholder != undefined) {
                $("#" + msgholder).html("<span class='warning'>" + msg + "</span>");
            } else {
                alert(msg);
            }
            $("#password" + subFix).val("");
            $("#password" + subFix).focus();
            return false;
        }

        var form = $("#login-form" + subFix);
        //改串化为逐个获取，主要将用户名escape编译 兼容中文名称
        var param = "loginId=" + escape(loginId) + "&password=" + pwd;
        param += "&entrance=" + encodeURIComponent(document.URL);
        var remember = $("#remember").attr("checked");
        if (remember) {
            param += "&remember=1";
        }
        $.ajax({
            url: form.attr("action"),
            data: param,
            type: "POST",
            success: function (res) {
                responseSuccess(res, msgholder, callback, subFix);
            }
        });
        return false;
    }
    function responseSuccess(res, msgholder, callback, subFix) {

            var msg = "登录信息不正确，请重新输入";
            if (msgholder != undefined) {
                $("#" + msgholder).html("<span class='warning'>" + msg + "</span>");
            } else {
                alert(msg);
            }
            $("#login-id" + subFix).select();
            return false;
        } else if (res == "-2") {
