		if (tsaLayerIsRunning) return;

			},

				var chatAvailability = data.chatAvailability;
				$("#help-layer .state0").addClass("hidden");

					$("#help-layer .state1_1").removeClass("hidden");

					$("#help-layer .state1_0").removeClass("hidden");
				}
			}
		});

		function getCookie(name) {
			var start = document.cookie.indexOf(name+"=");
			var len = start+name.length+1;
			if ((!start) && (name != document.cookie.substring(0,name.length))) return(null);
			if (start == -1) return(null);
			var end = document.cookie.indexOf(";",len);
			if (end == -1) end = document.cookie.length;
			return(unescape(document.cookie.substring(len,end)) );
		}
		
		function setCookie(name, value, expiredays, path, domain, secure) {
			var expires = new Date();
			expires.setDate( expires.getDate() + expiredays );
			var cookieString = name + "=" +escape(value) +
			( (expires) ? ";expires=" + expires.toGMTString() : "") +
			( (path) ? ";path=" + path : "") +
			( (domain) ? ";domain=" + domain : "") +
			( (secure) ? ";secure" : "");
			document.cookie = cookieString;
		}

			$.ajax({
				url : '/home.html?ajaxMode=true&ajaxAction=tsa_cs',

				async : true,
				cache : false,
				crossDomain : true,
				data : {},
				error : function (jqXHR, textStatus, error) {
				},
				complete : function(){
					$("#help-box").removeClass("loading");
					$("#help-box").addClass("with-back");
					tsaBoxIsRunning = false;
				}, 
				success : function (data, textStatus, jqXHR) {
					var chatAvailability = data.chatAvailability;
					if (chatAvailability > 0) {
						animateBox("state0", "state1_1");
					} else {
						$("#help-box").addClass("small-size");
						animateBox("state0", "state1_0");
					}
				}
			});
		}
		
		function animateBox(state0, state1) {
		
			state0_back = state0;
			state1_back = state1;
			
			$("#help-box ." + state0).addClass("fadeOutLeft");
			$("#help-box ." + state1).removeClass("hidden").addClass("fadeInRight");
			setTimeout(function(){
				// $("#help-box-full .item").removeClass("animated");
				$("#help-box ." + state0).addClass("hidden");
				$("#help-box ." + state0).removeClass("fadeOutLeft");
				$("#help-box ." + state1).removeClass("fadeInRight");
			}, 1000);
							
		}
		
		function animateBoxBack () {
		
			$("#help-box ." + state1_back).addClass("fadeOutRight");
			$("#help-box ." + state0_back).removeClass("hidden").addClass("fadeInLeft");
			setTimeout(function(){
				// $("#help-box-full .item").removeClass("animated");
				$("#help-box ." + state1_back).addClass("hidden");
				
				$("#help-box ." + state1_back).removeClass("fadeOutRight");
				$("#help-box ." + state0_back).removeClass("fadeInLeft");
			}, 1000);
							
		}
