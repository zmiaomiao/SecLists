function isMobileDevice() {

		try {
			if (jQuery.url.param("mobile")) {
				$('.mini_page_margins').css('max-width', '950px');

			//	isMobile = false;
			//	return isMobile;
			//}
		}
		catch (e) {
			//cannot access url from accessories showroom
		}
		
		// iPhone (all modells and iPod Touch)
		if (agent.search(/iphone/) != -1) {
			isMobile = true;
		}
		// Android (all versions, HTC Desire, Legend, Magic, Dream, Hero, T-Mobile G2, T-Mobile touch, Samsung I7500 Galaxy, Motorola Morrison, Philips V808)
		else if (agent.search(/android/) != -1) {
			$('#mobile_mainnav .ul_level_1, #mobile_mainnav .module_navigation_level').css('min-width', '950px');
			isMobile = true;
		}
		// BlackBerry (all versions)
		else if (agent.search(/blackberry/) != -1) isMobile = true;
		// Palm Pre
		else if (agent.search(/applewebkit.+pre/) != -1) isMobile = true;
		// Samsung M850
		else if (agent.search(/samsung.+sprint m850/) != -1) isMobile = true;
		// All Nokia (includes Nokia 6260)
		else if (agent.search(/nokia/) != -1) isMobile = true;
		// All SymbiosOS based mobiles (includes Nokia 6260)
		else if (agent.search(/symbian/) != -1) isMobile = true;
		// Return false for none mobile browsers
		else isMobile = false;
    }
	return isMobile;
}
