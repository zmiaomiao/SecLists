function changefader()
{
	_s = 	$(".faderADV");

	_alldiv = _s.find("div");
	_active = _s.find("div.active");
	index = _alldiv.index(_active);


	_size = _alldiv.size();
	
	if( index + 1 == _size  )
	{ _now = 0 }
	else
	{ _now = index + 1 }

	_s.find("div.active").removeClass("active").fadeOut("slow");
	$(_s.find("div").get(_now)).addClass("active").fadeIn("slow");

	setTimeout(changefader, 5000)

}
</script>
