/** Add commas for stat formatting */
function addCommas(nStr)
{
nStr += '';
x = nStr.split('.');
x1 = x[0];
x2 = x.length > 1 ? '.' + x[1] : '';
var rgx = /(\d+)(\d{3})/;
while (rgx.test(x1)) {
x1 = x1.replace(rgx, '$1' + ',' + '$2');
}
return x1 + x2;
}

if(link){ 
var a = $('<a/>'); a.attr('href', link); a.attr("target",target);
$newMasthead.wrap(a);
}
// Apply any caption to the image
if( title_text ) {
var title = $('<h2>' + title_text + '</h2>').wrapInner(a).get(0).outerHTML;
var caption = description && description.length ? $('<p>' + description +'<span></span></p>').wrapInner(a).get(0).outerHTML : "";

} else {
// If no caption exists, display a placeholder in case the next slide
// does contain one. If the placeholder doesn't exist, the .caption:last
// selector below will pick up the caption we want to display next and
// remove it.
$newMasthead.after( '<div class="caption"></div>' );
}
