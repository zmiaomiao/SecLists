    function QuantityChange() {
         var row_quantities = $("select[id^='checkout_tab_quantity-']");
         row_quantities.change(function() {
            TotalPrice();
            getTotalPrice();

            var quantity_ID = (this).id.split('-')[1];

function TotalPrice() {
	$("tr[id^='tabrow-']").each(function() {
		var num = (this).id.split('-')[1];
		var quantity = $("#checkout_tab_quantity-" + num).val();
		var checkout_price = $("#hidden_tab_price_" + num).val();
		
		var checkout_price_num =  checkout_price;
		var total = quantity * checkout_price_num;
		
		$("#checkout_tab_price_" + num + " span").html(total.toFixed(2));
	});
}

//Gets the prices from each row of the cart and adds it up
function getTotalPrice(do_request) {
    var prices_array = new Array(); // Where our prices are held 

    // For each .total_item_price - a <td> within my table.
    $(".checkout_tab_price").each(function(e){
        var text = $(this).text();                   // Get the value
        var prices = text; // Format it into a string
        prices_array.push(prices);                   // Push it onto our array
    });

    var result = eval(0);
    //  Add up our array
    for(i = 0; i < prices_array.length; i++)
    {
        temp = eval(prices_array[i]);
        result += temp;
    }
    // Round up our result to 2 Decimal Places
    result = Math.round(result*100)/100;

    $('#grandtotal').html("$" + result);

}

    function delete_item() {
     $("a[id^='removerow-']").click(function(event){

        /* 
            Workaround needed because IE 9 and lower triggers the placeholders to get reset when an item is removed.
            This command triggers the generation again
        */
        if (!$.support.placeholder) {
            $('input, textarea').trigger("blur.placeholder");
        }

        // Delete Row from Table
        var num = (this).id.split('-')[1];
        // Row Scope
        var row = $('#tabrow-' + num);
        var steptworow = $('#steptworow-' + num);
        var cookie_form = $('#goto_step1-' + num + "product");
        var formsubmit = $('#remove'+ num +'product');
        var product_name = formsubmit.data("product");

        var row_to_delete = row.length ===
                0 ? steptworow : row;

        // Effect & Remove from DOM
        row_to_delete.fadeOut(1000, function(){ 
            removeFromGiftBox(product_name);
            $(row).remove();
            $(cookie_form).remove();
            $(steptworow).remove();
            // Update the Prices (again)

            TotalPrice();
            getTotalPrice();
            updateCartCount();

            //If No Remove Rows Are Present, Cart Is Empty And Needs Different Text To Alert Customer
            if ($("a[id^='removerow-']").size() == 0) {
                $('#checkout_tab tr.border_bottom').after('<tr id="noitems2"><td colspan="6">You Have No Items In Your Shopping Bag</td></tr>');
                $('#submit_checkout_tab').remove();
                $('#continue-shopping').remove();
                $('#checkout-text').hide();
            }

            $(formsubmit).each(function(){
               var that = $(this);
               $.get(that.attr('action'), that.serialize());
           });

        /* 
            Workaround needed because IE 9 and lower triggers the placeholders to get reset when an item is removed.
            This command resets the placeholder functionality to allow it to work as usual
        */
        if (!$.support.placeholder) {
            $('input, textarea').addClass("placeholder");
        }

      });

            index = 4;
