            function uyeOlIlkAdimKontrol(){
                var hataGosterildi = 0;
                if ($('#username').val() == ''){
                    $('#uyeOlErrorMessage').html('Lütfen, email adresini giriniz.');
                    hataGosterildi = 1;
                }



                if ($('#memberDoc').prop('checked') == false){
                    if (hataGosterildi == 0){
                        $('#uyeOlErrorMessage').html('Lütfen, üyelik koşullarını kabul ediniz.');
                        hataGosterildi = 1;
                    }
                }

                if (hataGosterildi == 1){
                    $('#uyeOlErrorMessage').css('display','block');
                }else{
                    $.ajax({
                        type: "POST",
                        url: "http://www.butikbebe.com/ajaxRequests/",
                        data: { emailAddress: $('#username').val(), doJob: "isEmailAlreadyTaken" }
                    }).done(function( msg ) {
                        console.log('Kontrol edilen değere verilen yanıt : ' + msg);
                        if (msg == 'OK'){
                            $('.registerFirst').hide();$('.registerSecond').show();$('.fancybox-inner').height(570);
                        }else{
                            if (hataGosterildi == 0){
                                $('#uyeOlErrorMessage').html('E-Mail adresi zaten sistemimizde kayıtlı.');
                                $('#uyeOlErrorMessage').css('display','block');
                                hataGosterildi = 1;
                            }
                        }
                    });

                }
            }

        function showRemaining(divID,endTime) {

            var _minute = _second * 60;
            var _hour = _minute * 60;
            var _day = _hour * 24;
            var timer;

            var end = new Date(endTime);
            var now = new Date();
            var distance = end - now;
            if (distance < 0) {

                clearInterval(timer);
