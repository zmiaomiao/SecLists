        function setRadioFilter(s) {
            var pa = $(s).closest(".fill-box");
            var selectedval = parseInt($(s).val());
            if (selectedval >= 0) {
                pa.attr("data-value", pa.attr("data-fid") + "_" + $(s).val());
            } else {
                pa.attr("data-value", "");
            }
        }
        function setRadioFilterCate(s) {

        function setCheckboxFilter(s) {
            var pa = $(s).closest(".fill-box");
            var filterVal = "";
            var checkedArr = {};
            if ($(s).val() == "-1") {
                pa.find("input").not(s).removeAttr('checked');
                $(s).attr('checked', 'checked');
            } else {
                var checked = pa.find("input:checked");

                if (checked.length <= 0) {
                    pa.find("input[value='-1']").attr('checked', 'checked');
                } else if (checked.length > 1) {
                    pa.find("input[value='-1']").removeAttr('checked');
                }
                checked = pa.find("input:checked");
                checkedArr = checked.map(function () {
                    return $(this).val();
                }).get();
            }

            if (checkedArr.length > 0) {
                filterVal = pa.attr("data-fid") + "_" + checkedArr.join("v");
            }
            pa.attr("data-value", filterVal);
        }

        function setPriceFilter(s) {

        }

        $(".pop-fill .apply").click(function () {

        function builtFilterUrl() {
            var ctrs = $(".pop-fill .fill-box");
            var filterArr = ctrs.map(function () {
                return $(this).attr("data-value");
            }).get();
            var cateArr = filterArr.filter(function (n) { return n.indexOf("cat-") == 0; });
            var cate = (cateArr && cateArr.length > 0) ? cateArr[0] : "";
            filterArr = filterArr.filter(function (n) { return n != "" && n.indexOf("cat-") < 0; });

            var url = window.location.href;
            var urlitems = url.split('.h')[0].split('/');
            var filted = false;
            var sortStr = "";
            var isCateFilted = false;
            urlitems = urlitems.filter(function (n) { return n.indexOf("trang"); });
            for (var i = urlitems.length - 1; i >= 0; i--) {
                
                //sortStr = "";
                if (!isCateFilted) {
                    if (urlitems[i].indexOf("cat") == 0) {
                        if (cate != "") {
                            urlitems[i] = cate;
                        }
                        isCateFilted = true;
                    } else {
                        isCateFilted = false;
                    }
                }
                if (urlitems[i].indexOf("sort") == 0) {
                    filted = false;
                    sortStr = urlitems[i].substring(urlitems[i].indexOf("sort"), urlitems[i].length);
                    urlitems.splice(i, 1);
                    //break;
                } else {
                    if (urlitems[i].indexOf("__sort") >= 0) {
                        sortStr = urlitems[i].substring(urlitems[i].indexOf("__sort"), urlitems[i].length);
                        urlitems[i] = urlitems[i].replace(sortStr, "");
                    }
                    if (urlitems[i].indexOf("filter") == 0) {
                        filted = true;
                        urlitems[i] = (filterArr && filterArr.length > 0) ? "filter-" + filterArr.join(',') + sortStr : sortStr;
                        //if (!isCateFilter) urlitems[i] = cate + "/" + urlitems[i];
                        //break;
                    }
                }

            }
            if (!filted) {
                var newItem = "";
                if (filterArr && filterArr.length > 0) {
                    newItem = "filter-" + filterArr.join(',');
                }
                if (sortStr != "") {
                    newItem = (newItem != "") ? newItem + "__" + sortStr : sortStr;
                }
                if (newItem != "") {
                    urlitems.push(newItem);
                }
            }
            if (cate != "" && !isCateFilted) {
                if (urlitems.indexOf("s") >= 0) {
                    urlitems.splice(urlitems.indexOf("s") + 1, 0, cate);

                if (urlitems.indexOf("s") >= 0) {
                    urlitems.splice(urlitems.indexOf("s") + 1, 1);
                }
            }

        }

        function get_browser_info() {
            var ua = navigator.userAgent, tem, M = ua.match(/(opera|chrome|safari|firefox|msie|trident(?=\/))\/?\s*(\d+)/i) || [];
            if (/trident/i.test(M[1])) {
                tem = /\brv[ :]+(\d+)/g.exec(ua) || [];
                return { name: 'IE', version: (tem[1] || '') };
            }
            if (M[1] === 'Chrome') {
                tem = ua.match(/\bOPR\/(\d+)/)
                if (tem != null) { return { name: 'Opera', version: tem[1] }; }
            }
            M = M[2] ? [M[1], M[2]] : [navigator.appName, navigator.appVersion, '-?'];
            if ((tem = ua.match(/version\/(\d+)/i)) != null) { M.splice(1, 1, tem[1]); }

                name: M[0],
                version: M[1]
            };
        }
        function getOSName() {
            var OSName = "Unknown OS";
            if (navigator.appVersion.indexOf("Win") != -1) OSName = "Windows";
            if (navigator.appVersion.indexOf("Mac") != -1) OSName = "MacOS";
            if (navigator.appVersion.indexOf("X11") != -1) OSName = "UNIX";
            if (navigator.appVersion.indexOf("Linux") != -1) OSName = "Linux";
            return OSName;
        }
