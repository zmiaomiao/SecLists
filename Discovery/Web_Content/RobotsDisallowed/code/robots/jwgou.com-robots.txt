    function Jwgou_Time() {
        var TuanGouIds = $("#JwG_ids").val().split(',');
        var TuanGouSeconds = $("#JwG_secondNums").val().split(',');
        var JwG_status = $("#JwG_Hd_Status").val().split(',');
        var JwG_times = $("#JwG_Start_Time").val().split(',');
        for (var j = 0; j < TuanGouIds.length; j++) {
            if (JwG_status[j] == "OK") {
                var st = new showTime(TuanGouIds[j], TuanGouSeconds[j]);
                st.setTimeShow();
            }
            else {
                $("#jyg_timer" + TuanGouIds[j]).html(JwG_times[j]);
            }
        }
    }
    function showTime(tuanid, time_distance) {
        this.tuanid = tuanid;
        //PHP时间是秒，JS时间是微秒
        this.time_distance = time_distance * 1000;
    }
    showTime.prototype.setTimeShow = function() {
        var timer = $("#jyg_timer" + this.tuanid);
        var str_time;
        var int_day, int_hour, int_minute, int_second;
        time_distance = this.time_distance;
        this.time_distance = this.time_distance - 1000;
        if (time_distance > 0) {
            int_day = Math.floor(time_distance / 86400000);
            time_distance -= int_day * 86400000;
            int_hour = Math.floor(time_distance / 3600000);
            time_distance -= int_hour * 3600000;
            int_minute = Math.floor(time_distance / 60000);
            time_distance -= int_minute * 60000;
            int_second = Math.floor(time_distance / 1000);
            if (int_hour < 10)
                int_hour = "0" + int_hour;
            if (int_minute < 10)
                int_minute = "0" + int_minute;
            if (int_second < 10)
                int_second = "0" + int_second;
            str_time = int_day + ":" + int_hour + ":" + int_minute + ":" + int_second;
            timer.text(str_time);
            var self = this;
            setTimeout(function() { self.setTimeShow(); }, 1000); //D:正确
        } else {
            timer.text("已结束！");
            return;
        }
    }
    Jwgou_Time();
      
</script>
