	function fnReloadMainPage(searchType){
		var dynamicForm = document.createElement("form");        
		dynamicForm.name = "passForm";
        dynamicForm.method = "post";
        dynamicForm.action = "/index.html";

        document.body.insertBefore(dynamicForm, null);
        dynamicForm.submit();
	}
	/**
	 * add parameter to form
	 */
	function addParameter(dynamicForm , paramType , paramName , paramValue) {
		var paramObj = document.createElement("input");
		paramObj.type = paramType;
		paramObj.name = paramName;
		paramObj.value = paramValue; 
        dynamicForm.insertBefore(paramObj, null);
        return dynamicForm;
	}

	function fnBookmarkActionLogin(f_type,m_no,c_no) {
	    if ($.trim($('#loginForm #m_id').val()) == '') {
	        alert("Please enter ID/E-mail");
	        $('#loginForm #m_id').focus();
	    } else if ($.trim($('#loginForm #m_pwd').val()) == '') {
	        alert('Please enter PASSWORD');
	        $('#loginForm #m_pwd').focus();
	    } else {
	    	var expdate = new Date();
		    expdate.setTime(expdate.getTime() + 1000 * 3600 * 24 * 30); // 30ì¼		    
		    document.cookie = "yellow_saveid=" + escape ($('#loginForm #m_id').val()) + "; path=/; expires=" + expdate.toGMTString();
	    	
		    var data_ = "m_id="+$('#loginForm #m_id').val()+"&m_pwd="+$('#loginForm #m_pwd').val();
			
			$.ajax({
				type:"POST",
				url:"/yellow/login/login.do",
				data:data_,
				dataType:"json",
				success:function(data){
					if(data.message == 'SUCCESS'){
						document.body.removeChild(document.getElementById("layer_popup"));
						
				    	if(f_type == 'URL') fnMyFavoritesUrl();
				    	else fnMyFavoritesInsert(f_type,m_no,c_no);

				    	$('#loginForm #msg_content').html('Register as a member and get an email certificate<br />after which you may use');
				    	$('#loginForm #msg').show();
				    	$('#loginForm #m_pwd').val('');
					}else if(data.message == 'NO_FB_CONFIRM'){
				    	$('#loginForm #msg_content').html('You are currently enrolled with your Facebook account<br />Register as a member and get an email certificate<br />after which you may use');

				    }
				}
			});
	    }
	}
	
	function getLayerCookie(Name) {
	    var search = Name + "=";
	    if (document.cookie.length > 0) { // ì¿ í¤ê° ì¤ì ëì´ ìë¤ë©´
	        offset = document.cookie.indexOf(search);
	        if (offset != -1) { // ì¿ í¤ê° ì¡´ì¬íë©´
	            offset += search.length;
	            // set index of beginning of value
	            end = document.cookie.indexOf(";", offset);
	            // ì¿ í¤ ê°ì ë§ì§ë§ ìì¹ ì¸ë±ì¤ ë²í¸ ì¤ì 
	            if (end == -1)
	                end = document.cookie.length;
	            return unescape(document.cookie.substring(offset, end));
	        }
	    }
	    return "";
	}

	function fnBookmarkSite(){

			window.external.AddFavorite(ThisPageUrl,ThisPageTiTle);

		}
