function more()
{	var cand = document.getElementsByTagName('li');
	for(var n = 0 ; n < cand.length ; n++)
	{	var cls = cand[n].getAttribute('class')
		if(cls && /^(more|less|expand|collapse) id[0-9]+/.exec(cls))
		{	cand[n].onclick = domore;

{	var tgt = event.target;
