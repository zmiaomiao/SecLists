<script type="text/javascript">
	$(function() {
		setActiveLink("anaSayfaLink");
		
		// Turkiye haritasi
		$('#contentContainer > #turkiyeHaritasi').cssMap({
			'size': 660,
			'onClick': function(d) {
				d.removeClass("active-region");
				var href = d.children("a").attr("href");
				var ilKodu = href.charAt(0) == '#' ? href.substring(1) : href;
				initAndShowIlSaglikKurumlariDialog(ilKodu);
			}
		}).css({ visibility: 'visible' });
		
		// Duyurular slider
		$("#contentContainer > #duyurular").iSlider({
			allLabel: "Tüm Haberler",
			allUrl: "haberler.jsp",
			dataUrl: "Duyurular",
			slices: 2
		});
		
		// Bize ulasin form
		var $contactUsPanel = $("#contentContainer #contactUsPanel");
		var $icerik = $contactUsPanel.find("#txtIcerik");
		var $icerikCountDown = $contactUsPanel.find("#icerikCountDown");
		var $contactUsButton = $contactUsPanel.find("button");
		$contactUsPanel.find("#txtTelefonNo").mask("0 999 999 99 99");
		$contactUsPanel.find("#cbKonu").change(function() {
			if ($(this).val() == "-1") {
				return;
			}
			$.ajax({
				cache: false,
				type: "GET",
				url: "Konu",
				data: { "id": $(this).val() },
				dataType: "json",
				context: this,
				success: function(data) {
					if (data != null && data.icerik && data.icerik.length > 0) {
						$("#baseDialog").bind('mousewheel DOMMouseScroll', function(e) {
							disableOuterScroll(e, this);
						}).dialog({
							buttons: [{ text: "Tamam", click: function() { $(this).dialog("close"); } }],
							close: function() {
								$(this).empty();
								$contactUsPanel.find("#txtIcerik").focus();
							},
							closeText: 'Kapat',
							height: 450,
							modal: true,
							resizable : false,
							title: data.baslik,
							width: 600
						}).html(data.icerik);					
					}
				}
			});
		});
		$icerik.limiter(500, $icerikCountDown);
		$contactUsButton.bind('click', function(event) {
			$contactUsButton.attr('disabled', true).addClass('disabled');
			var validInputs = checkInputs("#contentContainer #contactUsPanel");
			if (validInputs) {
				var $emailAddress = $contactUsPanel.find("#txtEposta");
				if (!isValidEmailAddress($emailAddress.val())) {
					showAlert("Lütfen geçerli bir \"E-Posta Adresi\" giriniz!", function() {
						markInput($emailAddress);
						$emailAddress.focus();
					});
					$contactUsButton.removeAttr('disabled').removeClass('disabled');
					return;
				}
				if ($.trim($icerik.val()).length < 25) {
					showAlert("Lütfen iletmek istediğiniz konuyu daha detaylı belirtiniz!", function() {
						markInput($icerik);
						$icerik.focus();
					});
					$contactUsButton.removeAttr('disabled').removeClass('disabled');
					return;
				}
				$.ajax({
					cache: false,
					type: "GET",
					url: "MesajKayit",
					data: { 
						"ad": $contactUsPanel.find("#txtAd").val(),
						"soyad": $contactUsPanel.find("#txtSoyad").val(),
						"eposta": $contactUsPanel.find("#txtEposta").val(),
						"telefonNo": $contactUsPanel.find("#txtTelefonNo").val(),
						"konu": $contactUsPanel.find("#cbKonu option:selected").text(),
						"icerik": $contactUsPanel.find("#txtIcerik").val()
					},
					dataType: "json",
					context: this,
					success: function(data) {
						if (data == "1") {
							showInformation("<p>Mesajınız başarıyla iletilmiştir.</p><p>Geri dönüşler öncelikli olarak belirtmiş olduğunuz e-posta adresinize yapılacaktır.</p><p>Lütfen maillerinizi kontrol ediniz.</p>");
							clearInputs("#contentContainer #contactUsPanel");
							$icerik.limiter(500, $icerikCountDown);
						} else {
							showAlert(data);
						}
					}
				});
			} else {
				focusOnError("#contentContainer #contactUsPanel");
			}
			$contactUsButton.removeAttr('disabled').removeClass('disabled');
		});

<div class="ui-dialog" id="dlgRandevuNasilAlinir" title="Randevu Nasıl Alınır?">
	<h2>Randevu Nasıl Alınır?</h2>
	ALO182 aranarak;
	<ul>
		<li>Ev, iş, ankesör ve cep telefonlarından Sağlık Bakanlığına bağlı ALO182 çağrı merkezi aranır.</li>
		<li>ALO182 numaralı telefonu arayan vatandaş tarafından çağrıyı karşılayan asistana (canlı operatör) randevu talebinde bulunulan vatandaşın T.C. Kimlik Numarası verilir. Asistan vatandaş bilgilerini doğrular.</li>
		<li>Vatandaş randevu talebinde bulunduğu hastane, poliklinik ve hekim bilgilerinden en azından hangi polikliniğe gideceğini asistana bildirdikten sonra, asistan uygun tarih ve saat dilimlerini vatandaşa iletir.</li>
		<li>Vatandaş seçimini yaparak randevusunu alır.</li>
	</ul>
	Web'den;
	<ul>
		<li>Vatandaşlar, ALO182'nin yanı sıra Online Randevu Sistemi aracılığıyla da randevu alabilmektedir.</li>
		<li>Vatandaşlar, randevu almak istedikleri Hekim Randevu Sistemi kapsamında olan hastanelerin internet sitelerindeki Online Randevu linkine basarak, Vatandaş Giriş ekranından kayıt oluşturup randevu alabilmektedir.</li>
	</ul>
	
	<h2>Randevu Nasıl İptal Edilir?</h2>
	ALO182 aranarak;
	<ul>
		<li>Ev, iş, ankesör ve cep telefonlarından Sağlık Bakanlığına bağlı ALO182 çağrı merkezi aranır. Bu arama randevu zamanından en az 80 dakika önce yapılmalıdır.</li>
		<li>ALO182 numaralı telefonu arayan vatandaş tarafından çağrıyı karşılayan asistana randevu talebinde bulunulan vatandaşın T.C. Kimlik Numarası verilir. Asistan vatandaş bilgilerini doğrular.</li>
		<li>Vatandaş iptalini istediği randevusunun, tarih ve zamanını asistana iletir. Asistan tarafından ilgili randevu iptal edilir.</li>
	</ul>
	Web'den;
	<ul>
		<li>Hekim Randevu Merkezi Online Randevu ekranına T.C. Kimlik Numarası ve şifre ile giriş yapılmalıdır.</li>
		<li>Randevu Geçmişi kısmına tıklayarak alınmış olunan tüm randevu bilgilerine erişilir ve Randevu iptal işlemi bu kısımdan yapılır.</li>
		<li>Randevu iptali için Randevu iptal kısmına tıklayarak randevu iptal edilir.</li>
	</ul>
