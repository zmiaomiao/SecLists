function FBLogin(callback) {
    var html = '<div class="cnt">' +

        FB.login(function(response) {
            if (response.authResponse) {
                var uid = response.authResponse.userID;
                
                if (response.authResponse.grantedScopes.indexOf('email') == -1) {
                    var need_email_tpl = '<div class="cnt">' +

                            background: '#FFF',

                        },
                        closemodal: 'close',
                        width: 350,
                        zindex: 500,
                        content: need_email_tpl
                    });
                    $need_email_d.on('click', '.close',function() {
                        $need_email_d.trigger('close');
                    }).on('click', '.j-login', function(e) {
                        login();
                        $need_email_d.trigger('close');
                    });
                } else {
                                            $.get('/services/fbLoginStatus.php', {uid:uid, status:'skip'}, function() {
                            Hub.trigger('FBCB::pub', [uid]);
                            Hub.trigger('FBCB::sub', [uid]);
                            if (!doNotReFresh) {
                                $.pageReload();
                            }
                                                    });
                                    }
                
            }
        },{'scope':'user_birthday,user_friends,user_likes,email' , 'auth_type': 'rerequest', 'return_scopes': true});

        addClass: 'fv-lgbox tac',
        overlayCss: {background: '#FFF', opacity: 0.9},
        closemodal: 'close',
        width: 350,
        zindex: 500,
        content: html
    });

    $d.on('click', '.close',function() {
        $d.trigger('close');
