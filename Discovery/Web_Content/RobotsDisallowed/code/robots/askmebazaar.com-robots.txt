<script language="JavaScript" type="text/javascript">
    function  eventSave()
    {
        var pattern = /^\w+([-+.']\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/ ;
        var regExEmail = new RegExp(pattern) ;
        var validate_check=1;
        var focusName='';

        if($("#customer").val() == "")
            {
            $("#error_name").show().html("Please enter name"); 
            if(focusName==''){
                focusName='customer';    
            }            
            validate_check=0;
        }
        else 
            {
            $("#error_name").hide();   
        }

        if (($("#customer_email").val()== "" ) || ( !$("#customer_email").val().match(regExEmail))) 
            {
            $("#error_email").show().html("Please enter valid email"); 
            if(focusName==''){
                focusName='error_email';    
            }
            validate_check=0;
        }
        else 
            {
            $("#error_email").hide();   
        }
        if($("#star_name").val() == "")
            {
            $("#error_star").show().html("Please enter name of your favourite star"); 
            if(focusName==''){
                focusName='star_name';    
            }
            validate_check=0;
        }
        else 
            {
            $("#error_star").hide();   
        }

        if(validate_check==0){
            $("#"+focusName).focus();
            return false ;
        }
        else {

            $.ajax({
                type: 'POST',
                url: 'event_promotion.php',
                data: {'customer':$("#customer").val() , 'customer_email': $("#customer_email").val(), 'star_name': $("#star_name").val()},
                success: function(response) {
                    if(response=='success')
                        {
                        $('#closefrm').modal('hide');
                        $("#error_all").hide(); 
                    }
                    else if (response =='error_field')
                        {
                        $("#error_all").show().html("Please fill up all fields");  
                    }
                    else if(response=='failed')
                        {
                        /*$("#starmsg").css("color", "red");
                        $("#starmsg").html('Failed');*/
                        $("#error_all").hide(); 
                    }

                }  
            });   
        }   
    }
</script>

    function check_buy_now_pincode(pdp_checkout_url){
        $.ajax({
            url: "http://www.askmebazaar.com/pincodeVerify.php",
            type: "POST",
            data: ({ pin_code: $("#shipping_pincode").val(), product_id:''}),
            success: function(response)
            {
                var resp = $.parseJSON(response);
                var ndd  =1;
                if (resp.status.trim() == 'success'){

                    var get_pincode_status     =  localStorage.getItem("gb_pincode_status");
                    if(ndd==resp.ndd && get_pincode_status ==1)
                        {
                        window.location = pdp_checkout_url;
                    }
                    else {
                        $("#pinNo").show("slow");
                        $("#ndd_msg_not_ava").show("slow").html('Please check the availability of this product at your pin code before proceeding to checkout.');
                    }
                }
                else if (resp.status.trim() == 'fail'){                    
                    $("#pinNo").show("slow");
                    $("#ndd_msg_not_ava").show("slow").html('Please check the availability of this product at your pin code before proceeding to checkout.');
                }
            }
        });
    }
