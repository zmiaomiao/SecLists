		$(window).scroll(function(){
			var $nav = $('#mob_menu');
			if ($('body').scrollTop() > 0) {
				if ($nav.data('size') == 'big') {
					$('#right_menu').fadeOut(400);
					$('#logo_menu').animate({height:'25px', width:'81px'}, 600);
					$('#f_btn').animate({left:'90px', top:'2px'}, 600);
					
					$nav.data('size','small').stop().animate({
					height:'34px'
				}, 600);
			}
		} else {
			if ($nav.data('size') == 'small') {
				$('#right_menu').fadeIn(400);
				$('#logo_menu').animate({height:'50px', width:'162px'}, 600);

				$nav.data('size','big').stop().animate({
