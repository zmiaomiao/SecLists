function validateBankAccount(str) {
var value = str.val();
var regex = /(^\d{4}[-]\d{4}[-]\d{2}[-]\d{10}$)|(^\d{20}$)/;
if (!regex.test(value)) {
return "* Invalid format (0000-0000-00-0000000000)";
} else {
var value = value.replace(/\-/g, '');
var banco = value.substring(0,4);
var sucursal = value.substring(4,8);
var dc = value.substring(8,10);
var cuenta = value.substring(10,value.length);
if (!validarCcc(banco, sucursal, dc, cuenta)) {
return "* LABEL37#(en)<!-- Número de cuenta inválido -->";
}
}
}
function obtenerDigito(valor){
valores = new Array(1, 2, 4, 8, 5, 10, 9, 7, 3, 6);
control = 0;
for (i=0; i<=9; i++)
control += parseInt(valor.charAt(i)) * valores[i];
control = 11 - (control % 11);
if (control == 11) control = 0;
else if (control == 10) control = 1;
return control;
}
function validarCcc(banco, sucursal, dc, cuenta) {
if (banco.length != 4 || sucursal.length != 4 || dc.length != 2 || cuenta.length != 10) {
return false;
} else if ((banco == "0000") && (sucursal == "0000") && (dc == "00") && (cuenta == "0000000000")) {
return false;
} else {
if (!(obtenerDigito("00" + banco + sucursal) == parseInt(dc.charAt(0))) ||
!(obtenerDigito(cuenta) == parseInt(dc.charAt(1))))
return false;
else
return true;
}
