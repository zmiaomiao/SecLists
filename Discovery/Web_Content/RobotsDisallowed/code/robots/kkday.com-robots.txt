    $('#loginEmailErrMsg').html("");

/**
 * 檢查email
 */
function is_error_email()
{
    var is_error=false;
    var loginEmail = $('#loginEmail').val();
    if ( ! loginEmail)
    {
        $('#loginEmailErrMsg').html('You must provide the email address.');
        is_error=true;
    }
    else
    {
        var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
        if ( ! re.test(loginEmail))
        {
            $('#loginEmailErrMsg').html('You may have entered an unknown email address');
            is_error=true;
        }
    }
    return is_error;
}

/**
 * 檢查密碼
 */
function is_error_password()
{
    var is_error=false;
    var loginPassword = $('#loginPassword').val();
    if ( ! loginPassword)
    {
        $('#loginPasswordErrMsg').html('You must provide the password.');
        is_error=true;
    }
    else
    {
        if (loginPassword.length < 6 )
        {
            $("#loginPasswordErrMsg").html('Password must be at least 6 characters.');
            is_error=true;
        }
    }
    return is_error;
}

/**
 * 檢查密碼是否一致
 */
function is_error_repassword()
{
    var is_error=false;
    var loginPassword   =   $('#loginPassword').val();
    var loginRepass     =   $('#loginRepass').val();

    //check 1 : loginPassword == loginRepass
    if ( loginPassword != loginRepass )
    {
        // show error :
        $("#loginPasswordErrMsg").html('Passwords do not match.');
        is_error=true;
    }
    else
    {
        if (loginPassword.length < 6 )
        {
            $("#loginPasswordErrMsg").html('Password must be at least 6 characters.');
            is_error=true;
        }
    }

    return is_error;
}
