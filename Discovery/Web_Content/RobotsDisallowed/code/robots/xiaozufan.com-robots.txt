function getValue( $input ){
    var val = $input.val();
    return val == $input.attr('placeholder') ? '' : val;
}

// init place holder
if( document.createElement('input').placeholder === undefined ){
    $('input[placeholder]').focus(function(){
        if( this.value == this.getAttribute('placeholder') ){
            this.value = '';
        }
    })
    .blur(function(){
        if( !this.value ){
            this.value = this.getAttribute('placeholder');
        }
    })
    .trigger('blur');
}


function fixImage( $img, cb ){
    $('<img/>').load(function(){
        var width = this.width;
        var height = this.height;
        var $parent = $img.parent();
        var boxWidth = $parent.width();
        var boxHeight = $parent.height();
        var imgRatio = width / height;
        var imgCss = {};
        if( boxWidth / boxHeight > imgRatio ){
            imgCss = {
                width: '100%',
                height: 'auto',
                left: '0',
                top: - Math.min( 40,  ( boxWidth / imgRatio - boxHeight ) / 2 )};
        } else {
            imgCss = {
                width: 'auto',
                height: '100%',
                top: '0',
                left: - ( imgRatio * boxHeight - boxWidth ) / 2
            };
        }
        $img.css( imgCss );

        cb && cb();
    })
    .attr('src', $img.attr('src'));
}
