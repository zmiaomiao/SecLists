<script>
    var PartnerFocusBox = {
        init: function () {
            this.bindClick();
            this.resize();
        },

        bindClick: function () {
            var scope = this;
            $('#nextPartner').click($.proxy(scope.onClick, scope));
        },

        resize: function () {
            $('#partnerFocusBox').find('.ct').css('width', '295px');
            $('#partnerFocusBox').find('.partner-focus').css('width', '173px');

            //resize text
            $('#partnerFocusBox').find('.hellip').click(function(){
                $(this).hide();
                $('#partnerFocusBox').find('.hellip-after').show();
            });
        },

        onClick: function () {
            this.addLoading();
            var params = {
                'random' : 1
            };
            var scope = this;
            adcell.fn.send(
                '/default/index/partnerfocus',
                params,
                {
                    success: $.proxy(scope.changePartner, scope),
                    error: $.proxy(scope.loadCompleteError, scope)
                }
            );
            return false;
        },

        changePartner: function (data) {
            this.removeLoading();
            $('#partnerFocusBox').html(data);

            $('#partnerFocusBox').find('.btn-medium').button({
                icons: {
                    primary: 'ui-icon-arrowthick-1-e'
                }
            });

            this.resize();
        },

        loadCompleteError: function () {
            this.removeLoading();
        },

        addLoading: function () {
            this.removeLoading();

            var headline = $('#partnerFocusBox').find('h3').first();
            var loading = $('<img style="margin-left: 10px;" id="partnerFocusLoading" src="/themes/adcell-default/images/loading.gif" />');
            $(headline).append(loading);
        },

        removeLoading: function () {
            $('#partnerFocusLoading').remove();
        },
    }

    PartnerFocusBox.init();
</script>
