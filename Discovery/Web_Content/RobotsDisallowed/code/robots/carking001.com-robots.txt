	}

function getPhoneVerifyCode(node){
	$("#registerErrorInfo").html("");
	var phoneNumber=$("#mobilePhoneR").val();
	if(phoneNumber==null||phoneNumber==""){
		$(node).next().html("");
		$(node).next().attr("style",'color:red;');
		return;
	}else{
		$(node).next().html("");
	}
	var _randCode=$("#_randCode").val();
	$.ajax({
		url: ctx+"/member/validMobilePhone/"+phoneNumber+"?tmpStr="+(new Date()).getMilliseconds()+"&randCode="+_randCode,
		dataType:"json",
		async : false,
	    success:  function(result){
	   		resultStatus = result.status;
	   		resultDesc = result.message;
	    }
	});
}
