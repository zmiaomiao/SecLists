        function verifPass(value,container)
{
var globalValidationText;
globalValidationText = "Le mot de passe doit etre compose de 6 a 20 caracteres aplhanumeriques (a,b,c, (...) A,B,C, (...) 0,1,2, (...) )";
minLength = "- Le mot de passe doit avoir au moins 6 caractères.";
maxLength = "- Le mot de passe doit avoir au maximum 20 caractères.";
resValid = "- Le mot de passe choisi est valide.";
resNotValid = "- Le mot de passe choisi n'est pas valide.";
var regex=Array();
var regexText=Array();
var mustContainSymbol;
regex[0] = Array();
regex[0][0] = '^[a-zA-Z0-9]*$';
regexText[0] = "- Le mot de passe doit être compose que des caractéres alphanumériques.";
if(mustContainSymbol != 0 && mustContainSymbol != 1) mustContainSymbol=0;
var globalValid=true; 
document.getElementById(container).innerHTML = globalValidationText+'<br />';
if(value.length < 6)
{
   document.getElementById(container).innerHTML += minLength+'<br />';
   globalValid=false; 
}
if(value.length > 20)
{
   document.getElementById(container).innerHTML += maxLength+'<br />';
   globalValid=false; 
}
var valid=Array(); 
for(i=0;i<regex.length;i++){
   var reg = new RegExp(regex[i][0] );
   var match = reg.exec(value);
   if(regex[i][1])
   {
      if (match != null)
      {
         if(regexText[i]) document.getElementById(container).innerHTML += regexText[i]+'<br />';
         valid[i] = false;
      }
   }
   else
   {
      if (match == null)
      {
          if(regexText[i]) document.getElementById(container).innerHTML += regexText[i]+'<br />';
         valid[i] = false;
      }
   }
}
for(i=0;i<valid.length;i++){
   if(valid[i] == false)
   {
      globalValid=false; 
   }
}
if(globalValid == true)
{
   document.getElementById(container).innerHTML += resValid
}
else
{
   document.getElementById(container).innerHTML += resNotValid
}
}    </script>
