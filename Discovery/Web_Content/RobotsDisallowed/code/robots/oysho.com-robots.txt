ItxShippingPage.prototype._redraw = function() {
    
    var self = this;
    
    // Reload methods
    if (self.activeZipCode != null  || (!!self.isChina && self.activeDistrictCode != null)) {
        
        var continueFn = function(methods) {
            // Update
            self.shippingMethods = methods;

            // Reset
            self.activeZipCode = null;
            self.activeDistrictCode = null;

            // Reload methods
            redrawShippings();
        };
        
        var code = !!self.isChina ? self.activeDistrictCode : self.activeZipCode;
        if (Inditex.iStoreJSON.details.postalCodeRestriction == 1 && !!code) {
            inditex.xRestGetShippingMethodsByZipCode(code, false).done(function(shippingMethods) {
            continueFn(shippingMethods);
            }).Catch(function(error) {
               inditex.xOpenErrorPopUp(350,200);
            });

        } else continueFn(inditex.iXShipping.shippingMethods);
    
    } else {
        // Reload methods
        redrawShippings();
    }

inditex.categoryExistsOnTree = function(categories, categoryId) {
    var i;

    if ((categories != null) && (categories.length != 0)) {		
        for (i = 0; i < categories.length; i++) {
            if (categories[i].id == categoryId) {
                return true;
            } else {
                if (inditex.categoryExistsOnTree(categories[i].subcategories, categoryId) == true) {
                    return true;
                }
            }
        }
    }
        
    return false;
}

ItxRenderGridClass.prototype._getCategory = function(product) {
    var savedCatId = 0;
    if (product.relatedCategories && product.relatedCategories.length > 0) {
        for (var i = 0; i < product.relatedCategories.length; i++) {
            var cat = product.relatedCategories[i];
            if (cat.id === inditex.iCategoryId) {
                return inditex.iCategoryId;
            } else if (cat.urlCategory === true && inditex.categoryExistsOnTree(inditex.iCategoryTreeJSON.categories,cat.id)) {
                return cat.id;
            }
        }
        return savedCatId;
    } else {
        return inditex.iCategorId;
    }
};

	} else {
		base = self.iStoreJSON.details.imageBaseUrl.replace("https:", "http:");

	if (type == 2) {
		for (var i = 0; i < image.aux.length; i++) {
			result.push(base + image.url + "_" + type + "_" + image.aux[i]
					+ "_" + size + "." + extension + "?t=" + image.timestamp);
		}
	} else {
		result.push(base + image.url + "_" + type + "_1_" + size + "."
				+ extension + "?t=" + image.timestamp);
	}

	return result;

ItxBookingPageClass.prototype._init = function () {
    var self = this;
    $('form[name="iBookingForm"]').validationEngine({
        promptPosition: 'bottomLeft',
        autoPositionUpdate: true,
        showOneMessageByField: true,
        scroll: false,
        focusFirstField: false
    });
    $('#bookingButtonSubmit').unbind().bind(Inditex.iXBindType, function (e) {
        e.preventDefault();
		$('#bookingButtonSubmit').unbind();
        self._submit();
    });
    if (self.iProductInfo) {
        // -- se recuperan los sku
        self._findSkusAvailables();
        $("#tblBookShopAvailable tbody tr").remove();
        var table = $("#tblBookShopAvailable");
        var row = self._renderDetail(self.iProductInfo.id, self.iProductInfo.name, self.iProductInfo.detail.displayReference, self.color);
        if (row) {
            row.appendTo(table);
            // -- se carga el combo de tallas
            var sizeCombo = self._loadComboSize(self.iProductInfo.id, self.skus);
            $('#comboSize_' + self.iProductInfo.id).append(sizeCombo);
            Inditex.xLoadCustomSelect("customSelectTalla_" + self.iProductInfo.id, null, function (elementSelected) {
                var skuInfo = elementSelected.data("skuInfo");
                if (skuInfo) {
                    self.skuSelected = skuInfo;
                    var idDivParent = elementSelected.parent().parent().parent().attr('id').split('_');
                    $('#price_' + idDivParent[1]).html(Inditex.formatPrice(skuInfo.price));
                }
            });
        }
    }
};

ItxBookingPageClass.prototype._submit = function () {
    var self = this;
    var bool = true;
    if (self.skuSelected) {
        // -- TODO en self.skuSelected esta la información del sku seleccionado
        if ($("#iBookingForm").validationEngine('validate')) {
            var firstName = $("#iBookingForm #firstName").val();
            var lastName = $("#iBookingForm #lastName").val();
            var email = $("#iBookingForm #email").val();
            var phone = $("#iBookingForm #phone").val();
            var captcha = $("#iBookingForm #capchaBooking").val();
            var numAllBookingItems = $('input[name=bookingStoreItems_bamStoreId]').length;
            var bamStoreId = new Array();
            var catentryId = new Array();
            var unitsOrdered = new Array();
            bamStoreId.push(self.stlocId);
            unitsOrdered.push(1);
            catentryId.push(self.skuSelected.sku);
            // -- TODO en este punto se hace el submit
            Inditex.getRestBookingAddBookingCmd({
                captcha: captcha,
                email: email,
                fullName: firstName + ' ' + lastName,
                phone: phone,
                bamStoreId: bamStoreId,
                catentryId: catentryId,
                unitsOrdered: unitsOrdered,
                onSuccess: function (aJson) {
                    if ((aJson) && ((aJson.errorKey == undefined) || (aJson.errorKey == null))) {
                        Inditex.xCloseOverlay();
                        Inditex.xOpenSuccessPopUp(350, 250, Inditex.iProperties["BOOKING_TRANSACTION_OK"], Inditex.iProperties["BOOKING_EMAIL_CONFIRMATION"]);
                    } else {
                        self._refreshCaptcha();
						$('#bookingButtonSubmit').unbind().bind(Inditex.iXBindType, function (e) {
							e.preventDefault();
							$('#bookingButtonSubmit').unbind();
							self._submit();
						});
                        Inditex.xOpenErrorPopUp(350, 200, Inditex.iProperties["DEFAULT_ERROR_MESSAGE_TITLE"], aJson.errorMessage);
                    }
                },
                onFailure: function (aStatus) {
                    self._refreshCaptcha();
						$('#bookingButtonSubmit').unbind().bind(Inditex.iXBindType, function (e) {
							e.preventDefault();
							$('#bookingButtonSubmit').unbind();
							self._submit();
						});
                    Inditex.xOpenErrorPopUp(350, 200);
                }
            });
        } else {
            bool = false;
        }
    } else {
        bool = false;
        Inditex.xOpenErrorPopUp(350, 290, Inditex.iProperties['TRANSACTION_BREAK'], Inditex.iProperties['BOOKING_SIZE_NO_SELECTED']);
    }
    if (!bool) {
        self._refreshCaptcha();
		$('#bookingButtonSubmit').unbind().bind(Inditex.iXBindType, function (e) {
			e.preventDefault();
			$('#bookingButtonSubmit').unbind();
			self._submit();
		});
    }
    return bool;
};
