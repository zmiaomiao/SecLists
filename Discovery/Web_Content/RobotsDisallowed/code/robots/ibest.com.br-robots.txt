var geoComponent = (function() {
    "use strict";

    // Checando a variável global "editor".
    var editMode = editor || false;

    // Objeto jQuery do componente.
    var component = null;

    // Popula componente de regionalização,
    // com dados vindos do parceiro.
    function populateComponent(data) {
        var baseUrl = data.urlBase;
        var componentTitle = (data.tituloComponente) ? data.tituloComponente : data.title;
        var tagPublicidade = (data.tagPublicidade) ? data.tagPublicidade : '';
        var openUrlAsLayer = (data.layer === "1") ? true : false;
        var results = data.noticias.chamadas;

        // Definindo título, publicidade e link do header do componente.
        definePub(tagPublicidade);
        component.find(".geo-titulo").html(componentTitle);
        component.find(".geo-link").attr("href", baseUrl);
        /* id para inserção de publicidade de acordo com a região */
        component.find(".adRetranca").attr("id", normalize(componentTitle).replace(/-/g, '_'));

        // Percorrendo as chamadas.
        for (var attr in results) {
            if (results.hasOwnProperty(attr)) {
                var news = results[attr];

                // Percorrendo os tipos de conteúdo de cada
                // chamada (titulo, link, imagem, etc).
                for (var type in news) {
                    if (news.hasOwnProperty(type)) {

                        // Conteúdo do elemento.
                        var content = news[type] || "";

                        // Elemento que deverá ser populado por esse conteúdo.
                        var elem = component.find(".geo-" + attr + "-" + type);

                        // Se houver dados desse tipo de conteúdo
                        // e o elemento a ser populado existir.
                        if (content !== "" && elem.length > 0) {

                            // Populando cada elemento de
                            // acordo com o seu tipo.
                            var externo = news.externo;                            
                            switch(type) {
                                case "link":
                                    /* 
                                        exceção para o parceiro O Dia
                                        Solicitado pelo próprio parceiro para retirada de abertura das notícias em Lightbox
                                    */
                                    if ( (/odia/g).test(baseUrl) ) { // se for O Dia
                                        content = content;
                                    } else { // caso contrário mantemos o padrão
                                        content = (openUrlAsLayer == true && externo == "false") ? baseUrl + "?url_layer=" + content : content;                              
                                    }

                                    elem.attr("href", content);
                                    break;
                                case "imagem":
                                    elem.attr("src", content);
                                    break;
                                default:
                                    elem.html(content);
                            }
                        }
                    }
                }
            }
        }
    }

    function definePub(tag){
        // var urlPub1 = 'http://adserver.ig.com.br/RealMedia/ads/adstream_sx.ads/'+tag+'@Middle2!Middle2';
        // var urlPub2 = 'http://adserver.ig.com.br/RealMedia/ads/adstream_sx.ads/'+tag+'@Middle3!Middle3';



        var urlPub1 = 'http://i1.statig.com.br/adapp/homeiG.html?par='+tag+',esquerda';
        var urlPub2 = 'http://i1.statig.com.br/adapp/homeiG.html?par='+tag+',direita';
        $('iframe#publicidadeHOGEOPub1').attr('src', urlPub1);
        $('iframe#publicidadeHOGEOPub2').attr('src', urlPub2);
    }

    function normalize(str) {
        str = str.toLowerCase()
                 .replace(/ /g, "-")
                 .split('-');

        for (var j = 0; j < str.length; j++) {
            str[j] = str[j].replace(/[àáâãäå]/g, 'a')
                     .replace(/æ/g, 'ae')
                     .replace(/ç/g, 'c')
                     .replace(/[èéêë]/g, 'e')
                     .replace(/[ìíîï]/g, 'i')
                     .replace(/ñ/g, 'n')               
                     .replace(/[òóôõö]/g, 'o')
                     .replace(/œ/g, 'oe')
                     .replace(/[ùúûü]/g, 'u')
                     .replace(/[ýÿ]/g, 'y');
        }       
        
        str = str.join('-');

        return str;
    }

    function getDataByRegion(category, city, state, success, error) {

        // Título do componente recebe
        // o nome da cidade por padrão.
        var componentTitle = city;
        var stateShortName = getStateShortName(state);

        // URL do json dos parceiros.
        var url = "/_static/misc/parceiros/" + stateShortName;
        var options = {
            type:  "GET",
            url: url + "/" + normalize(city) + "/" + category + ".json",
            dataType: "json",
            success: function(data) {

                // Definindo o título do componente no json (cidade ou estado).
                data.title = componentTitle;
                success(data);
            },
            error: function() {

                // Se não houver sucesso na busca pelos dados 
                // por cidade, define a url apenas com o estado.
                options.url = url + "/" + category + ".json"; 

                options.error = function() {
                    if (typeof error === "function") {

                        // Executa o callback de erro caso 
                        // não haja dados por estado.
                        error();
                    }
                };

                // Definindo o título do componente com o nome do estado.
                componentTitle = state;

                // Faz o request dos dados do parceiro por estado.                
                $.ajax(options);            
            }
        };

        // Faz o request dos dados do parceiro por cidade.
        $.ajax(options);            
    }

    function getStateShortName(state) {

            "acre": "ac",

    }

    return {
        init: function() {
            
            // Definindo o componente.
            component = $(".HOGEO27712");

            // Faixa do componente.
            var container = component.parents(".faixa");

            // Categoria do conteúdo do parceiro.
            var category = "noticias";

            // Callback caso a geolocalização retorne com sucesso.
            var success = function(data) {
                var location = data.split(", ");
                var city = location[0];
                var state = location[1];
                var country = location[2].toLowerCase();

                if (country === "br") {

                    // Callback para quando houver dados 
                    // de parceiros para a região do usuário.
                    var callbackSuccess = function(data) {

                        populateComponent(data);

                        // Exibe o componente.
                        component.css("display", "block");
                    };

                    // Callback para quando os dados não existirem.
                    var callbackError = function() {

                        // Remove a faixa do componente.
                        container.remove();                        
                    };

                    // Requisitando dados do parceiro.
                    getDataByRegion(category, city, state, callbackSuccess, callbackError);
                } else {

                    // Remove a faixa do componente caso
                    // o usuário acesse fora do Brasil.
                    container.remove();
                }
            };

            // Callback caso dê algum problema com a geolocalização.
            var error = function() {

                // Remove a faixa do componente caso haja
                // algum problema no request dos dados.
                container.remove();
            };

            // Produção. 
            if (!editMode) {

                if (typeof geolocation === "object") {

                // Exibe o componente.

        }

		function tiraAcento(text) {
		var acentos =[
			/[\300-\306]/g, /[\340-\346]/g, // A, a
			/[\310-\313]/g, /[\350-\353]/g, // E, e
			/[\314-\317]/g, /[\354-\357]/g, // I, i
			/[\322-\330]/g, /[\362-\370]/g, // O, o
			/[\331-\334]/g, /[\371-\374]/g, // U, u
			/[\321]/g, /[\361]/g, // N, n
			/[\307]/g, /[\347]/g, // C, c
			];

		var chars = ['A','a','E','e','I','i','O','o','U','u','N','n','C','c'];

		for (var i = 0; i < acentos.length; i++) {
		    text = text.replace(acentos[i],chars[i]);
		}

		return text;
