					function SetCookie(name, value, days) {
						var today = new Date();
						expires = new Date(today.getTime() + days*24*60*60*1000);
						document.cookie = name + "=" + escape(value) + "; expires=" + expires.toGMTString();
					};

					function getCookie(name) {
						var matches = document.cookie.match(new RegExp(
						"(?:^|; )" + name.replace(/([\.$?*|{}\(\)\[\]\\\/\+^])/g, '\\$1') + "=([^;]*)"
						));
						return matches ? decodeURIComponent(matches[1]) : undefined ;
					};

	function checkemail(email)
	{
		var filter=/^([\w-]+(?:\.[\w-]+)*)@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$/i
		if (!filter.test(email))
		{
			return false;
		}
		else
		{
			return true;
		}
	}

	function checkButtons()
	{
		
		var state=false;
		var id = "";
		for (var i=0; i < document.forms['myFormSubscribe'].length; i++)
		{
		  id = document.forms['myFormSubscribe'].elements[i].id;
		  if (document.forms['myFormSubscribe'].elements[i].name=="news_channel[]") {
              if (document.getElementById(id).checked)
              {
                state=true;
              }
		  }
		}
        
        
        
		if(state==false)
		{
			document.getElementById('save').disabled=true;
			document.getElementById('undo').disabled=true;						
		}
		else
		{
			document.getElementById('save').disabled=false;
			document.getElementById('undo').disabled=false;			
		}
		return false;
	}
