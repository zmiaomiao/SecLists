    function FFParseISBN(txt)
    {
        var regex = /[^0-9]/gi;
        txt = txt.replace(regex, "");
        var len = txt.length;
        var isbn = "";
        var ean = "";
        switch (len)
        {
            case 9:
            case 10:
                {
                    isbn = txt.substr(0, 9);
                    ean = "978" + isbn;
                    break;
                }
            case 12:
            case 13:
                {
                    ean = txt.substr(0, 12);
                    if (txt.substr(0, 3) == "978")
                    {
                        isbn = txt.substr(3, 9);
                        ean = "978" + isbn;
                    }
                    break;
                }
        }
        if (ean != "" && $.isNumeric(ean))
        {
            var checkEan = 0;
            for (var i = 1; i < 12; i += 2)
            {
                checkEan += (parseInt(ean.substr(i, 1)) * 3);
            }
            for (var i = 0; i < 12; i += 2)
            {
                checkEan += parseInt(ean.substr(i, 1));
            }
            checkEan = checkEan % 10;
            if (checkEan != 0)
            {
                checkEan = 10 - checkEan;
            }
            txt = ean.toString() + checkEan.toString();
            if (txt.length == 13)
            {
                location.href = "http://www.unilibro.it/libro/" + txt;
                return false;
            }
        }
        return true;
    }

    function FFTypeFilter(product)
    {

            case "libro":
            case "scolastici":

                    filter = "01";
                    break;
                }
            case "libri in lingua":
            case "books":
                {
                    filter = "08";
                    break;
                }
            case "dvd":
                {
                    filter = "02";
                    break;
                }
            case "cd":
            case "colonne sonore":
                {
                    filter = "04";
                    break;
                }
            case "game":
            case "accessori":
                {
                    filter = "03";
                    break;
                }
            case "mappamondi":
            case "puzzle":
            case "giochi":
                {
                    filter = "05";
                    break;
                }
            case "poster":
                {
                    filter = "06";
                    break;
                }
            case "scrittura":
            case "moleskine":
                {
                    filter = "07";
                    break;
                }
            case "ebook":
                {
                    filter = "09";
                    break;
                }
            case "cofanetti":

                    break;
                }
            default:
                {
                    filter = "";
                    break;
                }
        }
        if (filter != "")
        {
            filter = "&filterProdGroup=" + filter;
        }
        return filter;
    }
