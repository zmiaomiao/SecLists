		var doubleClick  = new function () {
			this.indexLeft = 0;
			this.indexRight = 0;
			this.indexPopup = 0;
			this.indexCenter = 0;
			this.lenLeft = 0;
			this.lenRight = 0;
			this.lenPopup = 0;
			this.lenCenter = 0;
			this.timeOutLeft;
			this.timeOutRight;
			this.timeOutPopup;
			this.timeOutCenter;
			this.Init = function () {
				doubleClick.lenLeft = $('.advLeft').length;
				doubleClick.runAdvLeft();
				$('#al').mouseover(function () {
					$('#btnCl').show();
				});
				$('#al').mouseout(function () {
					$('#btnCl').hide();
				});
				
				
				
				doubleClick.lenPopup = $('.advPopup').length;
				doubleClick.runAdvPopup();
				$('#popup').mouseover(function () {
					$('#btnPopup').show();
				});
				$('#popup').mouseout(function () {
					$('#btnPopup').hide();
				});
				

				doubleClick.lenRight = $('.advRight').length;
				doubleClick.runAdvRight();
				$('#ar').mouseover(function () {
					$('#btnCr').show();
				});
				$('#ar').mouseout(function () {
					$('#btnCr').hide();
				});
				doubleClick.lenCenter = $('.advCenter').length;
				doubleClick.runAdvCenter();
			};
			this.runAdvLeft = function() {

				$('.advLeft').hide();
				$('.advLeft:eq(' + doubleClick.indexLeft + ')').show();
				doubleClick.indexLeft++;
				if (doubleClick.indexLeft == doubleClick.lenLeft)
					doubleClick.indexLeft = 0;

				doubleClick.timeOutLeft = setTimeout(doubleClick.runAdvLeft, 6000);
			}
			this.runAdvRight = function () {

				$('.advRight').hide();
				$('.advRight:eq(' + doubleClick.indexRight + ')').show();
				doubleClick.indexRight++;
				if (doubleClick.indexRight == doubleClick.lenRight)
					doubleClick.indexRight = 0;

				doubleClick.timeOutRight = setTimeout(doubleClick.runAdvRight, 6000);
			}
			this.runAdvCenter = function () {

				$('.advCenter').hide();
				$('.advCenter:eq(' + doubleClick.indexCenter + ')').show();
				doubleClick.indexCenter++;
				if (doubleClick.indexCenter == doubleClick.lenCenter)
					doubleClick.indexCenter = 0;

				doubleClick.timeOutRight = setTimeout(doubleClick.runAdvCenter, 6000);
			}
			
			this.runAdvPopup = function () {
				var num = Math.floor((Math.random() * 3));
				$('.advPopup').hide();
				$('.advPopup:eq(' + num + ')').show();
				doubleClick.indexPopup++;			
				if (doubleClick.indexPopup == doubleClick.lenPopup)
					doubleClick.indexPopup = 0;

				doubleClick.timeOutPopup = setTimeout(doubleClick.runAdvPopup, 6000);
			}
			
			this.closeLeft = function () {
				clearTimeout(doubleClick.timeOutLeft);
				$('#al').hide();
			}
			this.closeRight = function () {
				clearTimeout(doubleClick.timeOutRight);
				$('#ar').hide();
			}
			this.closePopup = function () {
				clearTimeout(doubleClick.timeOutPopup);
				$('#popup').hide();
			}
