function photocomment(id)
{
	var logid=""; 
	if(logid=="")
	{
	 $.fancybox.open({
     href : 'http://aze.az/signin/index/from/1/bookmark/1',
     type : 'iframe',
	 autoSize: false, 
     padding : 5,
      height: 340, 
     width: 634
      });
	}
	else
	{
		var comment=document.getElementById('comment').value;
		if(comment=="")
		{ alert("Enter Comment"); } else {
		$.ajax({  
		type: "POST", 
		url: "http://aze.az/article/savecomment", 
		data:({id:id,comment:comment}),
		cache: false,
		success: function(data){ 
		//alert(data);
		if(data==0) { alert("Commented already");  } else { 
		document.getElementById('sitecomm').innerHTML=data; }
		document.getElementById('comment').value="";
		//window.location=window.location;
       
		}
		});
		}
	}
}
