function ValidateFieldsMiniBuscador() {
    var validated = false;
    if ($(tipoCMSelectId)[0].selectedIndex == 0) {
        $(tipoCMSelectId).closest(".custom-form_cuadro-medico").addClass("fieldRequiredErrorBorderRed")
        $("[id$='txtObligatorioCM']").addClass("fieldRequiredErrorRed");
        validated = false;

        $(tipoCMSelectId).closest(".custom-form_cuadro-medico").removeClass("fieldRequiredErrorBorderRed")
        $("[id$='txtObligatorioCM']").removeClass("fieldRequiredErrorRed");
        validated = true;

        $(provinciaSelectId).closest(".custom-form_provincia").addClass("fieldRequiredErrorBorderRed")
        $("[id$='txtObligatorioProvincia']").addClass("fieldRequiredErrorRed");
        validated = false;

        $("[id$='txtObligatorioProvincia']").removeClass("fieldRequiredErrorRed");
        if (validated) {

        }
    }
    return validated;
