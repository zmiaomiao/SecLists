	function loginPopupCenter(action) {
		var window_top = $(window).scrollTop();
		var window_height = $(window).height();
		var popup_height = $('#loginPopup').height();
		if(popup_height < window_height || action == 'open') {
			var new_popup_top = window_top + Math.round((window_height-popup_height)/2);
			$('#loginPopup').stop().animate({'top':new_popup_top+'px'},300,'linear', function() {

					if($('#loginPopupFieldLogin').val()) {
						$('#loginPopupFieldPassword').focus();
					} else {
						$('#loginPopupFieldLogin').focus();
					}
				}	
			});
		}
