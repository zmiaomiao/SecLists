			function hideAll()
			{
				$('.hiddenContent').each(function(){				
					$(this).hide();				
				});				
			}
			function hideAll2()
			{				
				$('.hideContent').each(function(){				
					$(this).addClass('dropMenu');
					$(this).removeClass('blockimportant');		
				});
				
				$('.nav li.dropdown').each(function(){
					$(this).hide();
				});
			}
			function showAll()
			{
				$('.nav li.dropdown').each(function(){
					$(this).show();
				});
			}
			
			function removeAllActiveClass()
			{
				$('.shareheader').removeClass('active');
				$('.search').removeClass('active');
			}	

			function showHidden()
			{
				$('.showHidden').each(function(){				
					$(this).click(function(){	
						removeAllActiveClass();
						var relToHide = $(this).attr('rel');
						var isVisible = '0';
						if($('#'+relToHide).is(':visible')){
							isVisible = '1';
						}					
						hideAll();	
						if(isVisible == '0'){
							$('#'+relToHide).fadeIn();	
							$(this).addClass('active');
						}									
					});				
				})
			}		
			$(document).click(function(e) 
			{
				var t = (e.target)

				if(
					!$(".shareTopContent").is(t) // if the target of the click isn't the container...
					&& $(".shareTopContent").has(e.target).length === 0 // ... nor a descendant of the container
					&& t!=$(".search").get(0) && t!=$(".shareheader").get(0) 
					&& !$(".searchTab").is(t) && $(".searchTab").has(e.target).length === 0
					) 
					{
					removeAllActiveClass();
					hideAll();				
				}
			});

			$(window).scroll(function(){
				if($(window).scrollTop() > 100)
				{
					$('.heightmin').css('height',109);
					$("#header").css('position','fixed');
					$("#header").css('z-index','99999');
					$('.logo').addClass('logofixed');
					$(".logo img").attr("src","http://gov.ro/front/view/img/logofixed.png");
					$("#header").addClass("bgtop");
					$('.headerTop').hide();
					$('.headerCenter').css("margin-top", -4);
					$('.menuTop .nav').css('padding-top',0);
					$('.logo a img').css('width', 'auto');
					
					if($(window).width()>=800)
					{
						$('.menuTop .nav').css('padding-top',0);
					}
				}
				else
				{
					$("#header").css('position','relative');
					$("#header").removeClass("bgtop");
					$('.logo').removeClass('logofixed');
					$(".logo img").attr("src","http://gov.ro/front/view/img/logo.png");
					$('.headerTop').show();
					$('.headerCenter').css('margin-top', '0');
					$('.heightmin').css('height',0);
					$('.menuTop .nav').css('padding-top',0);
					$('.logo a img').css('width', 'auto');
					$('.logo').css('top', 27);
					
					if($(window).width()<=800)
					{
						$('.logo').css('top', 47);
						$('.logo a img').css('width', 80);
						$('.menuTop .nav').css('padding-top',20);
					}
					else
					{
						$('.logo a img').css('width', 'auto');
						$('.menuTop .nav').css('padding-top',0);
					}
					
					if($(window).width()<=900)
					{
						$('.logo').css('top', 46);
						$('.logo a img').css('width', 80);
					}
					
					if($(window).width()>=800)
					{
						$('.menuTop .nav').css('padding-top',0);
					}
				}
			});

			function resize()
			{
				
				var window_width = $(window).width();				
				$(window).resize(function()
				{		
					 window_width = $(window).width();	
				});
				
				if(window_width<=800)
				{	
					$('.menuResponsive').unbind().on('click',function(){
						$('.menuTop .nav').fadeToggle();
					});
				}
				else
				{
					hideAll();
					hideAll2();
					showAll();
					$('.menuResponsive .nav').show();
				}
			}	
			
			$(document).ready(function()
			{
				resize();
				$(window).resize(function()
				{						
					resize();
				});	

				showHidden();	

				var firstMeniuElement = 0;
				
				$('.nav_bottom').live('click', function() {	
					
					
				
					var isVisible = '0';
					if($(this).next().hasClass('blockimportant')){
						isVisible = '1';
					}
				
					if(isVisible == '0'){
						hideAll2();
						$(this).next().addClass('blockimportant');
						$(this).next().removeClass('dropMenu');
					}
					else
						{hideAll2();showAll();}
							
					$(this).parent('li').show();
				});
				
				$('ul.nav li.dropdown').hover(function(){
					$(this).addClass('active');
						if(firstMeniuElement === 0){
							$(this).find('.dropMenu').fadeIn();
						}else{
							$(this).find('.dropMenu').show();
						}
						firstMeniuElement = 1;
				}, function(){
					$(this).removeClass('active');
						$('.dropMenu').hide();
				});

				$('.menuTop').mouseleave(function(){
					firstMeniuElement = 0;
				});

				$('#searchSiteForm').submit(function(e){
					var searchText = $('#topSearchInput').val();
					if(searchText.length < 3){
					alert("Va rugăm să introduceți minim 3 caractere!");
					return false;
					}
				});		

			});			

		</script>

<script type="text/javascript">
$(document).ready( function()
{
	var window_width = $(window).width();
	
	var widget_width = 320;
	
	if(window_width<=800)
	{
		widget_width = 265;
	}
	
	if(window_width<600)
	{
		widget_width = 400;
	}
	
	var slider = $('.widget-slider').bxSlider({
		mode: 'horizontal',
		nextSelector: '#slider-next',
		prevSelector: '#slider-prev',
		minSlides: 1,
		infiniteLoop: true,
		hideControlOnEnd: true,
		maxSlides: 3,
		slideWidth: widget_width,
		adaptiveHeight: true,
		pager: false
	});
		
	$('.widget-slider li span').live('click',function()
	{
		var id_widget = $(this).attr('id_widget');		
		
		var items = $('.widget-slider').attr('items');
		item_nr = parseInt(items);
		item_nr--;
		$('.widget-slider').attr('items',item_nr);
		
		if(item_nr<=3)
		{
			$('#slider-next').remove();
			$('#slider-prev').remove();
		}
		
		$(this).parent('.p20').fadeOut('fast', function(){
		
			$(this).remove();
			
			$.ajax(
			{
				url : "http://gov.ro/ro/ajax/widgets/delete?id_widget="+id_widget,
				type: "GET",
				success:function(raspuns)
				{ 
					if(item_nr<=3)
					{
						$('#slider-next').remove();
						$('#slider-prev').remove();						
					}
					
					if(item_nr>=3)
						slider.reloadSlider();
					
					if(item_nr<3)
					{
						$.ajax(
						{
							url : "http://gov.ro/ro/ajax/widgets/add",
							type: "GET",
							success:function(raspuns)
							{
								$('.widget-slider').append(raspuns);
								slider.reloadSlider();
							}
						});
					}
				}
			});
		});
	});
});

</script>
