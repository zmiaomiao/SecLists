{
	$(button).removeClass('active');
	$(menu).hide();

function setTaskbarMenu(button, menu)
{
	$(button).hover(function() {
		taskbarMenuTimeout = setTimeout('showTaskbarMenu("'+button+'", "'+menu+'")', 400);
	}, function() {
		clearTimeout(taskbarMenuTimeout);
		hideTaskbarMenu(button, menu);
	});
	$(menu).hover(function() {
		showTaskbarMenu(button, menu);
	}, function() {
		clearTimeout(taskbarMenuTimeout);

function setShowBalance()
{
	$('.AccDrpCont').mouseover(function() {

			$(this).find('span.loading').removeClass('loading');
			$.get('https://www.nganluong.vn/?portal=nganluong&page=user_info&ajax=get_balance', function(data) {
				if (data) {
					var $obj = $(data).find('.AccDrpCont');

						if (!$obj.find('form').length) {
							var html = $obj.html();

							document.location.href = 'https://www.nganluong.vn/?portal=nganluong&page=home';
						}
					}

function checkTaskbarLogin(){

	}
