    function openSocialNetworkSharingWindow
    (
        event,
        url,
        socialNetworkName,
        width,
        height
    )
    {
        JS_Event_StopPropagation(event);

        var left = (screen.width/2)-(width/2);
        var top = (screen.height/2)-(height/2);
        window.open
        (
            url,
            socialNetworkName,             
            'toolbar=no, location=no, directories=no, status=no, menubar=no, scrollbars=no, resizable=no, copyhistory=no, width='+width+', height='+height+', top='+top+', left='+left
        );
        return false;

    function initCustomEventHadoop() {
        var initCustomEventCookie = ReadCookie("InitCustomEvent");
        if (initCustomEventCookie == null) {
            var BrowserGUID = ReadCookie("BROWSERGUID");
            var userid = ReadCookie("UE");
            if (userid != null && BrowserGUID != null) {

                //Appel à HADOOP
                CallHadoopServer("ue=" + userid, BrowserGUID, "u", "tpo");

                //Création du cookie InitCustomEvent
                CreateCookie("InitCustomEvent", "1", null);
            }
        }
    }

    function setCustomNavigation(url) {
        var BrowserGUID = ReadCookie("BROWSERGUID");
        CallHadoopServer("u=" + encodeURIComponent(url), BrowserGUID, "n", "tpo");
    }

    function CallHadoopServer(data, BGUID, type, applicationID) {

        var url = "http://cs.tracepartsonline.net";
        url += "/" + applicationID + "/" + type;
        url += "/i.gif?bguid=" + BGUID + "&" + data;

        if ($("#HadoopPictureCall") != null) $("#HadoopPictureCall").attr("src", url);

    }

    function setCustomEvent(Category, Action, Label) {

        if(typeof(setCustomEventGoogle)=="function") setCustomEventGoogle(Category, Action, Label);

        //_gaq.push(['_trackEvent', Category, Action, Label]);

        var BrowserGUID = ReadCookie("BROWSERGUID");

        CallHadoopServer("category=" + Category + "&action=" + Action, BrowserGUID, "c", "tpo");
    }

    function ReadCookie(nom) {
        var nom2 = nom + "=";
        var arrCookies = document.cookie.split(';');
        for (var i = 0; i < arrCookies.length; i++) {
            var a = arrCookies[i];

            while (a.charAt(0) == ' ') {
                a = a.substring(1, a.length);
            }
            if (a.indexOf(nom2) == 0) {
                return a.substring(nom2.length, a.length);
            }
        }
        return null;
    }

    function CreateCookie(nom, valeur, jours) {
        // Le nombre de jours est spécifié
        if (jours) {
            var date = new Date();
            // Converti le nombre de jour en millisecondes
            date.setTime(date.getTime() + (jours * 24 * 60 * 60 * 1000));
            var expire = "; expire=" + date.toGMTString();
        }
        // Aucune valeur de jours spécifiée
        else var expire = "";
        document.cookie = nom + "=" + valeur + expire + "; path=/";
    }
