    var cartItems = "";
    function loadCartSum(carNum) {
        cartItems = "";
        TotalCart();
        $('.cart_menu').unbind("click");
        $(".box-items,.box-all").click(function () {
            TotalCart(true);
        });
        $(".min").click(function () {
            var num = $(this).next().val() * 1 - 1;
            if (num < 1)num = 1;
            var scid = $(this).data("scid");
            $(this).next().val(num);
            TotalCart();
        });
        $(".add").click(function () {
            var num = $(this).prev().val() * 1 + 1;
            var scid = $(this).data("scid");
            $(this).prev().val(num);
            TotalCart();
        });
        $(".text_box").blur(function () {
            var num = $(this).val() * 1;
            var scid = $(this).data("scid");
            TotalCart();
        });
        $(".close_span").click(function () {
            var scid = $(this).data("scid");
            $("#check_thing_" + scid).hide(300, function() {
                $(this).remove();
                TotalCart();
            });
            UpdateCart("/ShoppingCart/RemoveCartItem", scid, 0); 
       
        });
        $(".check_thing").hover(function () {
 
            $(this).find(".find_add").show();
        }, function () {
            var num = $(this).data("num");
            var id = $(this).data("id");
            var vnum = $("#sum_" + id).val();
            if (num != vnum) {
                UpdateCart("/ShoppingCart/UpdateShoppingCart", id, vnum);
            }
            $(this).find(".find_add").hide();
        });
    }
    function UpdateCart(url, scid, num) {
        upDown = false;
        $("#CartCheck_" + scid).data("num", num);
        $("#CartCheck_" + scid).attr("checked", true);

        $.post(url, { shoppingCartItemId: scid, shoppingNum: num ,shoppingCartItemIds:scid}, function (data) {
            if (data.success) {
                upDown = true;
            } else {
                alert("该产品超出购买限制");
            }
        });
    }
    function TotalCart(changCart) {
        var total = 0;
        var count = 0;
        cartItems = "";
        var allcount = 0;
        $(".box-items").each(function () {
            var scid = $(this).data("scid");
            var num = $("#sum_" + scid).val() * 1;
 
            var price = $(this).data("scpr") * 1;
            if (this.checked) {
                total += num * price;
                count += num;
                if (cartItems != "")
                    cartItems = (cartItems + ",").replace("," + scid + ",", ",") + scid;
                else
                    cartItems += "," + scid;
            }
            $("#total_" + scid).html(format(num * price));
            allcount += num;
        });
        $(".cart_number").html(count);
        $(".car_price").html(format(total));
        $(".mui_bar").html(allcount);
        $(".c_count").html(allcount);
        if (!changCart) cartItems = "";
        $("#products").val(cartItems);
    }
    $(function () {
        $.ajax({
            url: "/ShoppingCart/MiniCart?"+new Date(),

            success: function (data) {
                $(".cart_menu").html(data);
                loadCartSum(data.sum);
                $("#shopcart_con").css("display", "none");
                TotalCart();
            }
        });
