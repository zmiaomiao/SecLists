    function calculateSavings(monthlyAmount, years, monthlyRate, tax) {

        var s = 0;
        var i = 0;
        var it = 0;

        for (month = 0; month <= 12 * years; month++) {
            i = (s + i) * monthlyRate;
            it = it + i;
            s = s + monthlyAmount;
        }

        return s + it - it * tax - monthlyAmount;
    }

    function instantCalculate(value, e, ForLoc) {
    
 

        value = value.replace(/\s/g, "")
        
        hasDot = value.indexOf(".")

        if (hasDot <= 0) {
            value = CommaToDot(value);
        }

        var keycode;
        if (window.event) {
            keycode = window.event.keyCode;
        }
        else if (e) {
            keycode = e.which;
        }
        if (keycode == 13) {
            window.location = document.getElementById('updateLink').href;
            return false;
        }
        if (isNaN(value)) {
            $('#tbAmountForeignCurrency').val(0);
            $('#tbAmountLocalCurrency').val(0);
            return;
        }
        if (value > 100000000) {
