						$(document).ready(function() {
							var dropdown_show_speed = 250;
							var dropdown_hide_speed = 100;
							var dropdown_hide_delay = 50;
							var dropdown_hide_grace_delay = 50;
							var tag_limit = 12;

							var debug = false;

							var _logo = $(".module.banner.badge");
							var pos_min_x = _logo.length ? _logo.offset().left + _logo.outerWidth() : 0;
							var _header = $(".wheader");
							var pos_y = _header.length ? _header.offset().top + _header.outerHeight() : 0;

							var dropdowns_all = [];
							var dropdowns_shown = [];

							var dropdown = function() {
								return {
									dropdown_container: $(".top-nav-dropdown-container"),
									li: null,
									cat: null,
									dropdown: $("<div style=\"display: none;\"><div class=\"tags\"></div></div>"),
									locked_show: true,
									locked_hide: true,
									locked_hide_grace: true,
									hide_timeout: null,
									hide_grace_timeout: null,

									init: function(li) {
										this.li = $(li);
										this.cat = this.li.attr("data-category");
										if (!this.cat) return;
										debug && console.log("starting %s", this.cat);

										this.dropdown.find(".tags").addClass(this.cat);

										var url = "/solo/slide/tag/cloud?display_as_list=yes&sort_by=size&reverse_sort=yes";
										url += "&restrict_to_category=" + this.cat;
										url += "&limit=" + tag_limit;
										if (window.location.search.match(/no-cache=1/)) {
											url += "&no-cache=1";
										}

										$.get(url, function(d) {
											debug && console.log("Got data for %s, trying to finish", this.cat);
											this.dropdown.find(".tags").append(d);
											this.finish();
										}.bind(this));
									},

									finish: function() {
										debug && console.log("Setting data for %s", this.cat);
										this.dropdown_container.append(this.dropdown);
										this.li.hover(function(event) {
											debug && console.log("Hover event on %s", this.cat);
											this.check_shown();
											if (event.type === "mouseenter") {
												this.show();
											} else {
												this.hide();
											}
										}.bind(this));
										this.dropdown.hover(function(event) {
											debug && console.log("Hover event on %s", this.cat);
											this.check_shown();
											if (event.type === "mouseenter") {
												this.cancel_hide();
												// this.show();
											} else {
												this.hide();
											}
										}.bind(this));
										this.locked_show = false;
										this.locked_hide = false;
										this.locked_hide_grace = false;
										dropdowns_all.push(this);
									},

									check_shown: function() {
										var really_shown = [];
										for (var i = 0; i < dropdowns_all.length; i++) {
											if (dropdowns_all[i].dropdown.is(":visible")) {
												really_shown.push(dropdowns_all[i]);
											}
										}
										dropdowns_shown = really_shown;
									},

									show: function() {
										if (this.locked_show) return;
										this.locked_show = true;
										this.cancel_hide();
										debug && console.log("Showing %s", this.cat);
										var li_center = this.li.offset().left + (this.li.outerWidth() / 2);
										var immediate = dropdowns_shown.length ? true : false;
										if (immediate) {
											debug && console.log("Other categories are shown, hiding them right now");
											var hidden_cats = 0;
											for (var i = 0; i < dropdowns_shown.length; i++) {
												if (dropdowns_shown[i] === this) continue;
												debug && console.log("Force hide %s", dropdowns_shown[i].cat);
												dropdowns_shown[i].hide(true);
												hidden_cats++;
											}
											dropdowns_shown = [];
											if (hidden_cats === 0) immediate = false;
										}
										this.dropdown
											.css({
												position: "fixed",
												top: pos_y,
												left: Math.max(pos_min_x, li_center - (this.dropdown.outerWidth() / 2))
											})
											.fadeIn({
												duration: immediate ? 0 : dropdown_show_speed,
												complete: function() { this.locked_show = false; }.bind(this),
												queue: "show"
											}).dequeue("show");
										dropdowns_shown.push(this);
									},

									hide: function(force_hide) {
										this.locked_hide_grace = true;
										if (this.hide_grace_timeout) window.clearTimeout(this.hide_grace_timeout);
										this.hide_grace_timeout = window.setTimeout(function(force_hide) {
											this.locked_hide_grace = false;
											this.hide_grace_timeout = null;
											var no_delay = this.locked_show;
											if (force_hide) {
												no_delay = true;
												this.cancel_hide();
											}
											if (this.locked_hide) return;
											this.locked_hide = true;
											var immediate = force_hide || (dropdowns_shown.length > 1 ? true : false);
											debug && console.log("Hiding %s, force: %s, now: %s", this.cat, force_hide, immediate);
											this.hide_timeout = window.setTimeout(function(immediate, dont_remove) {
												debug && console.log("Really hiding %s, now: %s, dont_remove: %s", this.cat, immediate, dont_remove);
												this.cancel_show();
												this.dropdown.fadeOut({
													duration: immediate ? 0 : dropdown_hide_speed,
													complete: function() { this.locked_hide = false; }.bind(this),
													queue: "hide"
												}).dequeue("hide");
												this.hide_timeout = null;
												if (dont_remove) return;
												debug && console.log("marking %s as hidden", this.cat);
												var idx = dropdowns_shown.indexOf(this);
												if (idx > -1) dropdowns_shown.splice(idx, 1);
											}.bind(this, immediate, force_hide), no_delay ? 0 : dropdown_hide_delay);
										}.bind(this, force_hide), force_hide ? 0 : dropdown_hide_grace_delay);
									},

									cancel_show: function() {
										if (!this.locked_show) return;
										this.dropdown.stop("show");
										this.locked_show = false;
									},

									cancel_hide: function() {
										if (!(this.locked_hide || this.locked_hide_grace)) return;
										this.dropdown.stop("hide");
										if (this.hide_timeout) {
											window.clearTimeout(this.hide_timeout);
											this.hide_timeout = null;
										}
										if (this.hide_grace_timeout) {
											window.clearTimeout(this.hide_grace_timeout);
											this.hide_grace_timeout = null;
										}
										this.locked_hide = false;
										this.locked_hide_grace = false;
									}
								};
							};

							var lis = $(".top-nav-categories li");
							for (var i = 0; i < lis.length; i++) {
								dropdown().init(lis[i]);
							}
						});
