	function initCookie( name ) {
	  var value = "";
	  var today = new Date();
	  today.setDate( today.getDate() -1);		
	  document.cookie = name + "=" + escape( value ) + "; domain=kbstar.com; path=/; expires=" + today.toGMTString() + ";";
	}

  function getCookie(name){
     var i,x,y,ARRcookies=document.cookie.split(";");
//alert(document.cookie); alert("getCookiename ["+name+"]");
     for(i=0;i<ARRcookies.length;i++){
       x = ARRcookies[i].substr(0,ARRcookies[i].indexOf("="));
       y = ARRcookies[i].substr(ARRcookies[i].indexOf("=")+1);
       x = x.replace(/^\s+|\s+$/g,"");
       if (x==name){
         return unescape(y);
       }
     }
  }

    function getvalDate(){
	    var now = new Date();
	    year = now.getYear();
	    if((year + "").length <4) year = year + 1900;
	    month = now.getMonth() +1;
	    if((month + "").length <2) month = "0" + month;
	    date = now.getDate();
	    if((date + "").length <2) date = "0" + date;
	    today = year+""+month+""+date;
	    return today;
    }
    
  function getCookieVal(offset){
    var endstr = document.cookie.indexOf (";", offset);
    if (endstr == -1) endstr = document.cookie.length;
    return unescape(document.cookie.substring(offset, endstr));
  }

function tabmenu(select_id, selector, showType) {
	//tab
//	alert($('.btn_toggle'));
	var tab_id = $(select_id);
	var tab_cont = tab_id.children(selector);
	var $container = $('.cont');
	var tab_height = $('>li>a', tab_cont).height();
	var startNum = 0;
	//alert(tab_height);
	tab_cont.removeClass('jsOff');
	$('>li', tab_cont).addClass('off');
	if(showType != undefined){
		if(showType == 'random'){
			var totNum = $container.size();
			startNum = Math.floor(Math.random()*totNum);
		} else {
			startNum = parseInt(showType)-1;
		}
	}
	$('>li',tab_cont).eq(startNum).removeClass('off').addClass('on');
	
	tab_cont.css('height', tab_cont.find('>li>div:visible').height()+tab_height);
	if($('>li', tab_cont).find('img').size() > 0) {
		$('>li:first-child>a>img',tab_cont).attr('src', function() {
			return this.src.replace('.gif', '_on.gif');
		});
	} else {}
	
	$('>li>a', tab_cont).bind('click focusin', function(ev) {
		var t = $(this);
		if(t.next('div').is(':hidden')) {// 펼치기
			t.parent('li').removeClass('off').addClass('on');
			tab_cont.css('height', t.next('div').height()+tab_height);
			t.find('img').attr('src', function() {
				return this.src.replace('.gif', '_on.gif');
			});
			
			t.parent().siblings().removeClass('on').addClass('off');
			t.parent().siblings().children('a').find('img').attr('src', function() {
				return this.src.replace('_on.gif', '.gif');
			});
			setTimeout('tabmenu()',250);
		} else {//접기
			
		}
		return false
	});
}
