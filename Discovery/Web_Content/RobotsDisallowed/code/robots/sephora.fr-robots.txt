function buildAddToCartIframeLink(index,sender) {
    var hrefValue = '/checkout/iframe_basket.jsp';
    hrefValue +=';jsessionid=' + 'F20EC35B9D6A354ACED4C65C665D650F.wfr2j';
    var res = true;
    hrefValue += '?dcs_action=additemtocart';
    if (document.getElementById('skuId' + index).value) {
        hrefValue += '&url_catalog_ref_id=' + document
        .getElementById('skuId' + index).value;
    }
    else {
        res = false;
    }
    hrefValue += '&url_product_id=' + document
        .getElementById('productId' + index).value;
    if (isNumeric(document.getElementById('skuQuantity' + index).value)) {
        
        
        if(document.getElementById('skuQuantity' + index).value != 0) {
            document.getElementById('errorMessage'+ index).style.display='none';
            hrefValue += '&url_quantity=' + document
                    .getElementById('skuQuantity' + index).value;
            //Dummy random param to prevent browsers from using their cache
            hrefValue += '&randomJsCount=' + Math.random();
            document.getElementById('skuAdd' + index).href = hrefValue;
            sender=hrefValue;
        }else{
            document.getElementById('errorMessage'+ index).style.display='none';
            document.getElementById('errorMessageBis'+ index).style.display='block';
            res = false;
            
        }
        
    }
    else {
        //on gere l'affichage d'un message d'erreur
        // TO DO affichage de la DIV d'erreur
        document.getElementById('errorMessage'+ index).style.display='block';
        document.getElementById('errorMessageBis'+ index).style.display='none';
        res = false;
    }
    // TO DO controle sur la quantité avec le nombre max authorisé pour la marque
    // TO DO affichage de la DIV d'erreur
    return res;

}
