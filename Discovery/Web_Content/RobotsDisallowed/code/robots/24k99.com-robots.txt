        /*
        这个方法是每次滚动一张图片。
        参数设置：
        diva:内容div
        divb:空div
        ScrollDiv1:外层div
        speed:频率 毫秒
        space:每次移动的像素
        imgnum:几张图片
        */



        var div1, div2, ScrollDiv, Speed, Space, Num, PageWidth, fill, MoveLock = false, MoveTimeObj, Comp = 0, type = 0, AutoPlayObj = null;
        function ScrollImg(diva, divb, ScrollDiv1, speed, space, imgnum) {
            Speed = speed;
            Space = space;
            Num = imgnum;
            div1 = diva;
            div2 = divb;
            ScrollDiv = ScrollDiv1
            GetObj(div2).innerHTML = GetObj(div1).innerHTML;
            PageWidth = GetObj(div1).offsetWidth;
            fill = GetObj(div1).offsetWidth / Num; //一次滚动的距离
            GetObj(ScrollDiv).scrollLeft = 0;
            AutoPlay();
            GetObj(ScrollDiv).onmouseover = function () { clearInterval(AutoPlayObj); clearInterval(MoveTimeObj); }
            GetObj(ScrollDiv).onmouseout = function () {
                if (type == 1) {
                    ISL_GoDown();
                } else { ISL_GoUp(); }
            }
        }

        function GetObj(objName) { if (document.getElementById) { return eval('document.getElementById("' + objName + '")') } else { return eval('document.all.' + objName) } }
        function AutoPlay() { //自动滚动 向左
            clearInterval(AutoPlayObj);
            clearInterval(MoveTimeObj);
            type = 1;
            AutoPlayObj = setInterval('ISL_GoDown();', 3500); //间隔时间
        }
        function AutoPlay1() { //自动滚动  向右
            clearInterval(AutoPlayObj);
            clearInterval(MoveTimeObj);
            type = 2
            AutoPlayObj = setInterval('ISL_GoUp();', 3500); //间隔时间
        }
        function ISL_GoUp() { //上翻开始
            type = 2;
            clearInterval(AutoPlayObj);
            clearInterval(MoveTimeObj);
            if (Comp > 0) {
                Comp = Comp - fill;
            }
            MoveTimeObj = setInterval('ISL_ScrUp();', Speed);
        }
        function ISL_ScrUp() { //上翻动作  向右
            if (GetObj(ScrollDiv).scrollLeft <= 0) { GetObj(ScrollDiv).scrollLeft = GetObj(ScrollDiv).scrollLeft + GetObj(div1).offsetWidth }
            if (Comp + fill > 0) {
                GetObj(ScrollDiv).scrollLeft -= Space; Comp--;
            } else {
                Comp = 0;
                clearInterval(MoveTimeObj);
                AutoPlay1()
            }
        }
        function ISL_GoDown() { //下翻  向左
            type = 1;
            clearInterval(AutoPlayObj);
            clearInterval(MoveTimeObj);
            if (Comp < 0) {
                Comp = Comp + fill;
            }
            MoveTimeObj = setInterval('ISL_ScrDown()', Speed);
        }
        function ISL_ScrDown() { //下翻动作
            if (GetObj(ScrollDiv).scrollLeft >= GetObj(div1).scrollWidth) { GetObj(ScrollDiv).scrollLeft = GetObj(ScrollDiv).scrollLeft - GetObj(div1).scrollWidth; }
            if (Comp < fill) {
                GetObj(ScrollDiv).scrollLeft += Space; Comp++;
            } else {
                Comp = 0;
                clearInterval(MoveTimeObj);
                AutoPlay()
            }
        }
