function getCountdownFormat(countData, saleType) {
  var untilType = new Date(countData);
  var time = untilType.getTime() - (new Date).getTime();
  var hours = time / 3600000;
  var dateFormat = "dhms";
  if (saleType == "upcoming" || saleType == "upcomingMap") {
    dateFormat = "ds";
    if (hours < 24) {
      dateFormat = "h";
    }
  } else if (hours < 48) {
    dateFormat = "hms";
  }
  return dateFormat;
}
