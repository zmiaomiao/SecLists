<script id="script-jquery-stickyElement">
    (function ($) {

        var methods = {
            defaults: {
                trackEl: null,
                distance: 50,
                stuckClass: 'is-sticky',
                suppressed: false
            },
            options: { },
            init: function(options) {
                var self = methods;

                $.extend(true, methods.options, methods.defaults, {element: this}, options);

                return this.each(function() {
                    var $this = $(this)
                            , data = $this.data('stickyElement');

                    // If the plugin hasn't been initialized yet
                    if (! data) {
                        $(this).data('stickyElement', methods.options);
                    }

                    self._bindScroll();
                });
            },
            _getOps : function(opt) {
                // Let's always get the options from the element data (not the options object)
                var $this = $(methods.options.element)
                        , data = $this.data('stickyElement');

                // If we are looking for a specific option, let's provide a shortcut
                if (opt != null) {
                    data = data[opt];
                }

                return data;
            },
            _bindScroll: function() {
                var self = methods;
                var options = self._getOps(),
                        $element;

                $element = options['element'],

                        $(window).scroll(function () {
                            self.showOrHide();
                        });
            },
            showOrHide: function() {
                var self = methods;
                var options = self._getOps(),
                        $element;

                $element = options['element'];
                $trackedElement = jQuery(options['trackEl']);

                if(!options['suppressed']) {
                    if(options['trackEl'] !== null && $trackedElement.length > 0) {
                        options['distance'] = $trackedElement.offset().top + $trackedElement.height()
                    }

                    if ($(window).scrollTop() > options['distance']) {
                        self.show();
                    } else {
                        self.hide();
                    }
                }
            },
            show: function() {
                var self = methods;
                var options = self._getOps(),
                        $element;

                $element = options['element'];

                $element.addClass(options['stuckClass']);
            },
            hide: function() {
                var self = methods;
                var options = self._getOps(),
                        $element;

                $element = options['element'];

                $element.removeClass(options['stuckClass']);
            },
            suppress: function() {
                var self = methods;
                self.options.suppressed = true;

                $(this).data('stickyElement', self.options);

            },
            resume: function() {
                var self = methods;
                self.options.suppressed = false;

                $(this).data('stickyElement', self.options);
            }
        }

        /* stickyElement PLUGIN DEFINITION
         * =========================== */

        $.fn.stickyElement = function(methodOrOptions) {
            if ( methods[methodOrOptions] ) {
                return methods[ methodOrOptions ].apply( this, Array.prototype.slice.call( arguments, 1 ));
            } else if ( typeof methodOrOptions === 'object' || ! methodOrOptions ) {
                // Default to "init"
                return methods.init.apply( this, arguments );
            } else {
                $.error( 'Method ' +  method + ' does not exist on jQuery.stickyElement' );
            }
        };

    }(window.jQuery));
</script>
