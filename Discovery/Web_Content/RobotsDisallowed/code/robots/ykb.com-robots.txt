                function requireImage(image) { 
                    var images = typeof (image) == "string" ? [image] : image;
                    var requireds = [];

                    function _preloadImage(url) {
                        var dfr = jQuery.Deferred();
                        var img = new Image();
                                                         
                        function unbind() {
                            img.onload = null; 
                            img.onabort = null;
                            img.onerror = null;
                        }
                        img.onload = function () {
                            unbind();
                            dfr.resolve();
                        };

                        img.onerror = img.onabort = function () {
                            unbind();
                            dfr.reject();
                        };

                        img.src = url;
                        return dfr;
                    }

                    $.each(images, function () {
                        requireds.push(_preloadImage(this));
                    });

                    return requireds.length > 1 ? requireds : requireds[0];
                }
