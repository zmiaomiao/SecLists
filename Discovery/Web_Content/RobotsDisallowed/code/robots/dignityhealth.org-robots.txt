function equalHeight(group) {
	tallest = 0;
	$(group).css("height", "auto");
	group.each(function() {
	thisHeight = $(this).outerHeight();
		if(thisHeight > tallest) {
			tallest = thisHeight;
		}
	});
	group.outerHeight(tallest);
}
