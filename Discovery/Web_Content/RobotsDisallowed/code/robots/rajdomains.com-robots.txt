  function subscribeMailChimp(){
	var firstname = '';
	var lastname = '';
	if($("#firstname").val() != ''){
		firstname = $("#firstname").val();
	}
	if($("#lastname").val() != ''){
		lastname = $("#lastname").val();
	}
	$("#sloader").show();
	$.ajax({
			url: 'index.php?route=common/footer/mailchimpSubscribe',
			type: 'POST',
			data: 'email=' + $("#email").val() + '&firstname=' + firstname + '&lastname=' + lastname,

				}

					$("#sloader").hide();
					$("#alert").html(data);
					$("#alert").show(300);
				}
			}
		});
  }
