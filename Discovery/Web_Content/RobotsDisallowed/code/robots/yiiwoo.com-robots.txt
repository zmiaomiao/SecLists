function eventBind()
{
	$(".mouseButNoMV").hover(function(){		
		$(this).find(".img").css("top",-1*$(this).height()+"px");		
	},function(){
		$(this).find(".img").css("top","0px");

		$(this).find(".img").stop().animate({

		   top: 0
		}, 300);

		}, 300);
		$(this).find(".txt").stop().animate({
		   top: 280
		}, 300);
		$(this).find(".line").stop().fadeOut();
	});
	
}
