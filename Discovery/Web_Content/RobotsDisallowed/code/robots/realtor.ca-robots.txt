        function startSearch(){

            // Invalidate if the user did not input text
            if( $("#m_hdr_glb_search_input").pVal() === '' )
            {
                //redirectToDisclaimer('/Map.aspx');
                getGeoLocated_EmptySearch();
                return false;
        
            }

            if (location.href.toLowerCase().indexOf('map.aspx') > -1) {
                HashSearch.remove('Area');
                HashSearch.set('Area', $('#m_hdr_glb_search_input').val());
                $('#m_hdr_glb_search_input').val('');
            } else {

                var areaSearch = new SubAreaSearch();
                areaSearch.input = $('#m_hdr_glb_search_input').val(); // INPUT TEXTBOX
                areaSearch.searchType = 'BASIC'; // BASIC || FILTERS
                areaSearch.searchCriteriaCommon += 'CultureId=' + ApplicationSettings.CultureID;
                areaSearch.searchCriteriaCommon += '&ApplicationId=' + ApplicationSettings.ApplicationId;
                areaSearch.searchCriteria = '&Area=' + encodeURIComponent($('#m_hdr_glb_search_input').val());
                areaSearch.Search();
                $('#m_hdr_glb_search_input').val('');
            }

            //$('#m_hdr_glb_search_input').focus();

            return false;

        }
