		if (window.addEventListener) {
			window.addEventListener(eventName, handler);
		} else {
			window.attachEvent('on' + eventName, function(evt) {				
				evt = evt || window.event;
				evt.target = evt.target || evt.srcElement;
				handler(evt);

	};

	var registerOccurred = function (eventName) {
		addEventListener(eventName, function () {
			occurredEvents.push(eventName);
		});
	};

	/* These are all the events you can require (every event has to occur on the window element) */
	registerOccurred('gamesys.events.mainReady');

	registerOccurred('gamesys.events.myGameReady');
	registerOccurred('gamesys.events.resizeVeediReady');

	expose.require = function (requiredEvent, handler) {

		/* Maybe the event already occurred, execute immediately */
		if (occurredEvents.indexOf(requiredEvent) > -1) {
			handler();
			return;
		}

		/* Or the event hasn't occurred yet, create a listener for when it does */
		addEventListener(requiredEvent, handler);
	};

	return expose;
})();

googletag.cmd.push(function () {
	/* This event only fires when ad is RENDERED, it doesn't when there is no ad received for a slot */
	googletag.pubads().addEventListener('slotRenderEnded', function(event) {
		var slotName = event.slot.getName().replace(/\/[0-9]+\//, '');

		if(event.isEmpty){
			return;
		}

		var	width = event.size[0],
			height = event.size[1];

		/* Do something with every adunit here. We loop through our adunits to see if the adunit.name
		 * matches the slotName. */
		for (var name in gamesys.adunits) {
			if (slotName === gamesys.adunits[name].name) {
				gamesys.requireEvent.require('gamesys.events.jsysReady', function () {
					var ad = document.getElementById('div-gpt-' + name);

					if (ad !== null) {
						$$.addClass(ad, 'loaded');

						if (gamesys.adunits[name].width != width || gamesys.adunits[name].height != height) {
							/* Unexpected size, adjust the size of the ad */
							ad.style.width = width + 'px';
							ad.style.height = height + 'px';

							/* Remove now unnecessary min-width and min-height */
							ad.style.minWidth = 0;
							ad.style.minHeight = 0;

							/* Adjust the size of the label, if there is one */
							if ($$.hasClass(ad, 'leaderboard-label')) {
								$$.addClass(ad, 'leaderboard-label-' + height);
							}
						}
					}
				});

				gamesys.adunits[name].slotRendered = true;
				break;
			}
		}

		if (slotName === 'GP-Right') {
			/* Resize gamepage when a different size than expected is received */
			if (width != gamesys.adunits.page_right.width) {
				gamesys.nonGameWidth = gamesys.nonGameWidth - gamesys.skyscraperWidth + width;
				gamesys.adContainerWidth = gamesys.adContainerWidth - gamesys.skyscraperWidth + width;

				document.getElementById('js-gamepage-container').style.width = gamesys.adContainerWidth + 'px';

				gamesys.requireEvent.require('gamesys.events.resizeGamepageReady', function () {
					gamesys.resizeGamepage();
				});
			}

			/* Show highest rated games for gamesfreak below the rectangle */
			if (width == 300 && gamesys.site.id == 40) {
				gamesys.requireEvent.require('gamesys.events.mainReady', function () {
					GAMESYS.gamesBelowRectangle();
				});
			}
		}
		else if (slotName === 'GP-Left') {
			/* Resize gamepage when a different size than expected is received */
			if (width != gamesys.adunits.page_left.width) {
				gamesys.nonGameWidth = gamesys.nonGameWidth - gamesys.skyscraperWidth + width;
				gamesys.adContainerWidth = gamesys.adContainerWidth - gamesys.skyscraperWidth + width;

				document.getElementById('js-gamepage-container').style.width = gamesys.adContainerWidth + 'px';

				gamesys.requireEvent.require('gamesys.events.resizeGamepageReady', function () {
					gamesys.resizeGamepage();
				});
			}
		}
		else if (slotName === 'GP-728x90-ATF-2' && width > 0) {
			document.getElementById('div-gpt-game_leaderboard_atf_2').style.width = width + 'px';
		}
		else if (width > 950) {
			for (var name in gamesys.adunits) {
				if (slotName === gamesys.adunits[name].name) {
					$$.addClass(document.getElementById('div-gpt-'+name), 'giant-ad');
				}
			}
		}

		if(typeof myGame != "undefined" && myGame != null) {
			myGame.alignLeaderboards();
		}
	});
});
