function Header(){
    this.modified = false;

    var main_content = $("#main-content");
    var main_brand = $("#main_brand");
    var header = $("#header");

    this.modify = function(){
        this.modified = true;
        this.original_data.logo_top_margin = $(logo).css('margin-top');
        this.original_data.search_icon_color = '#FDFDFD';
        $(search_icon).css('color', '#0E1D2E');
        this.original_data.input_border_color = $(input).css('border-color');

        this.original_data.input_id = $(input).prop('id');
        $(input).prop('id', 'inline_search_text_box2');
        this.original_data.main_brand_src = $(main_brand).prop('src');
                    $(main_brand).prop('src', 'http://cdn03.jobs.id/assets/images/logo-v3-dark.png')
                this.original_data.main_brand_width = $(main_brand).css('width');
        $(main_brand).css('width', '135px');
        this.original_data.header_add_class = 'headerbg';
        $(header).removeClass('headerbg');
        this.original_data.header_border = $(header).css('border-bottom');
        $(header).css('border-bottom', '1px solid rgba(44, 62, 80, .28)');
        this.original_data.class = 'in_large_header';
        $(list).addClass('in_large_header');
        $(header).css('position','absolute');
        $(header).hide();
                setTimeout(function(){
            $(header).fadeIn(300);
        }, 100);
    };

    this.unmodif = function(){
        this.modified = false;
        $(search_icon).css('color', this.original_data.search_icon_color);
        $(input).css('border-color', this.original_data.input_border_color);
        $(input).css('color', this.original_data.input_id_color);
        $(input).prop('id', this.original_data.input_id);

        $(header).css('border-bottom', this.original_data.header_border);
        $(header).css('position','fixed');
        $(list).removeClass(this.original_data.class);
        $(header).hide();

            $(header).fadeIn(300);
        }, 100);
    }
