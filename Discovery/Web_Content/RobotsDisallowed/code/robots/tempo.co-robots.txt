var loadmore_count=0;

function loadmore(idnews){
	loadmore_count++;
	var class_id = $("#loadmore");
	class_id.html('<img src="'+cdn+'/images/tempo-loading-50.gif">');		
	$.ajax({  
		type: "GET", 
		jsonp: "callback",
		dataType: "jsonp",
		url: "//m.tempo.co/rest/berita/terkini.jsonp",
		data: { limit: 12, last_idnews: idnews},
		success: function( msg ) {				
			var arr = [];
			$.each(msg, function(key, v){
				var img = 0;
				idnews = v.idnews;
				if(typeof v.images != 'undefined'){
					$.each(v.images, function(key2, v2){
						if(key2=='150') {
							img = v2;
						}else if(img==0 && key2=='default'){
							img = v2+'&width:150';
						}
					});
				}
				
				var gambar = (img==0 ? '':'<div class="box-gambar"><a href="'+v.subdomain+'"><img src="'+img+'" class="img-150" style="height: auto;"></a></div>');
				
				if(typeof v.subkanal != 'undefined'){
					subkanal = '<h5 class="kanal"><a href="'+v.subkanal_url+'">'+v.subkanal+'</a></h5>';
				}else{
					subkanal='';
				}
				
				arr.push('<li>'+
							gambar+
							'<div class="box-text">'+subkanal+
							'<h5 class="waktu">'+v.updated1+'</h5>'+
							'<h3><a href="'+v.subdomain+'">'+v.judul+'</a></h3>'+
							'<p>'+v.deskripsi+'</p>'+
							'</div>'+
						'</li>');
			});
			class_id.before(arr.join(' '));
			if(loadmore_count>2){
				class_id.html('<button class="indikator fade" onclick="window.location.assign(\'//www.tempo.co/indeks\')">Berikutnya</button>');
			}else{
				class_id.html('<button class="indikator fade" onclick="loadmore('+idnews+')">Berikutnya</button>');
			}
		}
	});
}
