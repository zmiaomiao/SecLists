  <script type="text/javascript">
    (function() {
      var $craneMoveHorizontal = $(".crane-set .crane-container");
      var $craneMoveVertical = $(".crane-set .crane-container .crane-cable");
      var $tShadow = $(".logo-t-shadow");
      var $win = $(window);
      var widthLimit = 1440;
      var isAnimationComplete = false;
      var xPos, yPos, shadowPos, prevSize;

      var positions = {
        yPosNormal : 120,
        yPosSmall : 101,
        shadowPosNormal : 52,
        shadowPosSmall : 27
      }

      var isSizeChanged = false;
      function moveUpDown() {
        var dur = 1000;

        $craneMoveVertical.animate({"height" : yPos + "px"}, {
          duration : dur,
          delay: 2000,
          complete : function() {
            isAnimationComplete = true;

            $craneMoveHorizontal.addClass("animation-complete");
            $craneMoveVertical.removeAttr("style"); // removed inline style attribute so that main CSS styles can take over position

          }
        })

        $tShadow.animate({"bottom" : shadowPos + "px", "opacity" : 1}, {
          duration : dur,
          delay: 2000,
          complete : function() {
            isAnimationComplete = true;

            $tShadow.removeAttr("style"); // removed inline style attribute so that main CSS styles can take over position
          }
        });
      }

      function setPositions() {
        var isNormalWidth = $win.width() > widthLimit;

        yPos = (isNormalWidth) ? positions.yPosNormal : positions.yPosSmall;
        shadowPos = (isNormalWidth) ? positions.shadowPosNormal : positions.shadowPosSmall;

        if (prevSize !== undefined && prevSize !== yPos) {
          isSizeChanged = true;
          console.log('changed');

          winSizeChanged();
        }

        prevSize = yPos;
      }

      function winSizeChanged() {
        stopMoveUpDown();

        // Restart Animation();
        moveUpDown();
      }

      function stopMoveUpDown() {
        $craneMoveVertical.stop();
        $tShadow.stop();
      }

      $win.resize(function() {
        if (!isAnimationComplete) {
          setPositions();
        }
      });

      // Init
      $(function() {
        setPositions();
        moveUpDown();
      });
    })();
  </script>
