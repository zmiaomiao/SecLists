function submitJobSearch(formJobSearch) {
var region = formJobSearch.elements['lieux'].value;
var libMetier = formJobSearch.elements['libMetier'].value;
var contrat = formJobSearch.elements['typeContrat'].value;
var debrayage = formJobSearch.elements['offresPartenaires'];
if (region == '01P') {
if (libMetier.length == 0) {
alert('Vous devez cibler davantage votre lieu de travail ou préciser l\'emploi recherché.');
return false;
} else {
if (libMetier.length < 3) {
alert('Le mot clé saisi doit contenir au moins 3 lettres.');
return false;
}
}
}
var newurl = '';
if (region == '0' && libMetier == '' && contrat == '0') {
newurl = 'http://offre.pole-emploi.fr/recherche';
} else {
newurl = 'http://offre.pole-emploi.fr/resultat';
}
var first = true;
if (region != '0') {
if (first) {
newurl = newurl + '?';
first = false;
} else {
newurl = newurl + '&';
}
newurl = newurl + 'lieux=' + region;
}
if (libMetier != '') {
if (first) {
newurl = newurl + '?';
first = false;
} else {
newurl = newurl + '&';
}
/* JSA : 20130506 : Sur demande coté Tapestry, plus besoin d'encoder le libMetier */
newurl = newurl + 'libMetier=' + libMetier;
}
if (contrat != '0') {
if (first) {
newurl = newurl + '?';
first = false;
} else {
newurl = newurl + '&';
}
newurl = newurl + 'typeContrat=' + contrat;
}
/* Permet de rajouter le paramètre pour indiquer que l'on veut aussi les partenaires de pole emploi dans la recerche d'offres */
if (debrayage != null) {
if (first) {
newurl = newurl + '?';
first = false;
} else {
newurl = newurl + '&';
}
/* JSA : 20130506 : Le bon test pour savoir si la case à cocher est coché c'est debrayage.checked et non debrayage.value */
if (debrayage.checked == true) {
newurl = newurl + 'offresPartenaires=true';
} else {
newurl = newurl + 'offresPartenaires=false';
}
}
window.open(newurl);
return false;
}
