		 <script type="text/javascript">
			paper.install(window);

			var distanceBetween = function (p1, p2) {
				if (!p1.x || !p2.x || !p1.y || !p2.y) return null;

				return Math.sqrt((p2.x - p1.x) * (p2.x - p1.x) + (p2.y - p1.y) * (p2.y - p1.y));
			};

			var angleBetween = function (p1, p2) {
				if (!p1.x || !p2.x || !p1.y || !p2.y) return null;

				return Math.atan2((p2.y - p1.y), (p2.x - p1.x));
			};

			window.onload = function() {
                var RADIUS_MIN = 95,
                    RADIUS_MAX = 105,
                    SPRING = 0.25,
                    FRICTION = 0.57;

				var hidden = true;

				var canvas = document.querySelector('#map canvas');
				paper.setup(canvas);

				var mouse = {};

				var project = paper.project;

				TweenMax.ticker.addEventListener('tick', project.view.update.bind(project.view));

				project.importSVG('/html/images/map.svg', function (item) {
					var centerX = canvas.width / 2,
						centerY = canvas.height / 2,
						minX = 200,
						minY = 150,
						maxX = canvas.width * 0.7,
						maxY = canvas.height * 0.7;

					item.children.world.opacity = 0;
					item.children.images.opacity = 0;

					var imageLoaded = function (image) {
						setTimeout(function () {
							var angle = angleBetween(image.position, {
								x: centerX,
								y: centerY
							}),
							x = -Math.cos(angle) * (minX + Math.random() * (maxX - minX)) + centerX,
							y = -Math.sin(angle) * (minY + Math.random() * (maxY - minY)) + centerY;

							image.position.x += image.image.naturalWidth / 2;
							image.position.y += image.image.naturalHeight / 2;

							image.originPosition = image.position.clone();
							image.vx = 0;
							image.vy = 0;
							image.opacity = 1 - y / canvas.height;

							TweenMax.to(image.position, 0.6, {
								delay: 0.2,
								x: x,
								y: y,
								ease: Expo.easeOut
							});

							TweenMax.fromTo(image.position, 0.6, {
								x: x,
								y: y
							}, {
								delay: 2.0,
								x: image.originPosition.x,
								y: image.originPosition.y,
								ease: Expo.easeOut
							});

							TweenMax.to(image, 0.6, {
								delay: 2.0,
								opacity: 1
							});
						});
					};

					setTimeout(function () {
						TweenMax.staggerFromTo('.title h1, .title h2', 0.6, {
							y: 10,
							opacity: 0
						}, {
							y: 0,
							opacity: 1
						}, 0.4);

						TweenMax.to(item.children.images, 0.6, {
							opacity: 1
						});

						for (var i = item.children.images.children.length-1; i >= 0; --i) {
							image = item.children.images.children[i];
							imageLoaded(image);
						}
					});

					TweenMax.to('.title', 0.6, {
						delay: 2.1,
						opacity: 0
					});
				});

				setTimeout(function () {
					project.view.on('frame', function () {

						var images = project.activeLayer.children['svg-map'].children.images.children;
						for (var i = images.length-1; i >= 0; --i) {
							var image = images[i],
								imageDistance = distanceBetween(image.position, mouse),
								originDistance = distanceBetween(image.position, image.originPosition),
								angle = angleBetween(image.position, mouse),
								delta = imageDistance - (RADIUS_MIN + ((distanceBetween(image.originPosition, mouse) / RADIUS_MAX) * (RADIUS_MAX - RADIUS_MIN))),
								effect = (!window.oppened && !hidden && imageDistance != null && imageDistance <= RADIUS_MAX);

							var dx = effect ? Math.cos(angle) * delta : image.originPosition.x - image.position.x,
								dy = effect ? Math.sin(angle) * delta : image.originPosition.y - image.position.y;

							image.vx += dx * SPRING;
							image.vy += dy * SPRING;

							image.vx *= FRICTION;
							image.vy *= FRICTION;

							image.position.x += image.vx;
							image.position.y += image.vy;
						}
					});

					var tool = new Tool(),
						child = 1;
					tool.onMouseMove = function(event) {
						if (window.oppened) return hidden = true;

						mouse = event.point;

						var result = project.activeLayer.children['svg-map'].children.world.hitTest(event.point);

						TweenMax.set('#macaron', {
							x: event.point.x,
							y: event.point.y
						});


						if (result) {
							if (hidden) {
								hidden = false;

								TweenMax.to('#macaron', 0.4, {
									opacity: 1
								});
							}
							
							if (window.countryCode !== result.item.name) {
								window.countryCode = result.item.name;

								var text = 'in ';

								switch (window.countryCode) {
									case 'AM':
										text += 'America';
										break;

									case 'AF':
										text += 'Africa';
										break;

									case 'EU':
										text += 'Europe';
										break;

									case 'OC':
										text += 'Oceania';
										break;

									case 'AS':
										text += 'Asia';
										break;
								}

								document.querySelectorAll('.country')[child-1].innerText = text;
								document.querySelectorAll('.country')[child-1].textContent = text;
								
								otherChild = (child === 1) ? 2 : 1;

								TweenMax.fromTo('.country:nth-child('+otherChild+')', 0.4, {
									opacity: 1
								},{
									opacity: 0
								});
								TweenMax.fromTo('.country:nth-child('+child+')', 0.4, {
									opacity: 0
								},{
									opacity: 1
								});
								
								child = (child === 1) ? 2 : 1;
							}
						}
						else if (!hidden) {
							hidden = true;

							TweenMax.to('#macaron', 0.4, {
								opacity: 0
							});
						}
					};
				}, 3000);
			}
