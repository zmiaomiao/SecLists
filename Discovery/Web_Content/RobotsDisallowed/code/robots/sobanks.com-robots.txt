function getKeyDown(val) {
	var tipindex = -1 ;
		var sval = val;
		var event = event ? event : window.event;
		var keyCode = event.keyCode;
		var vschool = $("#intelligent-regName");
		if (keyCode == 40 || keyCode == 38 || keyCode == 13) {
			var fobj;
			if (keyCode == 40) {
				tipindex++;
				if (tipindex == vschool.find("li").length) {
					tipindex = 0;
					vschool.find("li").eq(vschool.find("li").length - 1).css("background-color", "");
				}
				fobj = vschool.find("li").eq(tipindex);
				fobj.css("background-color", "#EEEEEE").siblings().css("background-color", "");
				$("#userEmail").val( fobj.html().replace(/<(\S*?)[^>]*>|<.*? \/>/g, "") );
				return;
			} else if (keyCode == 38) {
				tipindex--;
				if (tipindex <= -1) {
					tipindex = vschool.find("li").length - 1;
					vschool.find("li").eq(0).css("background-color", "");
				}
				fobj = vschool.find("li").eq(tipindex);
				fobj.css("background-color", "#EEEEEE").siblings().css("background-color", "");
				if (fobj.html() != null) {
					$("#userEmail").val(fobj.html().replace(/<(\S*?)[^>]*>|<.*? \/>/g, ""));
				}
				return;
			} else if (keyCode == 13) {
				tipindex = -1;
	
				return false;
			}
		} else {
			$("#morePinDiv").removeClass().addClass("intelligent-error hide");
			if (sval != "") {
				var userinput = sval;
				var oldSval = "";
				var pos = sval.indexOf("@");
				if (pos >= 0) {
					oldSval = sval.substring(0, pos);
				}
				tipindex = -1;
				var html = "";
				if (/[\u4E00-\u9FA5]/g.test(sval) || sval.indexOf("-") > -1 || sval.indexOf("_") > -1) {
					html = "<li>" + sval + "</li>";
				} else {
				
					if (oldSval != '') {
						sval = oldSval;
					}
					if (userinput.indexOf("@") == 0) {
						sval = "";
					}
					html = "<li>" + userinput + "</li>";
					var partSurfix = initEmailSurfixArray(userinput);
					if (partSurfix != null) {
						for (var i = 0; i < partSurfix.length; i++) {
							html += "<li>" + sval + partSurfix[i] + "</li>";
						}
					}
				}
				if (sval.length > 25) {
					$('#intelligent-regName').hide();
				}else {
					$('#intelligent-regName').show();
					$('#intelligent-regName').html(html).find("li").mousedown(function () {
						$("#userEmail").val($(this).html());
						tipindex = -1;
					});
				}
			} else {
				$('#intelligent-regName').html("").hide();
				tipindex = -1
			}
	}

function initEmailSurfixArray(str) {
	var pos = str.indexOf("@");
	if (pos < 0 || pos == (str.length - 1)) {
		return emailSurfixArray;
	}
	var inputSurfix = str.substring(pos, str.length);
	var suitableSurfixArray = [];
	var j = 0;
	for (var i = 0; i < emailSurfixArray.length; i++) {
		if (emailSurfixArray[i].indexOf(inputSurfix) == 0) {
			suitableSurfixArray[j] = emailSurfixArray[i];
			j++;
		}
	}
	return suitableSurfixArray;
}

function adAimate(){
	var adWrap = $( "#adWrap" ),
		time;
		
		
	adWrap.show().stop().animate({
		marginTop : -60,
		scrolltop : 0,
		height:460
	},500,function(){
		adWrap.find( ".maxImgMian" ).animate({bottom:0},500,"easeOutExpo")
	})
	
	
	time = setTimeout(function(){
		adWrap.animate({
			marginTop : 0,
			scrolltop : 400,
			height:60
		},500,function(){
			adWrap.find( ".maxImgMian" ).animate({bottom:0},500,"easeOutExpo")
		})
	},3000)
	
	adWrap.hover(function(e){
		
		e.stopPropagation()
		clearTimeout(time)
		adWrap.stop().animate({
			marginTop : -60,
			scrolltop : 0,
			height:460
		},500,function(){
			adWrap.find( ".maxImgMian" ).animate({bottom:0},500,"easeOutExpo")
		})
	},function(e){
		e.stopPropagation()
		clearTimeout(time)
		adWrap.stop().animate({
			marginTop : 0,
			scrolltop : 400,
			height:60
		},500,function(){
			adWrap.find( ".maxImgMian" ).css({bottom:400})
		})
	})
	
	
}
