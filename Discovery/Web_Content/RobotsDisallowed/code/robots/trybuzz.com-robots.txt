					});
					
					$(document).on('click','.header_form_submit_signup', function(){
						
						$(".header_form_validation").html("");
						
						$.ajax({
						    url: '/ajax_signup.php',
						    type: 'POST',
						    data: {
						        name: $(".header_form_text_name").val()
						        ,mailaddress: $(".header_form_text_mailaddress").val()
						        ,password: $(".header_form_text_password").val()
						    },
						    dataType: 'html'
						})
						.done(function( data ) {
							
							if($(data).find(".error").html()){
								
								$(".header_form_validation_signup_name").html($(data).find(".error_name").html());
								$(".header_form_validation_signup_mailaddress").html($(data).find(".error_mailaddress").html());
								$(".header_form_validation_signup_password").html($(data).find(".error_password").html());
									
							}else{
								
								$(".header_form_submit_signup").css({"display":"none"});
								
								$(".header_black_dialog_mailaddress_box_success").html($(".header_form_text_mailaddress").val() + " に確認メールを送信しました。<br>メール内のURLをクリックして登録を完了させてください。");
								
							}
							
						   	
						})
						.fail(function( data ) {
							alert("通信エラーが発生しました。更新してやり直してください。");
							location.reload();
						})
						.always(function( data ) {
							
						});
						
					});
					
					
					
				});
