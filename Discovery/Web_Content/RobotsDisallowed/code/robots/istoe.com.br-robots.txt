	function proximaFrase(){

		var atual = document.getElementById("atual").value;
		var size = document.getElementById("size").value;
		var idx_img = document.getElementById("idx_img").value;		

		var imgBalao =  document.getElementById("balaoBottom");

		
		if(atual < size) {
			if(imgBalao != null)
	            imgBalao.style.display = 'none';
			for(i = 0; i <= size; i++){
				var div = document.getElementById("div"+i);
				var img = document.getElementById("id_"+i);
				
				if(img != null)
					img.style.display = 'none';
								
				div.style.display = 'none';
			}			
			var divAnterior = document.getElementById("div"+atual);
			atual++;
			var divAtual = document.getElementById("div"+atual);

			idx_img++;
			var img_atual = document.getElementById("id_"+idx_img);

			if(img_atual != null){
				if(document.getElementById("isLarge_"+idx_img).value=="0"){
					img_atual.style.display = 'block';

					if(imgBalao != null)
						imgBalao.style.display = 'block';
					
				}
			}			

			
			
			divAnterior.style.display = 'none';
			divAtual.style.display = 'block';
	
			document.getElementById("atual").value++;
			document.getElementById("idx_img").value++;

			checarFoto();
			
		}				
	}

	function humane_date(date_str){
	    var time = ('' + date_str).replace(/-/g,"/").replace(/[TZ]/g," ");
	    var dt = new Date;
	    var seconds = ((dt - new Date(time) + (dt.getTimezoneOffset() * 60000)) / 1000);
	    var token = 'atrás', list_choice = 1;
	    
	    if (seconds < 0) {
	        seconds = Math.abs(seconds);
	        token = 'de agora';
	        list_choice = 2;
	    }
	    
	    if (seconds < 60)
	        return 'exatamente agora';
	    
	    var i = 0, format;
	    while (format = time_formats[i++]) if (seconds < format[0]) {
	        if (i == 0)
	            return format[1];
	        if (typeof format[2] == 'string')
	            return format[list_choice];
	        else
	            return Math.floor(seconds / format[2]) + ' ' + format[1] + ' ' + token;
	    }
	    
	    // overflow for centuries
	    if (seconds > 5806080000)
	        return Math.floor(seconds / 2903040000) + ' centuries ' + token;

	    //alert(date_str);
	    return date_str;
	};

	var time_formats = [
	    [120, '1 minuto atrás', '1 minute from now'], // 60*2
	    [3600, 'minutos', 60], // 60*60, 60
	    [7200, '1 hora atrás', '1 hour from now'], // 60*60*2
	    [86400, 'horas', 3600], // 60*60*24, 60*60
	    [172800, 'ontem', 'amanhã'], // 60*60*24*2
	    [604800, 'dias', 86400], // 60*60*24*7, 60*60*24
	    [1209600, 'ultima semana', 'next week'], // 60*60*24*7*4*2
	    [2419200, 'semanas', 604800], // 60*60*24*7*4, 60*60*24*7
	    [4838400, 'ultimo mês', 'next month'], // 60*60*24*7*4*2
	    [29030400, 'meses', 2419200], // 60*60*24*7*4*12, 60*60*24*7*4
	    [58060800, 'ultimo ano', 'próximo ano'], // 60*60*24*7*4*12*2
	    [2903040000, 'anos', 29030400], // 60*60*24*7*4*12*100, 60*60*24*7*4*12
	    [5806080000, 'last century', 'next century'], // 60*60*24*7*4*12*100*2
