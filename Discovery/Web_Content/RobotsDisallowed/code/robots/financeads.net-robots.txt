	function load_blognews()
	{
	var ajax_comment=false;
	
	// Firefox Klasse XMLHttpRequest.
	if (window.XMLHttpRequest)
	{
		ajax_comment=new XMLHttpRequest;
		
	}

	// Internet Explorer Klasse ActiveX-Objekt.
	else if (window.ActiveXObject)
	{
		try { ajax_comment=new ActiveXObject('Microsoft.XMLHTTP'); }
	        catch (e) { 

	        var fehler = 'ActiveX deactivated.';
	          window.document.getElementById('blognews').innerHTML=fehler;
	        }		
	}

		var opensuccess = false;		

		// Anfrage mit GET
		// Die angegebene URL muss vom selben Server stammen.
		
	        try {
				opensuccess = ajax_comment.open('GET', 'index_blognews.php');
	        }
	        catch (e) { }		

		if (! opensuccess)
		{
				
		// Das Ereignis onreadystatechange tritt ein, wenn sich dieser ändert.
		ajax_comment.onreadystatechange=function()
		{
			// Diese Eigenschaft enthält einen Code für den aktuellen Status.
			// Der Wert 4 steht dabei für complete.
			if (ajax_comment.readyState==4)
			{			
				var responseText = ajax_comment.responseText.replace(/^\s+|\s+$/g,"");				
				
				// Die Eigenschaft responseText enthält die Antwort.
					if (responseText=='false')
					{
						window.document.getElementById('blognews').innerHTML='Die Nachrichten stehen im Moment nicht zur Verf&uuml;gung';
					}	
					else
					{
						window.document.getElementById('blognews').innerHTML=responseText;
					}
			}			
		}

		// Die Anfrage wird gesendet.

	        try { ajax_comment.send(null); }
	        catch (e) { }

		} // ende if opensuccess
		else { alert('error'); }		
			
	} // ende load()
	
	load_blognews();
