        thumbSliderID = '#myCarouselSlider #containerThumb' + slideTo
        $(thumbSliderID).css({'border': '1px solid #F00'});

        $('#myCarouselSlider').carousel(carouselSliderPos);
    });

        var currentSliderId = $(this).parent().attr('id');
        currentSliderId = currentSliderId.toString();
        currentSliderPos = currentSliderId.replace('containerThumb', '');
        currentSliderPos = parseInt(currentSliderPos);
