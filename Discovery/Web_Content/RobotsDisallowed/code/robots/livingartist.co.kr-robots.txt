						prev();
					});

					btn_next.click(function(){
						next();
					});

					var start = function(){
						set_interval = setInterval(function(){
							next();
						}, 3000);
					}
					start();

					var hover = function(){
						$('.brand_box').hover(function(){
					   clearInterval(set_interval);
