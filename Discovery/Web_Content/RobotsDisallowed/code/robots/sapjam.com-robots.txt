    /** User our own callback function to update the profile link when the user has been retrieved from the server **/
    function updateProfileLink(user){
        /** Make sure user has been retrieved before updating the link **/
        if (!user.isEmpty()){
            var hello_message = 'Hi, ';
            var user_firstname = user.getFirstname();
            if ('en_US'.toLowerCase() == 'ja_jp') {
                hello_message = hello_message + user.getLastname() + ' ' + user_firstname + ' ' + String.fromCharCode(27096);
            } else {
                hello_message = hello_message + user_firstname;
            }
            $('#anonymous').hide();
            $('#loggedin').show();
            $('#welcome').text(hello_message);
            $('#logout').text('Log Out');
        } else {
            $('#loggedin').hide();
            $('#anonymous').show();
        }
    }
