function toggleLabelConnexion() {
	// Si l'on a une valeur par defaut (sauvegarde navigateur) alors on n'affiche pas les labels pour éviter la superposition
	if ($.trim($(".formConnexion input#user").val()) != "") {
		$(".formConnexion input#user").prev().hide();
		$(".formConnexion input#pass").prev().hide();
	}
	else {
		$(".formConnexion input#user").prev().show();
		$(".formConnexion input#pass").prev().show();

		$(".formConnexion input#user_prospect").prev().hide();

// Fonction qui permet de définir le formulaire à afficher en fonction du paramètre renseigné.
function afficherFormulaire(typeConnexion) {
	
	// On modifie les spécifications du formulaire en fonction du type de connexion attendu
	if (typeConnexion == "prospect") {
		$("#titreForm").html('<span class="lighter-cr" >Vous souhaitez</span>retrouver vos simulations'); // Titre du formulaire
		$("#formConnexionProspect").show();
		$("#formConnexion").hide();

		$("#titreForm").html('<span class="lighter-cr" >Vous êtes</span>Adhérent/<br />assuré social'); // Titre du formulaire
		$("#formConnexionProspect").hide();
		$("#formConnexion").show();
	}
	$("#liensAccesFormConnexion").hide();
	$("#imageFormConnexion").show();	
}

function retour() {
	$("#formConnexion").hide();
	$("#formConnexionProspect").hide();
	$("#imageFormConnexion").hide();
	$("#liensAccesFormConnexion").show();
}
