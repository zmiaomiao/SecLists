<script>
$(function() {
	var mid = docCookies.getItem("marktID");
	if(mid) {
		var markt_footer_sitemap = globusApp.getMarktFooterSitemap();
		var relData = markt_footer_sitemap[parseInt(mid)];
		for (i in relData) {
			if(relData[i]["Kontakt"]) {
				$(".markt_all_openings").attr("href",(relData[i]["Kontakt"]));
			}
		}
		var navigation_market_info = globusApp.getNavigationMarketInfo();
		renderInfoMI(navigation_market_info[parseInt(mid)]);
		hideMI(navigation_market_info[parseInt(mid)]);
	} else {
		infoMIFallback();
	}
})
function renderInfoMI(data) {
	$(".mic_mein_markt .markt_name").text(data.name);
	$(".mic_mein_markt .markt_street").text(data.street)
	$(".mic_mein_markt .markt_zip").text(data.postalcode);
	$(".mic_mein_markt .markt_city").text(data.city);
	$(".mic_mein_markt .markt_phone").text(data.phone);
	$(".mic_mein_markt .markt_opening").html(data.mainopening);
	$(".mic_mein_markt .markt_all_openings").attr('href', data.link);
}
function hideMI(data) {
	if(!data.hasgastro) {
		$(".mi_essen").remove();
	}
	if(!data.hasgasstation) {
		$(".mi_tanken").remove();
	}
}
function infoMIFallback() {
	var fallback = $('<div class="info_fallback">');
	fallback.append('<h3 class="mic_headline">Bitte wählen Sie zuerst einen Markt aus, um hier die Informationen Ihres Marktes zu sehen.</h3>');
	fallback.append('<a href="/de/maerkte/maerkte.html"class="button btn_orange">Zum Marktfinder</a>');
	$('.mic_mein_markt .panel').empty().append(fallback);
}
</script>
