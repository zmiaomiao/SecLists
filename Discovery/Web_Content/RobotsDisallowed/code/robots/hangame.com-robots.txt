function preRecSmartPhoneGame() {
	
	cPage--;
	if (cPage <= 0) {
		cPage = 1;
	}
	$j('.recSmartPhoneGame').each(function(n) {
		if ((cPage - 1) * 3 <= n && n < (cPage * 3)) $j('.recSmartPhoneGame').eq(n).show();
		else $j('.recSmartPhoneGame').eq(n).hide();
	});	
}

function nextRecSmartPhoneGame() {
	
	var size = $j('.recSmartPhoneGame').size();
	var maxPage = Math.floor(size / 3);

	cPage++;
	
	if (cPage > maxPage) cPage = maxPage;

		if ((cPage - 1) * 3 <= n && n < (cPage * 3)) $j('.recSmartPhoneGame').eq(n).show();
		else $j('.recSmartPhoneGame').eq(n).hide();
	});	
}

var PopularNavigator = new (function(){
	var keys = {"ALL":0, "M":1, "WEB":2, "A":3, "CA":4, "BO":5, "S":6, "GO":7, "PO":8, "SMART":9};
	var currentTabIdx = keys['PO'];
	var intervalId;
	var activate = true;
	var that = this;
	this.clickNavi = function (idx, force){
		if(!force && !activate)
			return;
		document.getElementById('pnavi'+currentTabIdx).className="";
		document.getElementById('pnavi'+idx).className="on";
		$j('#top_lst_'+currentTabIdx).hide();
		currentTabIdx = idx;
		$j('#top_lst_'+idx).show();
	}
	
	this.rollingGenreTab = function(){
		intervalId = setInterval(
								function(){
									that.clickNavi(that.nextGenre(), false);
								}, 5000);
	}
	
	this.nextGenre = function(){
		if(currentTabIdx == 9){
			return 0;
		} else {
			return currentTabIdx+1;
		}
	}
	
	this.clickNaviWithClearInterval = function (idx){
		clearInterval(intervalId);
		that.clickNavi(idx, true);
		that.rollingGenreTab();
	}
	
	this.stopRolling = function(){
		activate = false;
	}
	
	this.resumeRolling = function(){
		activate = true;
	}
});

	//이미 로딩이 완료되었다면 처리 중지함

		callback: SearchBox.onLazyLoad

	//결과로 리턴된는 자바스크립트를 로드한다
	AJAX.loadScript(ajaxOptions, ajaxParams);
