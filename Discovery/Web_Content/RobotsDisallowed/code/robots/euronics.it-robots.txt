		function verifyOrder(){
			$('#atg_store_guestEmail').find('span').remove();
			$('#atg_store_guestOrderId').find('span').remove();
			$('div#atg_store_guestEmail input[name=email]').removeClass('error');
			$('div#atg_store_guestOrderId input[name=orderId]').removeClass('error');
			
			var email=$('div#atg_store_guestEmail input[name=email]').val();
			var orderId=$('div#atg_store_guestOrderId input[name=orderId]').val();
			
			if(email == null || orderId == null) {
				alert("I campi sono obbligatori");
			}
			
			$('div#atg_store_guestEmail input[name=email], div#atg_store_guestOrderId input[name=orderId]').prop('disabled', true);
			$('div#atg_store_guestVerify input[name=verifyOrderSubmit]').attr("disabled", "disabled");
			
			$.ajax({
				type: "GET",
				url: '/acquistaonline/guest/myOrdersGuest.jsp',
				data: {email:email, orderId:orderId},
				dataType: "html",
				success: function(data) {
					var orderStatus = $(data).filter('.errorMessage').val();
					if (typeof orderStatus == 'undefined') {
						window.location.replace("/acquistaonline/guest/orderDetailGuest.jsp");
					}
					else {
						$(".checkOrderResult").html(data);
						$('.errorMessage span[data-ref]').each(function() {
							elem = $('#' + $(this).data('ref'))
							if ("SELECT|INPUT|TEXTAREA".indexOf(elem.prop("tagName")) > -1 ) {
								elem.addClass('error')
								elem.parent().append('<span class="error">'+$(this).html()+'</span>')
							}
							else {
								elem.after('<span class="error">'+$(this).html()+'</span>')
							}
							if (elem.length > 0) $(this).parent().remove()
						})
						$('div#atg_store_guestEmail input[name=email], div#atg_store_guestOrderId input[name=orderId]').prop('disabled', false);
						$('div#atg_store_guestVerify input[name=verifyOrderSubmit]').removeAttr("disabled");
					}
				}
			});
		}
