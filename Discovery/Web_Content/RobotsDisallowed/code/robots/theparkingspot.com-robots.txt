        var ImgHeader = document.getElementById('ctl00_imgHeader');
        var DivJump = document.getElementById('DivJumptolocation');
        DivJump.style.width = '680px';
        DivJump.style.display = "block";

        }
        else {
            LEFT = LEFT + ImgHeader.offsetWidth - document.getElementById('DivJumptolocation').offsetWidth + 'px';

        document.getElementById('DivShim').style.left = DivJump.style.left;
        document.getElementById('DivShim').style.zIndex = DivJump.style.zIndex - 10;

    }

    function SignIN() {
        bw = BrowserCheck();
        if (bw.ie) {
            var LEFT = findPosX(document.getElementById("ctl00_SignIn")) - 13 + "px";
            var TOP = findPosY(document.getElementById("ctl00_SignIn")) + 6 + document.getElementById("ctl00_SignIn").offsetHeight + "px";
        }
        else {
            var LEFT = findPosX(document.getElementById("ctl00_SignIn")) - 6 + "px";
            var TOP = findPosY(document.getElementById("ctl00_SignIn")) + 5 + document.getElementById("ctl00_SignIn").offsetHeight + "px";
        }
        document.getElementById("DivSign").style.display = "block";
        document.getElementById("DivSign").style.left = LEFT;
        document.getElementById("DivSign").style.top = TOP;
    }
    function findPosX(obj) {
        bw = BrowserCheck();

        var curleft
        if (bw.ie)
            curleft = 7;
        else
            curleft = 0
        if (obj.offsetParent)
            while (1) {
                curleft += obj.offsetLeft;
                if (!obj.offsetParent)
                    break;
                obj = obj.offsetParent;
            }
        else if (obj.x)
            curleft += obj.x;
        return curleft;
    }

    function findPosY(obj) {
        var curtop = 0;
        if (obj.offsetParent)
            while (1) {
                curtop += obj.offsetTop;
                if (!obj.offsetParent)
                    break;
                obj = obj.offsetParent;
            }
        else if (obj.y)
            curtop += obj.y;
        return curtop;
    }
    function BrowserCheck() {
        this.ver = navigator.appVersion;
        this.agent = navigator.userAgent;
        this.dom = document.getElementById ? 1 : 0;
        this.opera5 = this.agent.indexOf("Opera 5") > -1;
        this.ie5 = (this.ver.indexOf("MSIE 5") > -1 && this.dom && !this.opera5) ? 1 : 0;
        this.ie6 = (this.ver.indexOf("MSIE 6") > -1 && this.dom && !this.opera5) ? 1 : 0;
        this.ie4 = (document.all && !this.dom && !this.opera5) ? 1 : 0;
        this.ie = this.ie4 || this.ie5 || this.ie6;
        this.mac = this.agent.indexOf("Mac") > -1;
        this.ns6 = (this.dom && parseInt(this.ver) >= 5) ? 1 : 0;
        this.ns4 = (document.layers && !this.dom) ? 1 : 0;
        this.bw = (this.ie6 || this.ie5 || this.ie4 || this.ns4 || this.ns6 || this.opera5);
        return this;
    }
