$('h3.dest_foot').click(function(){
  if ($( document ).width() < 680){
    var t = this;
    if($(t).is('.aberto')){
     $(t).removeClass('aberto').next().slideUp();
    } else{
     $('h3.dest_foot').removeClass('aberto').next().slideUp();
     $(t).addClass('aberto').next().slideDown();
    }
  }
});

function ResizeColunas(){
  if ($( document ).width() > 980){
		// $('.cols-ipad .span4:even').addClass('margin-left-tablet');
		// $('.cols-ipad .span4:nth-child(2n)').removeClass('margin-left-mobile');
		$('.cols-ipad .span4:nth-child(3n+1)').addClass('remove_margin');
	}else if($( document ).width() < 980 && $( document ).width() > 640){
		$('.cols-ipad .span4:nth-child(4n)').removeClass('margin-left-pc');
		$('.cols-ipad .span4:nth-child(2n)').removeClass('margin-left-mobile');
		$('.cols-ipad .span4:even').addClass('margin-left-tablet');
	}else{
		$('.cols-ipad .span4:nth-child(4n)').removeClass('margin-left-pc');
		$('.cols-ipad .span4:even').addClass('margin-left-tablet');
		$('.cols-ipad .span4:nth-child(2n)').addClass('margin-left-mobile');
	}
}
