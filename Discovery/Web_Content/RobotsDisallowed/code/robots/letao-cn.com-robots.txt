function GetTime() { 
var dt = new Date();
var def = dt.getTimezoneOffset()/60;
var gmt = (dt.getHours() + def);
var ending = ":" + IfZero(dt.getMinutes()) + ":" +  IfZero(dt.getSeconds());

} 
function IfZero(num) {
return ((num <= 9) ? ("0" + num) : num);

function check24(hour) {
return (hour >= 24) ? hour - 24 : hour;
