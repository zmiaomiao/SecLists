				var add = Math.floor((vttot * (val / (per_minute))) * 10000) / 10000;
				
				vttot = Math.floor((vttot + add) * 10000) / 10000;
				vtnew = Math.floor((vttot - svttot) * 10000) / 10000; 
				
				newVal = vttot;
				updateNum(floatnewtot, newtot, floatnewtot_t, newtot_t, vtnew, num);
			}
			updateNum(floatObj, secFloatObj, floatObj_t, secFloatObj_t, newVal, num);
		}
	}

		var ftop = FROMTOP;

		if (num == 1) newVal = Math.round(newVal);

			floatObj_t.innerHTML = newVal;
			var timerUpdate = setInterval(function(){floatFromTop(floatObj, secFloatObj, floatObj_t, secFloatObj_t);}, 15);
		}
		
		function floatFromTop(floatObj, secFloatObj, floatObj_t, secFloatObj_t) {
			if (n < 23) {
				floatObj.style.top = ftop + "px";
				floatObj.style.opacity = fopac;
				ftop = ftop + 1;
				fopac = fopac + 0.05;
				if (fopac > 1) fopac = 1;
			}
			if ((n >= 5) && (n < 35)) {
				secFloatObj.style.top = fsectop + "px";
				secFloatObj.style.opacity = fsecopac;
				fsectop = fsectop + 1;
				fsecopac = fsecopac - 0.05;
				if (fsecopac < 0) fsecopac = 0;
			}		
			if (n > 40) {
				secFloatObj.style.top = (FROMTOP + 22) + "px";
				secFloatObj.style.opacity = 1;
				secFloatObj_t.innerHTML = floatObj_t.innerHTML;
				floatObj.style.top = (FROMTOP) + "px";
				floatObj.style.opacity = 0;
				clearInterval(timerUpdate);
			}
			
			n = n + 1;
		}
	}
