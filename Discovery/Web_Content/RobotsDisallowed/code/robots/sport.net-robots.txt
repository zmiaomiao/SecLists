    $(document).ready(function () {
        $.getJSON("/api/View/News/0/0", function (allData) {
                console.log(allData);
                viewModelMain = ko.mapping.fromJS(allData);
                ko.applyBindings(viewModelMain);
                if (window.history) {
                    if (window.history.replaceState)
                        window.history.replaceState({ Category: viewModelMain.Parent.Name() }, '');
                }
            });
            window.setTimeout(function () {
                if (window.addEventListener) {
                    window.addEventListener("popstate", function (e) {
                        if (e.state != null) {
                            ReturnBackState(e.state);
                        }
                    });
                }
            }, 5000);
        });
        function ReturnBackState(state) {
            console.log(state);
            if (state.Article !== undefined) {
                OpenArticle(state.Article);
            }
            else if (state.Video != undefined) {
                OpenVideo(state.Video);
            }
            else {
                Close(null);
            }
        }
        function ChangeCategory(id, name, event) {
            $.getJSON("/api/View/News/" + id + "/0", function (allData) {
                console.log(allData);
                ko.mapping.fromJS(allData, viewModelMain);
                if (window.history) {
                    if (window.history.pushState)
                        window.history.pushState(window.history.state, name, '/' + viewModelMain.Parent.Name());
                }
                window.document.title = name + " - Sport.Net";
            });
        }
        $(window).scroll(function () {
            if ($(window).scrollTop() >= $(document).height() - $(window).height() - 150) {
                viewModelMain.CurrentPage(viewModelMain.CurrentPage() + 1);
                $.getJSON("/api/List/News/" + viewModelMain.CategoryId() + "/" + viewModelMain.CurrentPage(), function (allData) {

                    var newItems = ko.mapping.fromJS(allData);
                    for (var i = 0; i < newItems().length; i++) {
                        viewModelMain.News.push(newItems()[i]);
                    }
                });
            }
        });
