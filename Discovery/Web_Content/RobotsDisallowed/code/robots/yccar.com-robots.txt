                     odd.style.right = '0px';
                 }
             } else {
                 oDl[i].onmouseover = function (i) {

                     var odd = this.getElementsByTagName('dd')[0];
                     var aa = this.getElementsByTagName('a')[0];
                     aa.className = 'hover';
                     odd.style.display = 'block';
                 }
             }
             oDl[i].onmouseout = function () {
                 var odd = this.getElementsByTagName('dd')[0];
                 var aa = this.getElementsByTagName('a')[0];
                 aa.className = '';
                 odd.style.display = 'none'
             }
         }
