		<h2>Instructions</h2>

		<p>
			Download the plugin, unzip it, copy files and include fancyBox script and stylesheet in your document (you will need to make sure the css and js files are on your server, and adjust the paths in the script and link tag). Make sure you also load the jQuery library. Example:
		</p>

<pre class="prettyprint">
&lt;!-- Add jQuery library --&gt;
&lt;script type="text/javascript" src="http://code.jquery.com/jquery-latest.min.js"&gt;&lt;/script&gt;

&lt;!-- Add mousewheel plugin (this is optional) --&gt;
&lt;script type="text/javascript" src="/fancybox/lib/jquery.mousewheel-3.0.6.pack.js"&gt;&lt;/script&gt;

&lt;!-- Add fancyBox --&gt;
&lt;link rel="stylesheet" href="/fancybox/source/jquery.fancybox.css?v=2.1.5" type="text/css" media="screen" /&gt;
&lt;script type="text/javascript" src="/fancybox/source/jquery.fancybox.pack.js?v=2.1.5"&gt;&lt;/script&gt;

&lt;!-- Optionally add helpers - button, thumbnail and/or media --&gt;
&lt;link rel="stylesheet" href="/fancybox/source/helpers/jquery.fancybox-buttons.css?v=1.0.5" type="text/css" media="screen" /&gt;
&lt;script type="text/javascript" src="/fancybox/source/helpers/jquery.fancybox-buttons.js?v=1.0.5"&gt;&lt;/script&gt;
&lt;script type="text/javascript" src="/fancybox/source/helpers/jquery.fancybox-media.js?v=1.0.6"&gt;&lt;/script&gt;

$(&quot;.fancybox&quot;).fancybox({

$(&quot;.fancybox&quot;).fancybox({
    helpers:  {
        overlay : null
    }
});

//Change title position; show overlay after content has loaded
$(&quot;.fancybox&quot;).fancybox({
    helpers:  {
        title : {
            type : 'inside'
        },
        overlay : {
            showEarly : false
        }
    }
});

//Enable thumbnail helper and set custom options
$(&quot;.fancybox&quot;).fancybox({
