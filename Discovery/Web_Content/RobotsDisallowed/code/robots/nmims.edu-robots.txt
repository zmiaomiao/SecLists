function populateStates1(index, stateElementId) {

function populateCountries1(countryElementId, stateElementId) {

	}
}

function checkForEmpty1(EleId) {
		var x = document.getElementById(EleId).value;
		if (x == null || x == "" || x == "-1") {
		document.getElementById(EleId+"Error1").classList.remove("error-No-msg1");
		document.getElementById(EleId+"Error1").classList.add("error-Yes-msg1");
		document.getElementById(EleId + "Error1").innerHTML = "Must Enter";
		return false;
		} else {
			console.log(EleId)
		document.getElementById(EleId+"Error1").classList.add("error-No-msg1");
		document.getElementById(EleId+"Error1").classList.remove("error-Yes-msg1");
		document.getElementById(EleId + "Error1").innerHTML = "";
		if(EleId == 'nmims_user_email_id1'){
				if(!validateEmail1(EleId))
					return;
				}
		if(EleId == 'nmims_user_mobile_number1'){
				if(!makeContactValid1(EleId))
					return;
			}
		return true;
		}
}
function validateEmail1(eleId) {
		var x = document.getElementById(eleId).value;
		var atpos = x.indexOf("@");
		var dotpos = x.lastIndexOf(".");
		if (atpos < 1 || dotpos < atpos + 2 || dotpos + 2 >= x.length) {
			document.getElementById(eleId+"Error1").classList.remove("error-No-msg1");
			document.getElementById(eleId+"Error1").classList.add("error-Yes-msg1");
			document.getElementById(eleId + "Error1").innerHTML = "Not a valid e-mail address";
			return false;
		} else {
			document.getElementById(eleId+"Error1").classList.add("error-No-msg1");
			document.getElementById(eleId+"Error1").classList.remove("error-Yes-msg1");
			document.getElementById(eleId + "Error1").innerHTML = "";
			return true;
		}
}
function makeContactValid1(eleId) {
		var phoneno = /^\d{10}$/;
		var inputtxt = document.getElementById(eleId).value;
		if (inputtxt.match(phoneno)) {
			document.getElementById(eleId+"Error1").classList.add("error-No-msg1");
			document.getElementById(eleId+"Error1").classList.remove("error-Yes-msg1");
			document.getElementById(eleId + "Error1").innerHTML = "";
			return true;
		} else {
			document.getElementById(eleId+"Error1").classList.remove("error-No-msg1");
			document.getElementById(eleId+"Error1").classList.add("error-Yes-msg1");
			document.getElementById(eleId + "Error1").innerHTML = "Not a valid phone number";
			return false;
		}
}

function populateStates(index, stateElementId) {

	var stateElement = document.getElementById(stateElementId);

	stateElement.length = 1; // Fixed by Julian Woods
	stateElement.selectedIndex = 0;
	var state_arr = s_a[index].split("|");

	for ( var i = 0; i < state_arr.length; i++) {
		stateElement.options[stateElement.length] = new Option(state_arr[i],
				"st_" + index + "_" + i);
	}
}

function populateCountries(countryElementId, stateElementId) {
	var countryElement = document.getElementById(countryElementId);
	if (countryElement) {
		countryElement.selectedIndex = 0;
		for ( var i = 0; i < country_arr.length; i++) {
			countryElement.options[countryElement.length] = new Option(
					country_arr[i], 'cn_' + i);
		}

		if (stateElementId) {
			countryElement.onchange = function() {
				populateStates(
						document.getElementById(countryElementId).selectedIndex,
						stateElementId);
			};
		}
	}
}

function checkForEmpty(EleId) {
	
	var x = document.getElementById(EleId).value;
	if (x == null || x == "" || x == "-1") {
	document.getElementById(EleId+"Error").classList.remove("error-No-msg");
	document.getElementById(EleId+"Error").classList.add("error-Yes-msg");
	document.getElementById(EleId + "Error").innerHTML = "Must Enter";
	return false;
	} else {
	document.getElementById(EleId+"Error").classList.add("error-No-msg");
	document.getElementById(EleId+"Error").classList.remove("error-Yes-msg");
	document.getElementById(EleId + "Error").innerHTML = "";
	if(EleId == 'nmims_user_email_id'){
			if(!validateEmail(EleId))
				return;
			}
	if(EleId == 'nmims_user_mobile_number'){
			if(!makeContactValid(EleId))
				return;
		}
	return true;
	}
	}

function validateEmail(eleId) {
	var x = document.getElementById(eleId).value;
	var atpos = x.indexOf("@");
	var dotpos = x.lastIndexOf(".");
	if (atpos < 1 || dotpos < atpos + 2 || dotpos + 2 >= x.length) {
		document.getElementById(eleId+"Error").classList.remove("error-No-msg");
		document.getElementById(eleId+"Error").classList.add("error-Yes-msg");
		document.getElementById(eleId + "Error").innerHTML = "Not a valid e-mail address";
		return false;
	} else {
		document.getElementById(eleId+"Error").classList.add("error-No-msg");
		document.getElementById(eleId+"Error").classList.remove("error-Yes-msg");
		document.getElementById(eleId + "Error").innerHTML = "";
		return true;
	}
}
function makeContactValid(eleId) {
	var phoneno = /^\d{10}$/;
	var inputtxt = document.getElementById(eleId).value;
	if (inputtxt.match(phoneno)) {
		document.getElementById(eleId+"Error").classList.add("error-No-msg");
		document.getElementById(eleId+"Error").classList.remove("error-Yes-msg");
		document.getElementById(eleId + "Error").innerHTML = "";
		return true;
	} else {
		document.getElementById(eleId+"Error").classList.remove("error-No-msg");
		document.getElementById(eleId+"Error").classList.add("error-Yes-msg");
		document.getElementById(eleId + "Error").innerHTML = "Not a valid phone number";
		return false;
	}
}
