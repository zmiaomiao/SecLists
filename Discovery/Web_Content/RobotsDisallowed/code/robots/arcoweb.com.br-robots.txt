	function validaNews(){
		var valida = true;
		if($('#txtNome').val() == ''){
			$('#txtNome').addClass('error');
			valida = false;
		}else{
			$('#txtNome').removeClass('error');
		}
		if($('#txtEmail').val() == ''){
			$('#txtEmail').addClass('error');
			valida = false;
		}else{
			var filtro = /^([\w-]+(?:\.[\w-]+)*)@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$/i;
			if(!filtro.test($('#txtEmail').val())){
				$('#txtEmail').addClass('error');
				valida = false;
			}else{
				$('#txtEmail').removeClass('error');
			}
		}

		if($('#txtCaptcha').val() == ''){
			$('#txtCaptcha').addClass('error');
			valida = false;
		}else{
			$('#txtCaptcha').removeClass('error');
		}
		return valida;
	}
