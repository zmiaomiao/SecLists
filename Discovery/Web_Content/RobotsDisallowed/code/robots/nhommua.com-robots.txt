        function subscribeEmailPopup() {
            var email = $.trim($("#txtEmailPopup").val());

            if ($.trim(email) == '') {
                $("#errorEmailPopup").html("Vui lòng nhập email");
                $("#txtEmailPopup").addClass("input-validation-error");
                $("#txtEmailPopup").focus();
                return;
            }
            if (!isValidEmailAddress(email)) {
                $("#errorEmailPopup").html("Email không hợp lệ");
                $("#txtEmailPopup").addClass("input-validation-error");
                $("#txtEmailPopup").focus();
                return;
            }

            $.post('/Shared/Subscribe', { 'email': email, 'cityId': cityIdSelect, 'isSendEmail':false })
                .done(function (data) {
                    if (data.success) {
                        hidePopup('#PopupEmail');
                        initAndShowPopupSmallCallback("Vui lòng kiểm tra email để hoàn tất đăng ký và nhận ngay <b>QUÀ CỰC HOT</b>. Nếu không thấy bạn vui lòng kiểm tra trong hộp thư rác (spam box)", 'gotoCity()', 'Kiểm tra ngay');
                    } else {
                        gotoCity();
                    }
                });

            //hoanglong.nguyen
            //Facebook tracking
            dynamicMarketing.subscribleEmail();
        }
