jQuery(document).ready(function() {
	var $splash = jQuery('#splash');
	if ($splash.find('.item').length < 2) {
		$splash.find('.navigation').hide();
		return;
	}
	var $selItem = $splash.find('.item').eq(0);
	var $selIndicator = $splash.find('.position_indicator_active').eq(0);

	var interval = window.setInterval(function () {
		changeItem();
	}, 5000);

	var changeItem = function ($item) {
		var $currItem = $selItem.css('z-index', 15);
		if ($item) {
			$selItem = $item;
		} else {
			if ($currItem.next().length == 0) {
				$selItem = $currItem.siblings().eq(0);
			} else {
				$selItem = $currItem.next();
			}
		}

		$currItem.siblings().css('z-index', 10); 
		$selItem.css('z-index', 20).fadeIn('500', function () {
			$currItem.hide();
		});

		if ($selIndicator) {
			var position = $selItem.children('h3').attr('class').replace(/[^0-9]+/,'');
			$selIndicator = jQuery('.position_indicator_' + position);
			$selIndicator.siblings().removeClass('position_indicator_active');
			$selIndicator.addClass('position_indicator_active');
		}
	};
	
	$selItem.siblings().hide();
	$splash.find('.button').click(function () {
		window.clearInterval(interval);
		var $this = jQuery(this);
		var $item;
		
		if ($this.is('.button-prev')) {
			$item = ($selItem.prev().length == 0) ? $selItem.siblings().eq(-1) : $selItem.prev(); 
		} else if ($this.is('.button-next')) {
			$item = ($selItem.next().length == 0) ? $selItem.siblings().eq(0) : $selItem.next();

			var position = $this.attr('class').replace(/[^0-9]+/,'');
			$item = $splash.find('.item').eq(position - 1);
		}
		changeItem($item);
		return false;
	});
});
