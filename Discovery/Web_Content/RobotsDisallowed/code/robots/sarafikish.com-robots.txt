        //==================================================================================fixdigit
        function fixdigits($num, $dig) {
            $zero = "000000000000"
            $str = $num.toString() + $zero;
            if ($str.indexOf(".") > 0) {
                $asharindex = $str.indexOf(".");
                $sahih = $str.substr(0, $asharindex);
                $str = $sahih + "." + $str.substr($asharindex + 1, $dig);
            } else {
                $str = $num.toString() + "." + $zero.substr(0, $dig);
            }
            return ($str);
        }
