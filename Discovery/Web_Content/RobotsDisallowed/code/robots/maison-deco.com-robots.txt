// SLIDESHOW BLOC SHOPPING HOMEPAGE
$.fn.slideshow = function(){

	return this.each(function(){
		// Variables
		var $shop = $(this),
			$ulShop = $shop.find('ul'),
			$liShop = $ulShop.find('li'),
			ulMarginShop = parseInt($ulShop.css('marginLeft'), 10),
			liWidthShop = parseInt($liShop.css('width'), 10) + parseInt($liShop.css('marginRight'), 10) + 1,
			//pour le liwidth le "+1" repr�sente le pixel du border-right
			liLengthShop = $ulShop.find('li').length,
			ulWidthShop = liWidthShop * liLengthShop,
			nextShop = $('.next', $(this)),
			prevShop = $('.prev', $(this));

			totalWidthShop = ulWidthShop - (ulWidthShop * 2) + (liWidthShop*4) + ulMarginShop;
			limitShop = liWidthShop * 4;

			$ulShop.css('width',ulWidthShop+"px");

			if ( (($ulShop.hasClass("list-diapo")) && (liLengthShop < 9)) || (ulWidthShop <= limitShop) ){
				$(nextShop).hide();
			}

		// Clic next (en evitant qu'il y ait plusieurs clics d'un seul coup)
		$('.next:not(.animated)', $shop).live('click', function(){
			$(nextShop).addClass("animated");

			var diapoNextShop = $ulShop.css('marginLeft');
				valueNextShop = parseInt(diapoNextShop, 10) - liWidthShop;
			if(valueNextShop == totalWidthShop){
				$(nextShop).hide();
			}
			if(valueNextShop < ulMarginShop){
				$(prevShop).show();
			}

			$ulShop.animate({ "margin-left": '-='+liWidthShop }, 500, function() {
				$(nextShop).removeClass("animated");
			});

		});

		// Clic prev
		$('.prev:not(.animated)', $shop).live('click', function(){
			$(prevShop).addClass("animated");
			var diapoPrevShop = $ulShop.css('marginLeft'),
				valuePrevShop = parseInt(diapoPrevShop, 10) + liWidthShop;
			if(valuePrevShop == ulMarginShop){
				$(prevShop).hide();
			}
			if(valuePrevShop > totalWidthShop){
				$(nextShop).show();
			}
			$ulShop.animate({ "margin-left": '+='+liWidthShop }, 500, function() {
				$(prevShop).removeClass("animated");
			});
		});

	});
};

$(function(){
