var CCountAnimate = true; //企业统计动画默认；

function CCont(_setNum, _animate) { //企业数统计
	
	var num = ['0', '-246', '-492', '-738', '-984', '-1230', '-1476', '-1722', '-1968', '-2214'];   //0-9背景定位
	var span = 41;   //数字牌高度
	var maxTop = -2419;   //背景定位限制
	var setNums = _setNum.toString().split('');   //开通数转化数组
	
	var CC = document.getElementById('CC');
	//var items = Fn.s('.num', CC);   //获取所有元素
	var items = $(".num");
	
	if (_animate) {   //翻牌动画
		
		var stop = {   //停止机制
			lock : false,
			index : items.length - 1
		};
		
		setTimeout(function() {
			stop.lock = true;   //半秒后开始停止
		}, 300);
		var speed = 15;   //-------> speed
		var stopSpeed = 100;      //-------> stopSpeed

		items.each(function(i, item) {
			var item = item;   //当前元素
			var i = i;   //当前索引
			var timeout = null;   //timeout回调函数初始
			var timer = null;   //setTimeout
			var setNum = setNums[i];   //获取对应数字
			
			timeout = function(){
				var currentTop = parseInt($(item).css("top")) || 0;   //获取当前TOP值
				timer = window.setTimeout(function(){
					timeout();
				}, speed);
				
				if (currentTop == maxTop){
					currentTop = 0;   //重复循环
				}
				
				if (stop.lock == true && stop.index == i) {   //是否开始停止
					stopSpeed = stopSpeed + 100;   //-------> stopSpeed
					//alert(stop.index+";"+i);
					window.clearTimeout(timer);
					item.style.top = num[setNum] + 'px';
					stop.index = i - 1;   //设置下一个数字索引
					stop.lock = false;
					window.setTimeout(function(){
						stop.lock = true;
					}, stopSpeed);
					
					return false;
				}
				item.style.top = currentTop - span + 'px';   //设置TOP值
			};
			timer = window.setTimeout(function(){
				timeout();
			}, speed);
		});

	}
	else{   //直接显示
		$(".num").each(function(i){
			var setNum = setNums[i];
			$(this).css("top", num[setNum]+"px");
		});
	}
	
}
