        function checkPartnerValue() {
            var url = new Uri(window.location);
            var partnerParam = url.getQueryParamValue('s');
            if (partnerParam != null) {
                var s = getCookie("sources").split(',');
                var index = $.inArray(partnerParam, s);
                if (index == -1 && (s + partnerParam + ",").length <= 100)
                    s += partnerParam + ",";

                    document.cookie = "sources=" + data;
                });
            }

        function getCookie(cname) {
            var name = cname + "=";
            var ca = document.cookie.split(';');
            for (var i = 0; i < ca.length; i++) {
                var c = ca[i].trim();
                if (c.indexOf(name) == 0) return c.substring(name.length, c.length);
            }
            return "";
