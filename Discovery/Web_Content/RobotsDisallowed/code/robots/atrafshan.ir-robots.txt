function commentRate (s, id, th){

    var th = $(th);
    $.ajax({
            url : "/comments/comment-rate-save",
            data : {"data" : s, "id" : id},
            "beforeSend" : function(){$(th).append( refresh );},
            type : "get",
            dataType : "json",
            success : function ( res ) {
                if(res.status == "ok"){
                    th.siblings(".active").removeClass("active");
                    th.addClass("active");
                    var num = $(th).parents(".rate").find(".num");
                    var a = parseInt(num.html());
                    if( num.hasClass("neg") ){
                        var v = s ? a-1 : a+1;
                    }else{
                        var v = s ? a+1 : a-1;
                    }
                    v = Math.abs(v);
                    num.html(v);
                    if( v == 1 ){
                        if( s && a!=2  )
                            num.removeClass("neg").addClass("pos");
                        else if( !s && a != 2)
                            num.removeClass("pos").addClass("neg");

                    }else if( v == 0){
                        num.removeClass("pos").removeClass("neg");
                    }
                    $(th).append("<i class='fa fa-check m-r checked'></i>");
                    window.setTimeout(function(){$(th).find(".checked").remove();},1000);
                }else if(res.status == "doLogin"){
                    showLogin();
                }else{
                    $(th).append("<i class='fa fa-times m-r times'></i>");
                    window.setTimeout(function(){$(th).find(".times").remove();},1000);
                }
            }
        }).always(function(){ $(th).find("i.refresh").remove(); });

    return false;
}

function inArray(needle, haystack) {
    var length = haystack.length;
    for(var i = 0; i < length; i++) {
        if(haystack[i] == needle) return true;
    }
    return false;
}
