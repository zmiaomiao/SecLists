    function closeEventPopup(obj)
    {
    	if (obj == "1" )
    	{
    		setEventCookie( "warpex_popup_201503", "done" , 20*365);
    	}
    
    	$("#warpex_event_popup").hide();
    }
    
    function setEventCookie( name, value, expiredays )
    {
    	var todayDate = new Date();
    	todayDate.setDate( todayDate.getDate() + expiredays );
    	document.cookie = name + "=" + escape( value ) + "; path=/; expires=" + todayDate.toGMTString() + ";"
    }
    function checkEventPopup()
    {
    	//쿠키값이 있으면 element를 가려준다
    	if(getEventCookie("warpex_popup_201503") == "done")
    	$("#warpex_event_popup").hide();
    	//쿠키값이 없으면 element를 보여준다
    	else
    	$("#warpex_event_popup").show();
    }
    function getEventCookie( name )
    {
    	var nameOfCookie = name + "=";
    	var x = 0;
    	while ( x <= document.cookie.length )
    	{
    		var y = (x+nameOfCookie.length);
    		if ( document.cookie.substring( x, y ) == nameOfCookie )
    		{
    			if ( (endOfCookie=document.cookie.indexOf( ";", y )) == -1 )
    			endOfCookie = document.cookie.length;
    			return unescape( document.cookie.substring( y, endOfCookie ) );
    		}
    		x = document.cookie.indexOf( " ", x ) + 1;
    		if ( x == 0 )
    		break;
    	}
    	return "";
    }
    checkEventPopup();
