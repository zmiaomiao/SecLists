                var self = this;

                self.init();
            };

            SpecialEffects.prototype = {
                init: function () {

                    var self = this,
                        readyCookie = Y.Cookie.get('new_index_cookie');

                    if (readyCookie !== 'xianjin') {
                        S.Cookie.set('new_index_cookie', 'xianjin', { expires: new Date('2017/08/20 00:00:00') });
                    } else {
                        return false;
                    }

                    self.topValue = 0;
                    self.createNode();
                    self.bindEvent();

                createNode: function () {
                    var self = this;

                    Nodes["#doc"] = Y.one('#doc');

                    self.overlayMask = Y.Node.create(ytMask);
                    self.overlayLove = Y.Node.create(ytLove);
                    self.viewport = Y.one('document').get('viewportRegion');

                    if (self.viewport.height > 400) {
                        self.topValue = (self.viewport.height - 400) / 2;
                    }
                    self.overlayLove.setStyle("left", (self.viewport.width - 574) / 2);

                    //IE6 下页面初始载入时若不是在最顶部位置，则需要设置top，但是IE到渲染页面实际打开位置需要时间，，所以延迟获取scTop
                    if (Y.UA.ie == 6) {
                        setTimeout(function () {
                            self.scTop = document.documentElement.scrollTop || document.body.scrollTop;
                            self.overlayMask.setStyle('top', self.scTop);
                            self.overlayLove.setStyle("top", self.scTop + self.topValue);
                        }, 700);
                    }


                    Nodes["#doc"].insert(self.overlayMask, "before");
                    Y.one("body").append(self.overlayLove);

                    Nodes["#yt-over-mask"] = Y.one("#yt-over-mask");
                    Nodes[".yt-love"] = Y.one(".yt-love");
                    Nodes[".yt-closebox"] = Y.one(".yt-closebox");
                    Nodes[".yt-eventlink"] = Y.one(".yt-eventlink");
                    Nodes["Win"] = Y.one(window);
                },
                bindEvent: function () {
                    var self = this;

                    var fallingAnimLoveOpen = new S.Anim({
                        node: Nodes[".yt-love"],
                        duration: .5,
                        to: { top: self.topValue }
                    });


                    var fallingAnimLoveClose = new S.Anim({
                        node: Nodes[".yt-love"],
                        duration: .5,
                        to: { top: -700 }
                    });

                    var fallingAnimMaskOpen = new S.Anim({
                        node: Nodes["#yt-over-mask"],
                        duration: .5,
                        to: { top: 0 }
                    });

                    var fallingAnimMaskClose = new S.Anim({
                        node: Nodes["#yt-over-mask"],
                        duration: .5,
                        to: { height: 0 }
                    });


                    fallingAnimMaskOpen.run();
                    //
                    fallingAnimLoveOpen.run();
                    function IE6Navfixed() {
                        self.scTop = document.documentElement.scrollTop || document.body.scrollTop;
                        Nodes["#yt-over-mask"].setStyle('top', self.scTop)
                        Nodes[".yt-love"].setStyle('top', self.scTop);
                    }
                    //IE6 滚动时
                    if (Y.UA.ie == 6) {
                        Nodes["Win"].on("scroll", IE6Navfixed);
                    }


                    Nodes[".yt-closebox"].on("click", function () {
                        fallingAnimLoveClose.run();
                        fallingAnimLoveClose.on("end", function () {
                            self.overlayLove.remove(true);
                            if (Y.UA.ie == 6) {
                                Nodes["Win"].detach('scroll', IE6Navfixed);
                            }
                        });
                        fallingAnimMaskClose.run();
                        fallingAnimMaskClose.on("end", function () {
                            self.overlayMask.remove(true);
                            if (Y.UA.ie == 6) {
                                Nodes["Win"].detach('scroll', IE6Navfixed);
                            }
                        });
                    });

                    Nodes[".yt-eventlink"].on("click", function () {
                        fallingAnimLoveClose.run();
                        fallingAnimLoveClose.on("end", function () {
                            self.overlayLove.remove(true);

                                Nodes["Win"].detach('scroll', IE6Navfixed);

                            if (Y.UA.ie == 6) {
                                Nodes["Win"].detach('scroll', IE6Navfixed);
                            }

                        });

                    });
                }
            };

            new SpecialEffects();
        });
