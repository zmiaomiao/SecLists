        function arrayShuffle(oldArray) {
            var newArray = oldArray.slice();
            var len = newArray.length;
            var i = len;
            while (i--) {
                var p = parseInt(Math.random() * len);
                var t = newArray[i];
                newArray[i] = newArray[p];
                newArray[p] = t;
            }
            return newArray;

        function pauseYTVideoOnMenu() {
            var obj = $('.menu_video:visible');
            if (obj.length > 0) {
                var player = swfobject.getObjectById(obj.attr('id'))
                var state;
                try {
                    state = player.getPlayerState();
                } catch (e) {
                    console.log(e);
                }
                if (player && state == 1) {
                    obj.attr('data-time', player.getCurrentTime())
                    obj.attr('data-status', player.getPlayerState())
                    player.pauseVideo();
                    player.stopVideo();
                }
            }
        }


        function setMenuVideo(options) {
            var params = { allowScriptAccess: "always", allowFullscreen: true };
            var atts = { "class": "menu_video" };
            var embed = String.format("//www.youtube.com/v/{0}?enablejsapi=1&playerapiid={1}", options.videoId, options.elementId)
            swfobject.embedSWF(embed, options.elementId, options.width, options.height, "8", null, null, params, atts);
        }

        function ytListener(newState) {
            var obj = $('.menu_video:visible');
            var player = swfobject.getObjectById(obj.attr('id'));
            switch (newState) {
                case 0:
                    obj.removeAttr('data-time')
                    break;
                case -1:
                case 5:
                    if (obj.data("time") && obj.data("time") != 0) {
                        if (obj.data('status') == 2) {
                            player.seekTo(obj.data("time"));
                            player.pauseVideo();
                        }
                        else
                            player.seekTo(obj.data("time"));
                    }
                    break;
                case 2:
                    obj.attr('data-status', player.getPlayerState());
                    break;
                default:
                    break;
            }
        }

        window.onYouTubePlayerReady = function (playerid) {
