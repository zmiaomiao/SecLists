function unescapeHtml (html) {
      var temp = document.createElement("div");
      temp.innerHTML = html;
      var result = temp.childNodes[0].nodeValue;
      temp.removeChild(temp.firstChild)
      return result;
  }
 
 


function removeEscapeCharacters(text){	
	//Remove all Escape characters like amerpsand,apostrophe,trademark,registered,etc  from the selected search suggestion
	//Because the escape characters in the search string would yield incorrect results 
	text=text.replace(/&[a-zA-Z]{1,7};/g, '');
	text=text.replace(/[^0-9a-zA-Z\s-]/g, '');
	
	return text;
}

}

function mycarousel_itemVisibleInCallback(carousel, item, i, state, evt)
{
    // The index() method calculates the index from a
    // given index who is out of the actual item range.
    var idx = carousel.index(i, currentItemList.length);
    carousel.add(i, mycarousel_getItemHTML(currentItemList[idx - 1]));
};


function mycarousel_itemVisibleOutCallback(carousel, item, i, state, evt)
{
    carousel.remove(i);
};

function mycarousel_getItemHTML(item)
{
    return '<a class="carouselImg" href="' + item.url + '"><img src="' + item.imageSRC + '" title = "' + item.title + '" alt="' + item.title + '" />' +
			'<p>' + item.title + '</p>' +
			'<p>' + item.price + '</p></a>' +
			'<a href="' + item.url + '" class="buyNow"></a>';
};

jQuery(document).ready(function() {
    jQuery('#mycarousel').jcarousel({
        wrap: 'circular',
        itemVisibleInCallback: {onBeforeAnimation: mycarousel_itemVisibleInCallback},
        itemVisibleOutCallback: {onAfterAnimation: mycarousel_itemVisibleOutCallback},
        initCallback: mycarousel_initCallback,
        visible: 5
    });
});

var carousel;
function mycarousel_initCallback(thisCarousel)
{
	carousel = thisCarousel;
};

function selectTab(newCategoryItemList, clickedTab)
{
	currentItemList = newCategoryItemList;
	carousel.reset()
	
	$(".categoryTab").removeClass('active');

	$(clickedTab).addClass('active');
};
