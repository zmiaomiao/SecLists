<script>
    $(document).ready(function() {
        $("#reg_form").submit(function(e){
            var selectedRole = $("input:checked");
            if(selectedRole.length == 0) {
                e.preventDefault();
                $("#roleSelector").css('color', 'red');
                $(".submitButton").removeAttr('disabled');
            } else {
                switch (selectedRole.val()) {
                    case 'publisher' : ga('send', 'event', 'button', 'click', 'fast_publisher_registration');
                        break;
                    case 'advertiser' : ga('send', 'event', 'button', 'click', 'fast_advertiser_registration');
                        break;
                }
            }
        })
        $(".role").click(function(e){
            $("#roleSelector").css('color', 'black');
        });
    });
</script>
