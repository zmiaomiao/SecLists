<script>
$(function(){
		var input = document.getElementsByTagName("input");
		if( input[0].value == '搜索:秋季满分冲刺班' ){
			input[0].style.color = '#CCCCCC';
		}
		input[0].onfocus = function(){
			input[0].value = (input[0].value == '搜索:秋季满分冲刺班') ? '' : input[0].value;
			input[0].style.color = '#333333';
		}; 
		input[0].onblur = function(){
			input[0].value = (input[0].value == '') ? '搜索:秋季满分冲刺班' : input[0].value;
			input[0].style.color = (input[0].value == '搜索:秋季满分冲刺班') ? '#CCCCCC' : '#333333';
		};
		
		fnKeyup('.search_suggest', '.head_top_search');
		function fnKeyup(obj,nbj){
			var listIndex = -1;
			//键盘enter和Backspace事件
			input[0].onkeyup = function(e){
				var e = window.event || e ;
				if(e.keyCode == 13){  
					search();  
				} else if (e.keyCode == 8){
					//删除时将listIndex重置
					listIndex = -1;
				}
			};

			$(nbj).attr('autocomplete','on');
			//obj=内容检索框
			//nbj=input输入框
			$(nbj).bind('keyup',function(e){
				//检索框定位
				$(obj).css({
					top:$(nbj).parent().offset().top+$(nbj).parent().height(),
					left:$(nbj).parent().offset().left	
				}).show();

				var searchSuggest = $(obj);
				var self = this;
				var size = searchSuggest.find('li').size();

				//键盘方向键按下要执行的方法
				var fn = function(index,isUpdate){
					if(isUpdate === true)self.value = searchSuggest.find('li').eq(index).html();
					searchSuggest.find('li').removeClass('cur');
					searchSuggest.find('li').eq(index).addClass('cur');
				}

				// 键盘方向键事件
				switch(e.keyCode){
				    case 38://up
				    if (size != 0) {
					    listIndex--;
					    if(listIndex < 0)listIndex = size - 1;
					    fn(listIndex, true);
					    return;				    	
				    } else {
				    	$(obj).hide();
				    	return;
				    }
				    break;
				    case 40://down
				    if (size != 0) {
					    listIndex++;
					    if(listIndex >= size )listIndex = 0;
					    fn(listIndex, true);  
					    return;				    	
				    } else {
				    	$(obj).hide();
				    	return;
				    }
				    break;
				}

				//获取搜索框中输入的值
				txt = $(nbj).val();
				if (txt == '') {
					listIndex = -1;
					$(obj).css('display', 'none');
					return;
				} else {
					//限制字符长度
					if (txt.length > 50) {
						txt = txt.substr(0,30);
					}				
					//ajax获取搜索联想词
					$.ajax({
						type:"GET",
						data:{'data':txt},
						url:"/SearchAll/getKeywords/",
						dataType:'jsonp',
						jsonp:'jsonpCallback',
						success:function(msg) {
							if (msg.stat == 1) {
								var keyword = msg.data;
								var count = msg.rows;
								$(obj).find('li').remove();
								//设置显示搜索联想词
								for (var i = 0; i < count; i++) {
									if (keyword[i]) {
										if (keyword[i].length > 20) {
											$(obj).find('ul').append('<li>' + keyword[i].substr(0,12) + '...'  + '</li>');					
										} else {
											$(obj).find('ul').append('<li>' + keyword[i] + '</li>');
										}
									}
								}
								$(obj).show();
							} else {
								$(obj).hide();
							}

							//当鼠标指向时只显示选择一个
							listIterm('.search_suggest');
							function listIterm(obj){
								//obj = search_suggest(内容检索框)
								$(obj).find('li').bind('mouseover', function (){
									$(this).addClass('cur').siblings().removeClass('cur');
									listIndex = $(this).index();
								}).bind('mouseout', function(){
									$(this).removeClass('cur');
									listIndex = $(this).index();
								});
							}

							//改变搜索框的值
							$(obj).find('li').each(function(){
								$(this).click(function(){
									$(nbj).val($(this).html());
									search();
								})
							});
						}
					});
				}
			});
		}
	});
		// 触碰页面其余位置检索框隐藏
		$(document).bind("click",function(){
			$('.search_suggest').hide();
		});
		// 点击搜索按钮执行搜索方法
		function search(){
			var url = 'http://www.xueersi.com/search';
			var keyword = $('#mq').val();
			if (keyword == '') {
				keyword = '秋季满分冲刺班';
			} else if (keyword.length > 50) {
				//限制字符长度
				keyword = keyword.substr(0,38);
			}
			url +='?keyword='+keyword;
			window.location.href = url;
		}
