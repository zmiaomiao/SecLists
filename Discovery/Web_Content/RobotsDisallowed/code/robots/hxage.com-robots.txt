function _showPicsss(index) { //普通切换
	var sWidth = $(".ts_right").width();
    var nowLeft = -index*sWidth; //根据index值计算ul元素的left值
	$(".ts_right > div").stop(true,false).animate({"left":nowLeft},300); //通过animate()调整ul元素滚动到计算出的position
	$(".ts_left > div").attr('class',"dir_2").eq(index).attr('class',"dir_1");
}
$(function(){
	var sWidth = $(".ts_right").width(); //获取焦点图的宽度（显示面积）
	var len = $(".ts_right > div > img").length; //获取焦点图个数
    var index = 0;
	var _picTimer;
	$(".ts_right > div").css('width',$(".ts_right").width()*$(".ts_right > div > img").length);
	$(".ts_leftbut > a").click(function(){
		index -= 1;
        if(index==-1) {index = len-1;}
        _showPics(index);
	});
	$(".ts_rightbut > a").click(function() {
        index += 1;
        if(index == len) {index = 0;}
        _showPics(index);
    });
	//鼠标滑上焦点图时停止自动播放，滑出时开始自动播放
    $(".tese").hover(function() {
        clearInterval(_picTimer);
    },function() {
        _picTimer = setInterval(function() {
	    if(index == len) {index = 0;}
            _showPics(index);
            index++;
        },3000); //此4000代表自动播放的间隔，单位：毫秒
    }).trigger("mouseleave");
	function _showPics(index) { //普通切换
        var nowLeft = -index*sWidth; //根据index值计算ul元素的left值
		$(".ts_right > div").stop(true,false).animate({"left":nowLeft},300); //通过animate()调整ul元素滚动到计算出的position
		$(".ts_left > div").attr('class',"dir_2").eq(index).attr('class',"dir_1");
    }
	//合作媒体
	var _speed=3000;
	var _slide = $("#media_div");
	var _slide_len = _slide.find(".media_c").length;
	if(_slide_len>8){
	var wid = _slide.find(".media_c").width()+15;
	var _slide_index = 0;
	_slide.width(wid*_slide_len*2);
	_slide.append(_slide.html());
	function Marquee(){
		if(_slide_index > _slide_len) {
			_slide_index = 0;
			_slide.css("left",0);
			_slide_index++;
		}
		var nowLeft = -_slide_index*wid; //根据index值计算ul元素的left值
		_slide.stop(true,false).animate({"left":nowLeft},300);
		_slide_index++;
	}
	//两秒后调用
		var sliding=setInterval(Marquee,_speed)
		_slide.hover(function() {
			//鼠标移动DIV上停止
			clearInterval(sliding);
		},function(){
			//离开继续调用
			sliding=setInterval(Marquee,_speed);
		});
	}
});
</script>
