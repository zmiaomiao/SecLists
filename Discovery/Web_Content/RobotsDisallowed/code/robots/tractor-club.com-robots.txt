$(document).on('submit', '#reg-form', function(event) {
	event.preventDefault();	
	var formD = {};
	var $fData = $('#reg-form input.require');
	$.each($fData, function(index, val) {
		 formD[""+$(val).attr('name')+""]=$(val).val()
	});		
	$.ajax({
		url: '/ajax/users/register',
		type: 'POST',
		dataType: 'json',
		data: {rcv:formD,json:1},
		success:function(data){
			if(data.code == -1){
				var str="";
				$('.input-wrp p').hide();
				$('.input-wrp input').removeClass('unvalidated');
			$.each(data.msg, function(index, val) {
				$("#reg-form INPUT[name='"+index+"']").siblings('p').text(val);
				
				str += " #reg-form INPUT[name='"+index+"']"+',';
			});	
				str = str.slice(0,-1);				
				$(str).siblings('p').show();		
				$(str).addClass('unvalidated');			
			}else if(data.code==1)
				$('#signUp').html(data.msg)			
			else
				$('#signUp .error-msg').html(data.msg)	
		}
	})	
	
});
