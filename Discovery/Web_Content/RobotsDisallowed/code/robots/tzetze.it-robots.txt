	/* Validate e-mail address */
	function validateEmailAddress(email) 
		{ 
			var emailPattern = /^[a-z0-9._\-][a-z0-9._\-\+]*@[a-z0-9][a-z0-9.\-]*[\.]{1}[a-z]{2,4}$/i;  
			return emailPattern.test(email);  
		} 

	/* Validate first chapter form */
	function validateNewsletterForm(formObj)
		{
			var errMsg = "";
			if(formObj.email.value == "" || !validateEmailAddress(formObj.email.value))
				errMsg = errMsg + "- Indirizzo e-mail non valido\n";
			if(formObj.name.value == "")
				errMsg = errMsg + "- Inserisci nome e cognome!\n";
			if(formObj.privacy.checked == false)
				errMsg = errMsg + "- Devi accettare l'informativa sulla privacy!";

			if(errMsg != "")
			{
				alert(errMsg);
				return false;
			}
			else
				return true;
		}
