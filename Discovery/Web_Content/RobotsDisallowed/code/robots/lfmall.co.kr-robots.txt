function jfReplaceCmd(cmd){
	var queryStr = "cmd=main&af=BR04&BROWSER=";

	var startIdx = queryStr.indexOf("cmd=");
	var endIdx = queryStr.indexOf("&", queryStr.indexOf("cmd=")) < 0 ? queryStr.length: queryStr.indexOf("&", queryStr.indexOf("cmd=")) ;

	var tmpQuery1 = queryStr.substring(0, startIdx);
	var tmpQuery2 = queryStr.substring(endIdx, queryStr.length);

	return tmpQuery1 + cmd + tmpQuery2;
}
