$(function(){
	var aryOnlineShop = [];
	aryOnlineShop.push('UR');
	aryOnlineShop.push('DOORS');
	aryOnlineShop.push('ROSSO');
	aryOnlineShop.push('kagure');
	aryOnlineShop.push('KBF');
	aryOnlineShop.push('SOP');
	aryOnlineShop.push('SLB');
	aryOnlineShop.push('SMELLY');
	aryOnlineShop.push('RODESKO');
	aryOnlineShop.push('URSELECT');

	// タイプ選択
	$('#categoryTab li').click(
		function () {
			if(!("request_count" in window)){
				request_count = 0;
			}

			if(request_count > 0){
				return false;
			}
			//タブ連打対策：スライダー状態変数をインクリメント（静的変数）
			request_count+=2;
			var selected = $(this).val();
			$('input[name="val[ProductsType]"]:hidden').val(selected);

			var clickIndex = $('#categoryTab li').index(this);

			$('#categoryTab li a img').each(function() {
				var beforeSrc = $(this).attr('src');
				if ($('#categoryTab li a img').index(this) == clickIndex) {
					// ※ smartRollover.jsも解除する必要があるため、removeAttrを追加
					$(this).removeAttr('onmouseover');
					$(this).removeAttr('onmouseout');
					$(this).unbind('hover');

					$(this).attr('src', beforeSrc.replace('_off.png', '_on.png'));

				} else {

					$(this).attr('src', beforeSrc.replace('_on.png', '_off.png'));
					$(this).hover(
						function () { $(this).attr('src', beforeSrc.replace('_off.png', '_on.png')); },
						function () { $(this).attr('src', beforeSrc.replace('_on.png', '_off.png')); }
					);

				}

				$(".boxProduct h3").tile();
				$(".boxProduct p.des").tile();
			});

			$("#rankingItemWrap").fadeOut(200,
				function () {
					$(".slider_ranking").html('').css('margin-left', '0');

					UnbindSlider("#btnSlideRight_ranking","#btnSlideLeft_ranking");
					setSlider(".slider_ranking","rankingitem","#btnSlideRight_ranking","#btnSlideLeft_ranking",500,21,{"page": 0});

					$("#rankingItemWrap").fadeIn(300);
					request_count--;
				}
			);

			$("#newItemWrap").fadeOut(200,
				function () {
					$(".slider_new").html('').css('margin-left', '0');

					UnbindSlider("#btnSlideRight_new","#btnSlideLeft_new");
					setSlider(".slider_new","new","#btnSlideRight_new","#btnSlideLeft_new",500,21,{"page": "0"});

					$("#newItemWrap").fadeIn(300);
					request_count--;
				}
			);

			var d = new Date();
			var previsit = d.toGMTString();
			d.setTime(d.getTime() + 30*24*60*60*1000); //30日後
			expire = d.toGMTString();

			if(selected == 1 || selected == 2){
					setCookie("tabProductsType",selected,expire);
			}else if(selected == 0){
					setCookie("tabProductsType",'',expire-1);
			}

			// (左メニュー)オンラインショップのURLの変更
			var url_type = '';
			if(selected == 0) {
				url_type = 'All/';
			} else if(selected == 9) {
				// url_type = 'KidsHome/';
			}
			$.each(aryOnlineShop,function(i) {
				setHref("shopLink_"+aryOnlineShop[i],"/"+aryOnlineShop[i]+"/"+url_type+"?link=1");
			});

			var selectedname = {};
			selectedname['0'] = '';
			selectedname['1'] = 'Men';
			selectedname['2'] = 'Women';
			selectedname['9'] = 'KidsHome';
			var newitemlink = '/search/?item=new&type=';
			var rankingitemlink = '/ranking/?ProductsType=';

			$('.NewItemList').attr('href',newitemlink + selectedname[selected]);
			$('.RankingItemList').attr('href',rankingitemlink + selected);
			return false;

		}
	);

	//cookieを保存
	function setCookie(item, value, d) {
		document.cookie = item + "=" + value + ";expires=" + d + ";path=" + URL_BASE;
	}

	//Hrefを設定
	function setHref(id, url) {
		$("#" + id).attr("href", url);
	}

});
</script>
