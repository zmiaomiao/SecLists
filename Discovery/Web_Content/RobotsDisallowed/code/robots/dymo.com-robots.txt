function updateModelSelectBox(family_select_element, categoryJSON) {
	
	var modelSelectElem = dojo.query(family_select_element).next('.product_model_select')[0];
    if(modelSelectElem != undefined && modelSelectElem != null)	{
    	dojo.empty(modelSelectElem);
    	
		dojo.create("option", {innerHTML: 'Model', value:''}, modelSelectElem);
		
		
		if(categoryJSON != undefined && categoryJSON != null && categoryJSON.category != undefined && categoryJSON.category != null) {
			dojo.forEach(categoryJSON.category.subCategories,function(category) {
				if(category.identifier == family_select_element.value) {
					dojo.forEach(category.catalogEntries,function(catEntry) {
						var URL = catEntry.URL + "#tab=" + "Support";
						dojo.create("option", {innerHTML: catEntry.description.name, value:URL}, modelSelectElem);
					});
				}
			});
		}
    	dojo.query(modelSelectElem).removeAttr("disabled");
    }
}


dojo.addOnLoad(function(){
		
		MessageHelper.setMessage("REGISTRABLE_PRODS_SELECT_TYPE", "Please select a product type");
		
		
		MessageHelper.setMessage("REGISTRABLE_PRODS_SELECT_MODEL", "Please select a product model");
		
	
	var categoryJSONElem = dojo.byId('softwareDriversCategoryJSON');
	var categoryJSON = eval('(' + categoryJSONElem.innerHTML + ')');
	
	
	dojo.query("select.product_family_select").forEach(function(selectTag){
		dojo.empty(selectTag);
		
		dojo.create("option", {innerHTML: 'Product Type', value:''}, selectTag);
		
		
		if(categoryJSON != undefined && categoryJSON != null && categoryJSON.category != undefined && categoryJSON.category != null) {
			dojo.forEach(categoryJSON.category.subCategories,function(category) {
				if(category.catalogEntries.length > 0) {
					dojo.create("option", {innerHTML: category.description.name, value:category.identifier}, selectTag);
				}
			});
		}
		
		dojo.connect(selectTag, 'onchange', function (e){
			updateModelSelectBox(this, categoryJSON);
		});
	});
	
	dojo.query(".software_drivers_find_download_btn").forEach(function(button){
		
		dojo.connect(button, 'onclick', function (e){
			var familySelectBox = dojo.query(this).prevAll('.product_family_select')[0];
			var modelSelectBox = dojo.query(this).prevAll('.product_model_select')[0];
			var isButtonDisabled = dojo.query(modelSelectBox).attr('disabled');
			if(familySelectBox.value != null && familySelectBox.value == '') {
				MessageHelper.formErrorHandleClient(familySelectBox.id, MessageHelper.messages["REGISTRABLE_PRODS_SELECT_TYPE"]);
				return false;
			}
			if(modelSelectBox.value != null && modelSelectBox.value == '') {
				MessageHelper.formErrorHandleClient(modelSelectBox.id, MessageHelper.messages["REGISTRABLE_PRODS_SELECT_MODEL"]);
				return false;
			}
			if(isButtonDisabled != 'true') {
				var modelSelectBoxId = modelSelectBox.id;
				if(modelSelectBoxId == 'SoftwareDriverModelSelectionHeader'){
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads Header')  ;
				}else if(modelSelectBoxId == 'SoftwareDriverModelSelectionFooter'){
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads Footer')  ;
				}else{
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads')  ;
				}
				location.href = modelSelectBox.value;
			}
		});
	});
});

function updateModelSelectBox(family_select_element, categoryJSON) {
	
	var modelSelectElem = dojo.query(family_select_element).next('.product_model_select')[0];
    if(modelSelectElem != undefined && modelSelectElem != null)	{
    	dojo.empty(modelSelectElem);
    	
		dojo.create("option", {innerHTML: 'Model', value:''}, modelSelectElem);
		
		
		if(categoryJSON != undefined && categoryJSON != null && categoryJSON.category != undefined && categoryJSON.category != null) {
			dojo.forEach(categoryJSON.category.subCategories,function(category) {
				if(category.identifier == family_select_element.value) {
					dojo.forEach(category.catalogEntries,function(catEntry) {
						var URL = catEntry.URL + "#tab=" + "Support";
						dojo.create("option", {innerHTML: catEntry.description.name, value:URL}, modelSelectElem);
					});
				}
			});
		}
    	dojo.query(modelSelectElem).removeAttr("disabled");
    }
}


dojo.addOnLoad(function(){
		
		MessageHelper.setMessage("REGISTRABLE_PRODS_SELECT_TYPE", "Please select a product type");
		
		
		MessageHelper.setMessage("REGISTRABLE_PRODS_SELECT_MODEL", "Please select a product model");
		
	
	var categoryJSONElem = dojo.byId('softwareDriversCategoryJSON');
	var categoryJSON = eval('(' + categoryJSONElem.innerHTML + ')');
	
	
	dojo.query("select.product_family_select").forEach(function(selectTag){
		dojo.empty(selectTag);
		
		dojo.create("option", {innerHTML: 'Product Type', value:''}, selectTag);
		
		
		if(categoryJSON != undefined && categoryJSON != null && categoryJSON.category != undefined && categoryJSON.category != null) {
			dojo.forEach(categoryJSON.category.subCategories,function(category) {
				if(category.catalogEntries.length > 0) {
					dojo.create("option", {innerHTML: category.description.name, value:category.identifier}, selectTag);
				}
			});
		}
		
		dojo.connect(selectTag, 'onchange', function (e){
			updateModelSelectBox(this, categoryJSON);
		});
	});
	
	dojo.query(".software_drivers_find_download_btn").forEach(function(button){
		
		dojo.connect(button, 'onclick', function (e){
			var familySelectBox = dojo.query(this).prevAll('.product_family_select')[0];
			var modelSelectBox = dojo.query(this).prevAll('.product_model_select')[0];
			var isButtonDisabled = dojo.query(modelSelectBox).attr('disabled');
			if(familySelectBox.value != null && familySelectBox.value == '') {
				MessageHelper.formErrorHandleClient(familySelectBox.id, MessageHelper.messages["REGISTRABLE_PRODS_SELECT_TYPE"]);
				return false;
			}
			if(modelSelectBox.value != null && modelSelectBox.value == '') {
				MessageHelper.formErrorHandleClient(modelSelectBox.id, MessageHelper.messages["REGISTRABLE_PRODS_SELECT_MODEL"]);
				return false;
			}
			if(isButtonDisabled != 'true') {
				var modelSelectBoxId = modelSelectBox.id;
				if(modelSelectBoxId == 'SoftwareDriverModelSelectionHeader'){
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads Header')  ;
				}else if(modelSelectBoxId == 'SoftwareDriverModelSelectionFooter'){
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads Footer')  ;
				}else{
					cmCreateElementTag(modelSelectBox.options[modelSelectBox.selectedIndex].text,'Software Drivers & Downloads')  ;
				}
				location.href = modelSelectBox.value;
			}
		});
	});
});
