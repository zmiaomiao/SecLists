function validateForm() {
	var name    = document.getElementById('name').value;
	var email   = document.getElementById('email').value;
	var subject = document.getElementById('subject').value;
	var message = document.getElementById('message').value;

	var type = document.getElementById('type').value;


	var formOK  = document.getElementById('formOK');
	var formErr = document.getElementById('formErr');

	formOK.style.display  = 'none';
	formErr.style.display = 'none';

	if (name == '' || email == '' || message == '' || type == 'I am a...' || subject == 'Subject...') {
		formErr.innerHTML = 'Please fill out all the required fields';
		formErr.style.display = 'block';
		return false;
	}

	if (!validateEmail(email)) {
		formErr.innerHTML = 'Your email address is invalid. Please re-type it and try again.';
		formErr.style.display = 'block';
		return false;
	}

	if (message.length < 20) {
		formErr.innerHTML = 'Your message is too short.';
		formErr.style.display = 'block';
		return false;
	}

	return true;
}

function validateEmail(address) {
	address = trim(address);
	var reg = /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/;
	if(reg.test(address) == false) return false;
	return true
}

function trim(str) {
	var newstr;
	newstr = str.replace(/^\s*/, "").replace(/\s*$/, "").replace(/\s{2,}/, " ");
	return newstr
