    function initSubscribe()
    {

        var errorTemplate = new Template('<div class="validation-advice signup-modal-error">#{message}</div>');

        $j('#signup-modal-email').focus();

        $j('#modal-subscribe-form').submit(function(e) {
            e.preventDefault();

            $j('#signup-modal-error-container').addClass('hidden');

            if (!modalForm.validator.validate()) {
                return false;
            }

            // copy email address to step 2
            $j('.newsletter .step-2 input[name="emailAddress"]').val( $j('#signup-modal-email').val() );

            // disable buttons
            $j('#signup-modal-content .step-1 button.shop-now-btn').prop("disabled",true);

            var request = $j.ajax({
                type: 'POST',
                url: 'http://www.rebeccaminkoff.com/basignup/subscribe/',
                data: $j('#modal-subscribe-form').serialize()
            });

            request.done(function(data) {
                data = data ? data.evalJSON(true) : '';

                if (!data || data && typeof data.error === 'undefined') {
                    initRegistration();

                    // Hides step 1, shows step 2
                    $j('#signup-modal-content .step-1').addClass('hidden');
                    $j('#signup-modal-content .step-2').addClass('visible');

                    // insert coupon code into HTML for step 3
                    $j('.newsletter .step-3 #code').html(typeof data.coupon === 'undefined' ? '' : data.coupon);

                    // Remove disabled property from button
                    $j('#signup-modal-content .step-1 button.shop-now-btn').prop("disabled",false);
                } else {
                    $j('#signup-modal-error-container').html(errorTemplate.evaluate({message : data.error}));
                    $j('#signup-modal-error-container').removeClass('hidden');
                    $j('#modal-subscribe-form input').addClass('validation-failed');
                    $j('#signup-modal-content button.shop-now-btn').prop("disabled",false);
                }
            });
        });
    }
