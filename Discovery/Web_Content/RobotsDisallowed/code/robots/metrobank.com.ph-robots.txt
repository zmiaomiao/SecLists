            && (!subjectVal || !subjOther)
            && !message) {

                window.location.replace(window.location);
            }

                url: "web/service/emailForm.ashx?a="
                    + agree + "&e="
                    + existEmp + "&c="
                    + captchaVal + "&t="
                    + title + "&ef="
                    + empFname + "&em="
                    + empMname + "&el="
                    + empLname + "&ea="
                    + emailAdd + "&s="
                    + subjectVal + "&so="
                    + subjOther + "&m="
                    + message + "",
                success: function (data) {
                    if (data == "success") {
                        $("#chkAgreement").attr("checked", false);

                        $("#txtSubjectOthers").attr("background", "#aaa");

                        $('input[name="rdoExistMbtc"]').attr("checked", false);
                        $("#drpTitleVal").text('');
                        $("#txtFnameVal").text('');
                        $("#txtMnameVal").text('');
                        $("#txtLnameVal").text('');
                        $("#txtEmailAddVal").text('');
                        $("#drpSubjectVal").text('');
                        $("#txtSubjectOthersVal").text('');
                        $("#txtMessageVal").text('');

                        $("#emailVerifyNextHidden").trigger("click");
                        $("button[type='reset']").trigger("click");
                        captcha.val('');
                        $('.captchaSpan').trigger("click");
                    }
                    else {
                        $('.errorMessage').text(data);
                        captcha.val('');
                        $('.captchaSpan').trigger("click");
                        $("#emailVerifyErrorHidden").trigger("click");
                        	$("button[type='reset']").trigger("click");
                    }
                },
                error: function (data) {
                       alert(data);
                       $('.errorMessage').text(data);
                       captcha.val('');
                       $('.captchaSpan').trigger("click");
                       $('label.error').hide();
                       $("#chkAgreement").attr("checked", false);
                  	  $("button[type='reset']").trigger("click");
                }
            })
        });

            });

		doc.action = linked;
		if (target=="window"){

		}else{
			doc.target = "_self";
		}

}
