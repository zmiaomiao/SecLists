            function setCookie(cName, exDays) {
                var exDate = new Date();
                exDate.setDate(exDate.getDate() + exDays);
                var cValue = (exDays == null) ? "" : exDate.getTime(); //exDate.toUTCString()
                document.cookie = cName + "=" + cValue;
            }

            function getCookieExpiryTime(c_name) {
                var c_value = document.cookie;
                var c_start = c_value.indexOf(c_name + "=");
                var c_end;

                if (c_start == -1) {
                    c_value = null;
                }
                else {
                    c_start = c_value.indexOf("=", c_start) + 1;
                    c_end = c_value.indexOf(";", c_start);
                    if (c_end == -1) {
                        c_end = c_value.length;
                    }
                    c_value = parseInt(c_value.substring(c_start, c_end));
                }
                return c_value;
            }

            function checkCookieExpired() {
                var exTime = getCookieExpiryTime("delay_app_reminder");
                var now = (new Date()).getTime();
                return (now > exTime);
            }

            function setCookieIfExpired() {
                if (checkCookieExpired()) {
