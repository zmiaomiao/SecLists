function setActiveMenu(arr, crtPage) //active state
{
var found = false;
var myinnerHTML = "";

	for (var i=0; i < arr.length; i++)
	{
	if(arr[i].innerHTML != "") myinnerHTML = arr[i].innerHTML;
	found = false;
	if(crtPage.indexOf('/home/') != -1 || crtPage.indexOf('/for-home/') != -1)		//for home
	{
		if(myinnerHTML == "For Home") found = true;
	}

	if(crtPage.indexOf('/business/') != -1 || crtPage.indexOf('/small-business/') != -1 || crtPage.indexOf('/security-products/') != -1 || 
		crtPage.indexOf('/enterprise/') != -1 || crtPage.indexOf('/mid-enterprise/') != -1 || crtPage.indexOf('/service-providers/') != -1)	//for business 
	{
		if(myinnerHTML == "For Business") found = true;	
	}

	if(crtPage.indexOf('/security-intelligence/') != -1)		//security intelligence
	{
		if(myinnerHTML == "Security Intelligence") found = true;
	}

	if(crtPage.indexOf('/technology-innovation/') != -1)		//why trend
	{
		if(myinnerHTML == "Why Trend Micro") found = true;
	}
	if(crtPage.indexOf('esupport.trendmicro.com') != -1)		//support
	{
		if(myinnerHTML == "Support") found = true;
	}
	if (found)
	{
		if (arr[i].parentNode.tagName != "DIV")
			arr[i].parentNode.className = "main-cat nav-active";
	}		
	}
}

	if (document.getElementById("drop-nav") !=null )

<script type="text/javascript">
var Open_Close_Fix = function( button_classname, open_text, close_text )
{
	this.open_state = false;
	this.open_text = '.' + open_text;
	this.close_text = '.' + close_text;
	this.button_classname = '.' + button_classname;
	
	// Setup a pointer back to this class for use in events
	var self = this;
	
	// Bind the click event on our buttons to this class's click() function
	$(this.button_classname).unbind( 'click' ).bind( 
		'click', 
		function() {
			self.click( open_text, close_text );
		}
	);
	
}

Open_Close_Fix.prototype.click = function( open_text, close_text )
{
	if( !this.open_state )
	{
		this.classname = '.' + 'pane';
		this.open_all_tabs();
		this.classname = '.' + 'pane-title';
		this.open_all_titles();
		var allp = document.getElementsByTagName('p');
		var text = "";
		var pchildren; 
		for (var i=0; i<allp.length; i++) {
			pchildren = allp[i].innerHTML;
			if (pchildren == open_text) {
				document.getElementsByTagName('p')[i].innerHTML = close_text;
			}
		}
	}
	else
	{
		this.classname = '.' + 'pane';
		this.close_all_tabs();
		this.classname = '.' + 'pane-title';
		this.close_all_titles();
		var allp = document.getElementsByTagName('p');
		var text = "";
		var pchildren; 
		for (var i=0; i<allp.length; i++) {
			pchildren = allp[i].innerHTML;
			if (pchildren == close_text) {				
				document.getElementsByTagName('p')[i].innerHTML = open_text;
			}
		}		
	}
	
	this.open_state = !this.open_state;
}


Open_Close_Fix.prototype.open_all_tabs = function()
{
	$(this.classname).css( 'display', 'block' );
}

Open_Close_Fix.prototype.open_all_titles = function()
{
	$(this.classname).addClass("current");
}

Open_Close_Fix.prototype.close_all_tabs = function()
{
	$(this.classname).css( 'display', 'none' );
}
Open_Close_Fix.prototype.close_all_titles = function()
{
	$(this.classname).removeClass("current");

}
</script>
