<script>
    $('#tickerSubscribe').on("click",function()
    {
        var email = $('#tickerEmail').val();
        var ticker = $('#ticker_subscribe').text();
        if(email == "")
        {
            $('#tickerEmail').focus();
            $('#tickerEmail').tooltip({ title : 'Enter Email Address.' });
            $('#tickerEmail').tooltip('show');
            setTimeout(function(){ $('#tickerEmail').tooltip('destroy'); }, 1000);
            return false;
        }

        var atPos = email.indexOf("@");
        var dotPos = email.lastIndexOf(".");
        if (atPos < 1 || ( dotPos - atPos < 2 ))
        {
            $('#tickerEmail').focus();
            $('#tickerEmail').tooltip({ title : 'Invalid Email Address.' });
            $('#tickerEmail').tooltip('show');
            setTimeout(function(){ $('#tickerEmail').tooltip('destroy'); }, 1500);
            return false;
        }

        $.ajax({
            type: "POST",
            url:  "/post/ticker_subscribe.php",
            data: "email="+email+"&ticker="+ticker,
            success: function(msg)
            {
                if(msg == 'insert')
                {
                    $('#tickerEmail').val('');
                    $('#tickerEmail').tooltip({ title : 'You are Now Subscribed to '+ticker });
                    $('#subscribeModal').modal('hide');
                }
                else
                {
                    $('#error_ticker').html('Something Goes Wrong');
                }
            }
        });
    });
</script><!-- /container -->
