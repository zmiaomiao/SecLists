function hookShare()

function closeShareOptions()
{

	{

var isMobileLayout = false;
function toggleShareOptions(sharearrow, alink, atitle)
{
	if(shareArrowActive == $(sharearrow))
	{
		shareArrowActive = null;
	}
	else
	{
		shareArrowActive = $(sharearrow);
	}
	
	if(typeof atitle === 'undefined')
		atitle = '';
		
	if(shareArrowLink==alink)
	{
		shareArrowLink = '';
		closeShareOptions();
		return;	
	}
	
	shareArrowLink = alink;
	shareArrowTitle = atitle;
	
	if(!isMobileLayout)
	{
		if(timShare!=null)
		{
			clearInterval(timShare);	
			$(".global_share").hide();
			$(document).off("click");
		}
		timShare = setInterval(hookShare, 50);
	}
	else
	{
		
		$('body').bind('touchmove', noTouch);
		$(".mobileShareBG").removeClass('hide');
		$(".mobileShareBG").show();
		$(".mobileShareBG").animate({opacity:0.5});
		$(".mobileSharePanel").removeClass("hide");
		$(".mobileSharePanel").animate({height:"95px"});	
	}
}

function validateEmail(email) {
    var re = /^([\w-]+(?:\.[\w-]+)*)@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$/i;
    return re.test(email);
}

function isNumber(evt) {
    evt = (evt) ? evt : window.event;
    var charCode = (evt.which) ? evt.which : evt.keyCode;
    if (charCode > 31 && (charCode < 48 || charCode > 57)) {
        return false;
    }
    return true;
}
