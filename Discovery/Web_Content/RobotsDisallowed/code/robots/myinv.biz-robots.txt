  function initBallAnimation() {
  
    try {
    
        nodes = d3.range(numBalls).map(function(d,i) { 
            
            return {radius: Math.random() * 12 + 4}; 
        });
        root = nodes[0];	
    
        root.radius = 0;
        root.px = width*0.5;root.py = height*0.5;
        root.fixed = true;
        
        force = d3.layout.force()
            .gravity(0.05)
            .charge(function(d, i) { return i ? 0 : -2500; })
            .nodes(nodes)
            .size([width, height]);
        
        force.start();
    
        // SVG ///////////////////////////////////////////////////
    
        svg = d3.select("#WEAnimation").append("svg")
            .attr("width", width)
            .attr("height", height);
        
        // "ball" light effect:
        var gradient = svg.append("svg:defs")
            .append("svg:radialGradient")
            .attr("id", "gradient")
            .attr("cx", "25%")
            .attr("cy", "25%")
            .attr("fx", "25%")
            .attr("fy", "25%")
            .attr("r", "25%")
            ;
        
        gradient.append("svg:stop")
            .attr("offset", "0%")
            .attr("stop-color", "#FFFFFF")
            .attr("stop-opacity", 0.75);
        
        gradient.append("svg:stop")
            .attr("offset", "100%")
            .attr("stop-color", "#FFFFFF")
            .attr("stop-opacity", 0);
            
        // all balls
        svg.selectAll("circle")
            .data(nodes.slice(1))
          .enter().append("circle")
            .attr("r", function(d) { return d.radius; })
            .style("fill", function(d, i) { return ar_color[i % ar_color.length]; });
        
        // light effect on each ball
        svg.selectAll("x1")
            .data(nodes.slice(1))
          .enter().append("circle")
            .attr("r", function(d) { return d.radius; })
            .style("fill", "url(#gradient)");
            
            
        force.on("tick", function(e) {
          var q = d3.geom.quadtree(nodes),
              i = 0,
              n = nodes.length;
        
          while (++i < n) q.visit(collide(nodes[i]));
        
          svg.selectAll("circle")
              .attr("cx", function(d) { return d.x; })
              .attr("cy", function(d) { return d.y; });
        });
    
      colorTextHighlight();
      
	} catch (ex) { ; }
      
  } // end ballAnimation()

  function collide(node) {
    var r = node.radius + 16 + 8,
        nx1 = node.x - r,
        nx2 = node.x + r,
        ny1 = node.y - r,
        ny2 = node.y + r;
    
    // wall collision:
    /** /
      var minWallR = node.radius; if (minWallR < 10) minWallR = 10;
      if (node.x-minWallR < wallDist) node.x = wallDist+minWallR;
      if (node.y-minWallR < wallDist) node.y = wallDist+minWallR;
      if (node.x+minWallR > width-wallDist) node.x = width-wallDist-minWallR;
      if (node.y+minWallR > height-wallDist) node.y = height-wallDist-minWallR;
      /**/
    
    return function(quad, x1, y1, x2, y2) {
      if (quad.point && (quad.point !== node)) {
        var x = node.x - quad.point.x,
            y = node.y - quad.point.y,
            l = Math.sqrt(x * x + y * y),
            r = (node.radius + quad.point.radius)*1.32;
        if (l < r) {
          l = (l - r) / l * .5;
          node.x -= x *= l;
          node.y -= y *= l;
          quad.point.x += x;
          quad.point.y += y;
        }
      }
      return x1 > nx2 || x2 < nx1 || y1 > ny2 || y2 < ny1;
    };
  }

  function colorTextHighlight() {
    var $texts = $('#WEWhyAnswersTexts b');
    var $svg = $('svg');
    var svgX = $svg.offset().left, svgY = $svg.offset().top;
  
    var lastY = 0;

    $texts.each(function(){ // coordinates, dimensions and colors

        $this.data("w", $this.width()+parseInt($this.css('margin-left'))+parseInt($this.css('margin-right'))).data("h", $this.height()+parseInt($this.css('margin-top'))+parseInt($this.css('margin-bottom')));
        var x = $this.offset().left-svgX;
        var y = $this.offset().top-svgY;
        $this.data("x", x).data("y", y);
        if (lastY && lastY !== y) currColorInd += 1; // change color for each line
        lastY = y;
        $this.data('color', colors_rgb[currColorInd % 3]);
    });
  
    $texts.each(function(){ // calculate centers
        var $this = $(this);
        $this.data("xw", $this.data('x')+$this.data('w')*0.5).data("yh", $this.data('y')+$this.data('h')*0.5);

        var mx = ev.pageX - $(this).offset().left;
        var my = ev.pageY - $(this).offset().top;
        
      svgMouseMove(mx, my); // affect ball animation
      
        var maxDist = width;
        
        var colorText = ['', '', ''];
        $texts.each(function(){
            var $this = $(this);
            var dx = mx - $this.data('xw'), dy = my - $this.data('yh');
            var relDist = Math.sqrt(dx*dx + dy*dy)/200;
            if (relDist > 1) relDist = 1;
            
            var colorFar = 240; // color (grey composant) when mouse far
            var colorClose = $this.data('color');
            var color = '#';
            for (var i=0; i<3; i++) {
              var colorPart = parseInt(colorClose.substr(1+i*2, 2), 16); // ex FF => 255
              colorText[i] = parseInt(colorPart * (1-relDist) + colorFar * relDist).toString(16); // colorClose if mouse is over the text, colorFar when mouse is far
              color += ((colorText[i].length < 2) ? '0' : '') + colorText[i];
            }
          
            $this.css({'color' : color});
        });
    
    });
  
    $('#WEAnimation').mouseleave(function(ev){
      svgMouseLeave();
    });
  
  }
