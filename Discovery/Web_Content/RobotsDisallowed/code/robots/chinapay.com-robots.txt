    function checkSubmit(form) {
        if (form.account.value.isEmpty()) {
            alert("请输入帐户名");
            form.account.focus();
            return false;
        }
        if (form.password.value.isEmpty()) {
            alert("请输入密码");
            form.password.focus();
            return false;
        }
        if (!form.password.value.isChar()) {
            alert("密码无效！");
            form.password.focus();
            return false;
        }
        if (form.password.value.trim().getLength() < 6
                || form.password.value.trim().getLength() > 20) {
            alert("密码长度至少为6位,至多为20位");
            form.password.focus();
            return false;

            alert("验证码长度必须为4位");
