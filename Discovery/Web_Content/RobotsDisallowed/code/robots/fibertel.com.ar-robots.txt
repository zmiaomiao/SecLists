<script>
    var NavidadController = {

    $holders : new Array(),
	$totalParticles : 0,
	$intervalMove : 0,
	$intervalParticlesGenerator : 0,
	$speedIntervalParticlesGenerator: 200,
	
	$speedIntervalMove: 22,
	$particles : new Array(),
	$count: 0,
	
	
	$speedY_Max: 5,
	$speedY_Min: 2,
	
	$speedX_Max: 2,
	$speedX_Min: 1,
	
	$maxItemsInterface : 30,
	

	
	init: function() {
		
		
		
		this.$holders.push($('body .navidadContainerLeft'));
		this.$holders.push($('body .navidadContainerRight'));
		this.$holders.push($('body .navidadContainerCenter'));
		
		
		/*
		console.log( this.$holders[0] );
		console.log( this.$holders[1] );
		*/
		this.$intervalMove = setInterval(this.doIntervalMove, this.$speedIntervalMove);
		
		this.$intervalParticlesGenerator = setInterval(this.doIntervalParticle, this.$speedIntervalParticlesGenerator);
		
		
		this.setButtonsFunctions();
		
	},
	
	
	clear : function ()
	{
		//console.log( "clear!" );
		clearInterval(this.$intervalMove);
		this.$intervalMove = 0;
		
		clearInterval(this.$intervalParticlesGenerator);
		this.$intervalParticlesGenerator = 0;
		
	},
	
	
	doIntervalMove : function()
	{
		
		for (var i=0; i<NavidadController.$particles.length; i++)
		{
			
			var particle = NavidadController.$particles[i];
			
			if(particle.positionX>particle.widthContainer || particle.positionX<0 || particle.positionY>particle.heightContainer)
			{
				NavidadController.clearParticle(i);
				
			}else
			{
				NavidadController.doMoveParticles(particle);
			}
			
		}
	},
	
	doMoveParticles: function(particle)
	{
				var particleHolder = $("."+particle.name);
				
				
				particle.positionY += particle.speedY;
				particle.positionX += particle.speedX;
			
				particleHolder.css("top",particle.positionY+"px");
				particleHolder.css("left",particle.positionX+"px");
				
	},
	
	
	clearParticle : function (indexOfArray)
	{
		var particle = this.$particles[indexOfArray];
		var particleHolder = $("."+this.$particles[indexOfArray].name);
		particleHolder.remove();
		
		
		//console.log("particle.name:"+ (particle.name));
		particle = undefined;
		
		this.$particles.splice(indexOfArray, 1);
		
		//console.log("this.$particles.lenght:"+ (this.$particles.length));
	},
	
	doIntervalParticle : function()
	{
		if(NavidadController.$particles.length >= NavidadController.$maxItemsInterface)
		{
			//console.log( "hay muchas particulas!" );
			return;
		}
		NavidadController.createParticle()
	},
	
	createParticle: function()
	{
		this.$count++;
		//console.log( "doInterval!" );
		this.addParticle();
	},
	
	setButtonsFunctions: function ()
	{
		//console.log( "setButtonsFunctions!" );
		$( ".qa1Button" ).click(function() {
		  NavidadController.clear();
		});
	},
	
	
	addParticle : function()
	{
		
		var $tmpParticle = new Object();
		$tmpParticle.id = this.$count;
		$tmpParticle.name = "particle"+this.$count;
		$tmpParticle.speedY = Math.floor((Math.random() * (this.$speedY_Max-this.$speedY_Min))+this.$speedY_Min);
		$tmpParticle.speedX = Math.floor((Math.random() * (this.$speedX_Max-this.$speedX_Min))+this.$speedX_Min);
		
		$tmpParticle.positionY = -30;
		
		
		$tmpParticle.positionX = 0;
		
		$tmpParticle.elementInterface = "sarasa.png"+this.$count;
		
		
		
		this.$particles.push($tmpParticle);
		this.$totalParticles = this.$particles.length;
		
		this.drowParticle(this.$particles[this.$particles.length-1]);
		this.doMoveParticles(this.$particles[this.$particles.length-1]);
		
	},
	
	onMouseOverParticle: function (idParticle)
	{
		
		
		for (var i=0; i<this.$particles.length; i++)
		{
			
			var particle = this.$particles[i];
			
				
				if(particle.id == idParticle)
				{
				particle.speedY+=2;
				particle.speedX=particle.speedX*-1;
				return;
				/*
				this.clearParticle(i);
				this.createParticle();
				*/
				}			
		}
	},
	
	drowParticle : function (particle)
	{		
		var indxParticleImage = Math.floor((Math.random() * 3));			
		var strParticle = "<div class='particle "+particle.name+" snow"+indxParticleImage+"'  onmouseover='NavidadController.onMouseOverParticle("+particle.id+")'></div>";
		var holder;	
		
		/* se setea random en que contenedor se aplica el copo de nieve*/
		var indxContainer = Math.floor((Math.random() * 3));
		holder = this.$holders[indxContainer];
		/*
		if(this.$count % 2 == 0)
		{
			holder = this.$holders[0];
		}else
		{
			holder = this.$holders[1];
		}
		*/
		/*holder = this.$holders[0];*/
		
		particle.holder = holder;
		
		var widthContainer = holder.css("width").replace("px", "");
		var heightContainer = holder.css("height").replace("px", "");
		
		particle.widthContainer = widthContainer;
		particle.heightContainer = heightContainer;
		particle.positionX = Math.floor((Math.random() * widthContainer))-50;
		
		holder.html(holder.html()+strParticle);		
	}		
};
