                var prev_btn = $(".frPage .prev");

                jQuery.checkPos = function(curCount){
                    if( curCount == 1 ){
                        prev_btn.addClass("hot");
                        goMask.css({"right":"14px"});
                        next_btn.removeClass("hot");
                    }else if( curCount == totalCount ){
                        next_btn.addClass("hot");
                        prev_btn.removeClass("hot");
                        goMask.css({"right":"0"});
                    }else{
                        prev_btn.removeClass("hot");
                        next_btn.removeClass("hot");
                        goMask.css({"right":"-14px"});
                    }
                };
                if( totalCount < 1 ){
                    $(".frPage").hide();
                }else if( totalCount == 1){
                    prev_btn.hide();
                    next_btn.hide();
                }
                else{
                    jQuery.checkPos(curCount);
                }

                $("[flash6=flash6] .next").bind("click",function(){
                    curCount = curCount > (totalCount - 1) ? 1 : curCount+1;
                    $("#curcount").text(curCount);
                    jQuery.checkPos(curCount);
                })
