    }
  });
  
  var requesturi = 'http://version.doit.im/json/versions.json';
  
  $.ajax({
      async: false,
      url: requesturi,//跨域请求的URL
      dataType: 'jsonp',
      jsonpCallback: "success_jsonpCallback",
      success: function (result) {
        var android = result.android;
        var $android = $('#download_android');
        $android.find('.name').text(android.version + ' for Android');
        $android.find('a').attr('href',android.download_url);
        var android_intl = result.android_intl;
        var $android_intl = $('#download_android_intl');
        $android_intl.find('.name').text(android_intl.version + ' for Android');
        $android_intl.find('a').attr('href',android_intl.download_url);
        var windows = result.win7;
        var $windows = $('#download_windows');
        $windows.find('.name').text(windows.version + ' for Windows');
        $windows.find('a').attr('href',windows.download_url);
        var iphone = result.iphone;
        var $iphone = $('#download_iphone');
        $iphone.find('.name').text(iphone.version + ' for iPhone');
        $iphone.find('a').attr('href',iphone.download_url);

   });
});
